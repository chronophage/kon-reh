\section{Creature Stat Blocks}

\emph{For use with the Fate's Edge system. These statistics are drawn from Saikou's case files.}

\subsection{Echo (Class I Spirit)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & I \\
\textbf{Cap:} & 1 (non−combatant) \\
\textbf{Traits:} & Incorporeal, Repetitive, Trapped \\
\textbf{Attacks:} & None (cannot harm directly) \\
\textbf{Complications:} & Distraction (SB: 1 to create a sensory illusion), Delay (SB: 2 to cause a time loop) \\
\textbf{Resolution:} & Witness a truth about the death (DV 3) to banish permanently. \\
\textbf{Timers:} & Echo's Loop [4] −−− when filled, the Echo expands its range. \\
\end{tabular}

\subsection{Anchor (Class II Spirit)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & II−−III \\
\textbf{Cap:} & 2−−4 \\
\textbf{Traits:} & Incorporeal, Intelligent, Bound \\
\textbf{Attacks:} & Manifestation (DV 3, Fatigue 1), Poltergeist Shove (DV 4, Harm 1) \\
\textbf{Complications:} & Bargain Required −−− cannot be fought; must negotiate. \\
\textbf{Resolution:} & Fulfill the Anchor's unfinished business, or perform a witnessed substitution ritual (DV 4). \\
\textbf{Timers:} & Anchor's Patience [6] −−− when filled, the Anchor becomes a Poltergeist. \\
\end{tabular}

\subsection{Poltergeist (Class IV Spirit)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & III−−IV \\
\textbf{Cap:} & 3−−5 \\
\textbf{Traits:} & Incorporeal, Raging, Hungry \\
\textbf{Attacks:} & Telekinetic Throw (DV 3, Harm 2), Possession Attempt (DV 4, Spirit+Resist or Controlled) \\
\textbf{Complications:} & Aura of Fear (all actions Desperate within Near range), Environmental Collapse (timers advance) \\
\textbf{Resolution:} & Containment ritual (The Sentinel) followed by negotiation with the underlying Anchor. \\
\textbf{Timers:} & Poltergeist Escalation [8] −−− when filled, the spirit reaches Stage Four. \\
\end{tabular}

\subsection{Ghoul (Starved Anchor)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & II−−III \\
\textbf{Cap:} & 2−−4 \\
\textbf{Traits:} & Corporeal, Hungry, Intelligent \\
\textbf{Attacks:} & Claw (DV 2, Harm 1 + Fatigue 1), Bite (DV 3, Harm 2 + poison) \\
\textbf{Complications:} & Regeneration (ignore first 2 Harm each scene), Ravenous (must feed or frenzy) \\
\textbf{Resolution:} & Silver weapon to the heart, or fire. Negotiation possible if the ghoul is fed. \\
\textbf{Timers:} & Hunger [6] −−− when filled, the ghoul attacks indiscriminately. \\
\end{tabular}

\subsection{Lycanthrope (Linn Sea−Wolf)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & III \\
\textbf{Cap:} & 4 \\
\textbf{Traits:} & Shapechanger, Amphibious, Solitary \\
\textbf{Attacks:} & Claw (DV 2, Harm 2), Drowning Grasp (DV 4, Harm 1 + Suffocation) \\
\textbf{Complications:} & Silver Vulnerability (+2 Harm from silver), Transformation Trigger (first winter storm after a kill) \\
\textbf{Resolution:} & Silver weapon to the heart or brain. Aconite poultice can prevent transformation if applied within the hour. \\
\textbf{Timers:} & Moon's Hunger [6] −−− when filled, the lycanthrope enters a frenzy. \\
\end{tabular}

\subsection{Vampire (Lesser Bloodline)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & III \\
\textbf{Cap:} & 4 \\
\textbf{Traits:} & Undead, Hypnotic Gaze, Blood Drain \\
\textbf{Attacks:} & Claw (DV 2, Harm 1), Blood Drain (DV 3, Harm 1 + 1 Fatigue to victim, heals vampire 1 Harm) \\
\textbf{Complications:} & Sunlight Vulnerability (1 Harm per round in direct sun), Invitation Required (cannot enter a home without invitation) \\
\textbf{Resolution:} & Stake through the heart (DV 5, must be wooden), decapitation, fire. \\
\textbf{Timers:} & Thirst [6] −−− when filled, the vampire loses control and feeds openly. \\
\end{tabular}

\subsection{Demon (Class C Possessor)}
\begin{tabular}{@{}ll}
\textbf{Tier:} & IV \\
\textbf{Cap:} & 5 \\
\textbf{Traits:} & Incorporeal (unless possessing), Intelligent, Malicious \\
\textbf{Attacks (possessing):} & Use host's physical stats +2, plus fear aura (DV 3, Fear effect) \\
\textbf{Complications:} & Cannot be harmed directly while possessing; exorcism required. \\
\textbf{Resolution:} & The Sentinel's exorcism ritual (DV 5, requires three witnesses and the demon's true name). \\
\textbf{Timers:} & Possession Depth [8] −−− when filled, the host is permanently erased. \\
\end{tabular}

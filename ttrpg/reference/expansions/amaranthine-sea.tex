\section{The Amaranthine Sea and Titan’s Throat}
The \textbf{Amaranthine Sea} is a semi-enclosed inland ocean of busy trade lanes, sudden fogs, and spirit-haunted shoals. Its primary western mouth is throttled by a labyrinth of islands, reefs, and boom-chains called \textbf{Titan’s Throat}: a natural choke that the Kahfagian Admiralty has engineered into an artificial bottle.

\medskip
\noindent\textbf{Key Theaters (node map):}
\begin{itemize}
  \item \textbf{Western Reach} --- long swell, fast water; Ykrul longboats probe here.
  \item \textbf{Reef Cuts} --- Vilikari eel-channels; only low-draft pilots dare them.
  \item \textbf{Amber Straits} --- spirit buoys, pilot tests, and rangefinders.
  \item \textbf{Bottle Chain} --- lantern posts and boom-gates maintained by Kahfagia.
  \item \textbf{Saffron Lane} --- Oshiiran convoy corridor with timed windows and lamp codes.
  \item \textbf{Harbor Underbelly} --- dockside warrens; Linns stevedores control lists/rumor.
  \item \textbf{Dolmis Narrows} --- treacherous eastern strait: city-league pilots, tolls, courts.
  \item \textbf{Eastern Reach (Chain of Reeds)} --- Ashaan delta approaches; reed-winds and hidden inlets.
  \item \textbf{Free Havens} (e.g., \textit{Sable Key}, \textit{Indigo Shoals}) --- neutral anchorages and gray courts.
\end{itemize}

\subsection*{Titan’s Throat at a Glance}
\begin{itemize}
  \item \textbf{Boom-Anchors:} chained logs, iron links, and embedded purity bells.
  \item \textbf{Lantern Pylons:} weekly phrases; failure ticks \textit{Papers Scrutiny}.
  \item \textbf{Spirit Buoys:} weigh “soul-scent” for contraband, blood-debt, or bondage.
  \item \textbf{Pilot Rights:} only registered pilots may shift Position safely in the channels.
\end{itemize}

\section{Naval Movement and Positioning}
% ---------------------------------------------------------

\subsection*{Two Scales of Play}
\begin{itemize}
  \item \textbf{Theater Scale (Node Map).} Move between zones (Western Reach, Reef Cuts, Amber Straits, etc.). Each move is a \emph{leg} resolved by one \textbf{Pilot/Navigator} test; hazards may tick clocks.
  \item \textbf{Tactical Scale (Range Bands).} When exact distance matters for guns, rites, or boarding, switch to \textbf{Range Bands}.
\end{itemize}

\subsection*{Range Bands at Sea}
\begin{tabularx}{\linewidth}{@{}l X@{}}
\toprule
\textbf{Band} & \textbf{Description and Implications}\\
\midrule
\textbf{Close} & Boarding distance; grapples, cutlass work, harpoons, powder \& fire pots. Boarding occurs at Close and becomes its own scene.\\
\textbf{Near} & Same vessel cluster; shouting distance, coordinated maneuvers, ballista/harpoon range.\\
\textbf{Far} & Visual contact; signal flags/lantern codes, long-range ballista or storm-calling.\\
\textbf{Absent} & Beyond horizon; requires a \textbf{Travel Clock [4]} to close, or kite/lantern relay, spirit messenger, or courier.\\
\bottomrule
\end{tabularx}

\subsection*{Naval Movement Actions (Tactical)}
\begin{itemize}
  \item \textbf{Sail.} Shift one Range Band on success; \emph{Dash} attempts a second shift at –1 Position if it fails.
  \item \textbf{Maneuver.} Complex positioning (flank, cut-off, evade) uses \textbf{Pilot + Navigation} vs DV 2–4; success may also grant +1 Effect or impose –1 Position on a foe.
  \item \textbf{Drift.} Controlled drift masks wake/noise using \textbf{Rigger + Survival}; on failure a nearby picket hears you or \textit{Patrol Net} +1.
  \item \textbf{Anchor / Drop Sail.} Secure the ship under pressure with a \textbf{Crew Muster} test (Presence + Command); failure scatters crew or marks \textit{Rigging Damage}.
\end{itemize}

\subsection*{Theater Movement (per Leg)}
\begin{enumerate}[leftmargin=*,itemsep=2pt]
  \item \textbf{State Speed Profile:} \emph{Stealth Drift} (–1 leg; +Position), \emph{Standard Run} (normal), \emph{Press the Sails} (+1 leg; –Position on a failure).
  \item \textbf{Pilot Test:} \textbf{Wits + Navigation} (+1 die for relevant leverage: pilot-token, eel-chart, convoy slot).
  \item \textbf{Set Start Position by Successes:} 0–1 \emph{Desperate}, 2–3 \emph{Risky}, 4+ \emph{Controlled}. (A 10 counts as 2 successes.)
  \item \textbf{Hazards:} If a leg lists hazards (boom, fog, reef), resolve or tick a clock (GM).
\end{enumerate}

\subsection*{Chases on the Node Map}
Track \textbf{Quarry Lead [6]} and \textbf{Pursuer Closure [6]}; opposed \textbf{Pilot} tests at each leg tick the winner’s clock +1. \textbf{Catch} at 6 ⇒ shift to Tactical at \emph{Near}. \textbf{Escape} at 6 ⇒ chase ends.

\subsection*{Position at Sea}
\begin{itemize}
  \item \textbf{Controlled} — clear weather, pilots, known water; errors usually tick paperwork, not blood.
  \item \textbf{Risky} — contested water, moderate weather, eyes upon you; errors cost time or minor harm.
  \item \textbf{Desperate} — storms, shoals, superior force; the acting ship re-rolls one success on that action.
\end{itemize}

\subsection*{Common Maneuvers (apply TAGS to justify bonuses)}
\begin{tabularx}{\linewidth}{@{}p{3.3cm} X@{}}
\toprule
\textbf{Maneuver} & \textbf{Mechanical Effect}\\
\midrule
\textbf{Beat to Windward} & Improve Position +1 this round; fail ⇒ lose one band \emph{or} mark \textit{Rigging Damage [1]}.\\

\section{Ship Combat}
% ---------------------------------------------------------

\subsection*{Conflict Modes}
\begin{itemize}
  \item \textbf{Side-Based Turns (Recommended).} Each side chooses up to \emph{two} distinct ship actions per round (different roles): \textbf{Helm} (change Range/maneuver), \textbf{Rig} (repairs, grapples, smoke), \textbf{Guns} (ballista/harpoon/chain-shot), \textbf{Board} (grapple/cross/secure), \textbf{Ritual} (weather/ward/veil).
  \item \textbf{Initiative Variant.} Roll \textbf{Wits + Command}; highest acts first. Each PC/NPC ship role takes one action/round.
\end{itemize}

\subsection*{Combat Procedure (both modes)}
\begin{enumerate}[leftmargin=*,itemsep=2pt]
  \item \textbf{Declare Engagement.} GM sets Position from fiction (weather, surprise, Range).
  \item \textbf{Assign Roles.} Pilot, Rigger/Engineer, Gunner, Boarder/Cutter, Whisper/Ritualist, Signal/Face.
  \item \textbf{Take Actions.} Resolve rolls; apply Range/Position; spend/earn SB.
  \item \textbf{Resolve Consequences.} Mark damage clocks, shift Position, tick scene clocks.
\end{enumerate}

\subsection*{Gunnery \& Naval Weapons}
\begin{tabularx}{\linewidth}{@{}p{3.1cm} p{2.1cm} X@{}}
\toprule
\textbf{Weapon} & \textbf{Range} & \textbf{Tags / Notes}\\
\midrule
Light Ballista & Near & \tagpill{ACCURATE} \tagpill{CREW 2} \tagpill{RELOAD} — tends to mark \textit{Rigging}.\\
Heavy Ballista & Far/Near & \tagpill{PIERCING} \tagpill{SLOW} \tagpill{CREW 3} — marks \textit{Hull}; crit may also mark \textit{Crew Morale}.\\

\noindent \textbf{Resolve Gunnery.} Roll \textbf{Body or Wits + Gunnery/Craft}. \emph{Position} from Range and fiction; \emph{Effect} from TAGS and ammo. \tagpill{ARMORED} reduces \textit{Hull} marks by 1 (min 0) per hit; \tagpill{SPIRIT-WARD} negates the first ritual/fire mark once/scene.

\subsection*{Boarding Sequence (three beats)}
\begin{enumerate}[leftmargin=*,itemsep=2pt]
  \item \textbf{Grapple} (Rig/Gunner): \textbf{Craft or Gunnery} at \emph{Close}. Success ⇒ proceed.
  \item \textbf{Cross} (Cutter): \textbf{Athletics or Melee}; \tagpill{BOARDING NETS} +1 die; fail ⇒ mark \textit{Crew Morale} or \textit{Rigging}.
  \item \textbf{Secure} (All): clear deck leaders, cut lines if needed; success shifts to personal combat or surrender test.
\end{enumerate}

\subsection*{Signals, Ramming, and Command}
\begin{itemize}
  \item \textbf{Signal.} \textbf{Presence + Sway/Etiquette} to coordinate allies or demoralize foes; on success, grant +1 die or impose –1 die for a round.
  \item \textbf{Ram (Desperate).} \textbf{Body + Athletics/Pilot} vs target \textbf{Hull}; on success mark both ships’ \textit{Hull}; great Effect may also \textit{Grapple}.
  \item \textbf{Command Presence.} \textbf{Presence + Command} to steady crew, cancel a panic, or seize initiative for one role.
\end{itemize}

\subsection*{Ship Condition: Quick or Detailed}
\textbf{Quick Play:} \textbf{Ship Integrity [6]} — when full, the ship is \emph{Unseaworthy}; further marks risk sinking or forced beaching.

\medskip
\textbf{Detailed Tracks (replace Integrity):}
\begin{itemize}
  \item \textbf{Hull Damage [4]} — breaches/leaks; on fill: \emph{Sinking}.
  \item \textbf{Rigging Damage [4]} — on fill: no Range changes.
  \item \textbf{Crew Morale [4]} — on fill: panic/surrender test.
  \item \textbf{Fire/Explosion [4]} — on fill: catastrophic loss unless contained.
\end{itemize}
\noindent \textbf{Hull Integrity Test:} \textbf{Body + Survival} to resist catastrophic damage. \textbf{Crew Loyalty Test:} \textbf{Presence + Command} to maintain order.

\subsection*{Magic at Sea}
Magic is potent but volatile on water:
\begin{itemize}
  \item \textbf{Unstable Platform.} Casting at \emph{Risky}: –1 die; at \emph{Desperate}: –2 dice unless anchored/lashed.
  \item \textbf{Backlash Amplification.} Rites that misfire tick \textit{Ship Integrity} (or \textit{Fire}/\textit{Morale}) in addition to personal harm.
  \item \textbf{Patron Interference.} Naval battles increase omen/demand frequency at GM discretion.
  \item \textbf{Ritual Constraints.} Major rites require a stabilized anchor point and mark Obligation.
\end{itemize}

\section{Inspection and Paperwork Systems}

\subsection*{Checkpoint Scene Procedure}
\begin{enumerate}[leftmargin=*,itemsep=2pt]
  \item \textbf{Declare Cover.} What papers/story are you presenting?
  \item \textbf{Set Position.} Etiquette, bribes, or favors can improve Position.
  \item \textbf{Primary Test.} \textbf{Presence + Etiquette/Deception} \emph{or} \textbf{Wits + Lore/Craft} for paperwork.
  \item \textbf{Secondary Checks.} Keel drag, muster, sigil scan; each failure ticks \textit{Papers Scrutiny}.
  \item \textbf{Outcomes.} \emph{Controlled}: pass \& reduce \textit{Blockade Awareness}. \emph{Risky}: pass with a condition (escort, stamp, toll). \emph{Desperate}: board/search, duel, or seizure court.
\end{enumerate}

\section{Faction Standing and Reputation}
% ---------------------------------------------------------

\subsection*{Faction Standing Clocks [6]}
Each major faction keeps a Standing clock with your crew.
\begin{itemize}
  \item \textbf{–6 to –5 (Hostile).} Bounties, blacklists, kill-on-sight or seizure.
  \item \textbf{–4 to –3 (Unfavorable).} More inspections, higher tolls, patrol “attention.”
  \item \textbf{–2 to 2 (Neutral).} Standard treatment, normal prices, limited help.
  \item \textbf{3 to 4 (Favorable).} Easier inspections, better prices, occasional favors.
  \item \textbf{5 to 6 (Favored).} Convoy slots, legal shelter, special services (pilot tokens, codes).
\end{itemize}

\subsection*{Changing Standing}
\begin{itemize}
  \item \textbf{Favors/Jobs.} +1–2 for completing marked work; –1–2 for insults or unpaid debts.

\section{Conducting Magic Aboard}

\subsection*{The Ship as Circle}
A seaworthy hull is a moving ritual diagram. Treat \textbf{Mast}, \textbf{Bells}, \textbf{Lanterns}, \textbf{Keel}, and \textbf{Figurehead} as \emph{anchor points}. A rite needs three of five to be \emph{stable}. Fewer anchors: +1 DV and counts as \emph{Risky}. No anchors: \emph{Desperate} and primed for backlash.

\subsection*{Casting Windows \& Position}
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item \textbf{Controlled.} Steady way on, trimmed canvas, crew at stations. Standard DV.
  \item \textbf{Risky.} Maneuvering hard, cross chop, or contested water. –1 die or +1 DV (choose).
  \item \textbf{Desperate.} Storm-on-bow, reefs, boarding melee. –2 dice \emph{or} +2 DV; on backlash, extend to crew (tick Ship Integrity or Morale).
\end{itemize}

\subsection*{The Barometer Track \([6]\)}
Place a visible track between \textbf{Breath (Raéyn)} on the left and \textbf{Pressure (Khemesh)} on the right. At scene start, mark 1–2 toward whichever force dominates (weather, depth, fiction). \emph{Effects:}
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Breath 1–2: Wind rites –1 DV; constricting effects +1 DV.
  \item Breath 3–4: Free \emph{Wind Call} once; chain/grapple actions are \emph{Risky}.
  \item Pressure 1–2: Binding, ballast, and holdfast rites –1 DV; sailing Maneuvers +1 DV.
  \item Pressure 3–4: Free \emph{Holdfast} once; capsize checks +1 tier of consequence.
\end{itemize}
Shift the marker when omens land, when Cantors sing properly (toward Breath) or when debts are enforced (toward Pressure).

\section{Watches, Fatigue, and Duty Cycle}
% ---------------------------------------------------------

\subsection*{Watches}
Ships run two or three watches. Mark \textbf{Watch Fatigue [4]} when you: sail through a full night at \emph{Risky/Desperate}, fight a battle, or work storm drills.
\begin{itemize}
  \item At 2: –1 die to \emph{Rig} or \emph{Guns} until rest.
  \item At 4: first \textit{Crew Morale} tick each scene is free (auto-tick) until anchored rest.
\end{itemize}
\noindent \textbf{All-Hands!} Clear \emph{one} Watch Fatigue tick at end of crisis if food \& hot drink are served (Purser test).

\subsection*{Berths \& Overmanning}
If crew aboard exceeds \textbf{Rec} by 25\%+, mark \textbf{Cramped Berths [4]}. At 2: +1 \textit{Crew Morale} tick on any failure; at 4: disease check (Surgeon test) or Harm 1 to random hands.

\section{Leadership Challenges and Mutiny}
% ---------------------------------------------------------

\subsection*{The Grievance Engine}
Track three clocks that point at trouble:
\begin{itemize}
  \item \textbf{Late Shares [4]} — unpaid prize, dockside “adjustments,” missing wages.
  \item \textbf{Hard Usage [4]} — cruel orders, needless risks, taboos broken.
  \item \textbf{Bad Stores [4]} — foul water, weevils, short rations.
\end{itemize}
\noindent When any two fill, start the \textbf{Mutiny Clock [6]}. While it runs, \textbf{Command} actions are –1 die unless supported by Purser/Surgeon/Chaplain.

\subsection*{Crisis Procedures}
\paragraph{Address the Crew.} \textbf{Presence + Command}; success lowers one Grievance by 1 and pauses \textit{Mutiny} this scene. Bring proof, prize shares, or apologies for +1 die.
\paragraph{Distribute Shares.} Spend 1–2 Cred to clear \textit{Late Shares} entirely; also clear 1 \textit{Crew Morale} mark.
\paragraph{Call the Hands (Duel or Vote).} Courtesy duel with the spokesman (Duel/Etiquette/Melee). Win: \textit{Mutiny –2}. Lose: \textit{Mutiny +2} and choose—step down, make restitution, or fight on deck (Desperate).

\subsection*{Mutiny Resolution}
When \textbf{Mutiny [6]} fills, resolve as a scene: PCs may parley, duel, bribe, or split the company. On a clean success, choose two:
\begin{itemize}
  \item Clear one Grievance entirely.
  \item Promote a loyal petty officer (create a \emph{Face} or \emph{Bosun} with Seasoned CQ).
  \item Keep the ship intact; otherwise \textit{Rigging} +1 and \textit{Crew Morale} resets to 2 ticks.
\end{itemize}

\textbf{Collapse:} \emph{Obligation} hits 6 and a Doctrine is broken on screen: Patron withdraws; convert to \textbf{Break the Titan} (secular reform) or \textbf{Pirate Kings} (fallen zeal).

\subsection*{1. The Scent of Fear}
\textbf{Hook.} Contract-buoys report \emph{unlawful bondage} aboard several “clean” hulls queued for Oshiiran transit.\\
\textbf{Objective.} Trace the false positives without tipping off the saboteurs.

\textbf{Key Faces.} Scribe-Pilot \emph{Orel Vass} (by-the-lantern), Free Haven broker \emph{Sola Nemet}, covert Kahfagian runner \emph{The Quiet Bosun}.\\
\textbf{Procedure \& Clocks.} \emph{Audit Chain [6]} (dock, manifest, buoy), \emph{Saboteur Window [4]} (they move once filled). Investigate with Craft/Lore; stealth with Heist Inside team.\\
\textbf{Complications (d4).} 1) The buoy key was swapped in court; 2) A rival crew frames the PCs; 3) A witness vanishes at sea; 4) The “victims” are embedded agents pinging the net.\\
\textbf{Rewards.} –1 \emph{Papers Scrutiny} for one lantern cycle; Standing +1 with Free Havens.\\
\textbf{Scaling.} Tier I: one pier. Tier II–III: three synchronized docks across two ports.

\subsection*{2. Pilot’s Gambit}
\textbf{Hook.} Dolmis pilots strike; harbor idle.\\
\textbf{Objective.} Move a priority hull through the Narrows without official pilotage.

\section*{Scene Construction (Sea)}
% ------------------------------
\begin{enumerate}[leftmargin=*,itemsep=2pt]
  \item \textbf{One Clear Aim} (escape, seize, pass, rescue).
  \item \textbf{One Map Element} (reef, boom, fog, tower arc).
  \item \textbf{One Pressure Clock} (Patrol Net, Weather Window, Morale).
  \item \textbf{One Signature Move} for the foe (signal trick, chain-shot, spirit net).
  \item \textbf{One Reward} that alters procedure (safe cut, face, rite).
\end{enumerate}

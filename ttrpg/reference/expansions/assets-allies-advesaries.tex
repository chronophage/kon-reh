\section*{Assets, Allies, and Adversaries}
A Fate's Edge Module for Holdings, Companions, and Organizations

\subsection*{Design Goals}
\begin{itemize}[leftmargin=*,noitemsep]
\item \textbf{Make Holdings Matter:} Assets have tags, strings, upkeep, and project clocks that change scenes. Patronage brings Position, seals, and coin—and obligations, audits, and scandals.
\item \textbf{Companions with Teeth:} Followers act on-screen and off-screen with clear risks, clocks, and costs. Social fabric (Loyalty, Morale, Bonds) generates scenes and consequences.
\item \textbf{Organizations as Levers:} Cohorts and institutions move through clocks, strings, and orders without excessive bookkeeping.
\item \textbf{Use the Core:} Leverages core Fate's Edge mechanics: Position/\dv, \sbtext, Clocks, Favor/Leverage/Exposure, Reputation, Strings. No new dice math.
\item \textbf{Campaign Continuity:} Assets and relationships persist and evolve across story arcs.
\end{itemize}

\subsection*{Quickstart (5 minutes)}
\begin{enumerate}[leftmargin=*,noitemsep]
\item \textbf{Pick an \asset:} Choose a type (safehouse, workshop, etc.). Mark Integrity \clock{6} and Resources \clock{6}. Attach 1-2 Strings (permits, seals). Choose 1-2 Project Clocks \clock{4}-\clock{8}.
\item \textbf{Name a \patron (optional):} Set Patron \tier and Obligations.
\item \textbf{Create 1-2 \followers:} Name them, pick a Playbook (Combat Ally, Specialist, etc.). Mark Loyalty \clock{6} and Morale \clock{6}. Attach Strings they control.
\item \textbf{Pick an \organisation (if any):} Mark Cohesion \clock{6}-\clock{8}, Exposure \clock{6}, and Bandwidth (1-3 orders/cycle).
\item \textbf{Each Session:} Issue \asset Orders, run \follower roles on-scene, confront Events \& Audits, and tick Upkeep.
\end{enumerate}

\section{Assets and Worldly Patrons}

\subsection{Asset Sheet (Template)}

\textbf{[\asset NAME]} \\
\textbf{Type:} safehouse / barge company / workshop / archive / shrine / etc. \\
\textbf{Locale:} district \& city \\
\textbf{Strings (1--3):} permits, seals, routes, rites, keys \\
\textbf{Tags (2--4):} (See Tag List below) \\
\textbf{Tracks:}
\begin{itemize}[noitemsep]
\item \textbf{Integrity \clock{6}:} condition/standing; fill $\rightarrow$ shutdown, collapse, or seizure.
\item \textbf{Resources \clock{6}:} cashflow, inventory, staff slack; empty $\rightarrow$ shortages/concessions.
\item \textbf{Heat \clock{6} (optional):} locals' attention; fill $\rightarrow$ protest, inquiry, rough visit.
\end{itemize}
\textbf{Projects (0--3):} named \clock{4}-\clock{8} clocks to upgrade, expand, or pivot. \\
\textbf{Upkeep:} coin/favor/rites required each cycle. \\
\textbf{Legacy Notes:} campaign events, reputation effects. \\
\textbf{Notes:} staff names, neighbors, rivals.

\subsubsection{Asset Tags (pick 2--4)}
\begin{itemize}[noitemsep]
\item \tag{Fortified}: Shield/Petition here starts Dominant.
\item \tag{Sanctified}: Breaking rites here ticks Exposure +1 (offender).
\item \tag{Licensed}: Broker/Petition \dv -1 in scope of license.
\item \tag{Secret}: First Expose against you here starts Desperate for the attacker.
\item \tag{Crowd-Facing}: Audience tags created here persist an extra scene.
\item \tag{Hazardous}: On 1s, GM may start Accident \clock{4} instead of other \sbtext spend.
\item \tag{Mobile} (barge/caravan): Can act in adjacent districts without penalty.
\item \tag{Prestige}: Petition here \dv -1 with patricians; Blackmail +1 \sbtext against you if scandal hits.
\item \tag{Ward-Woven}: Infiltrate here starts Desperate unless key is held.
\item \tag{Unionized}: Strike is a valid Event; Broker with unions here starts Dominant.
\item \tag{Shadowed}: Smuggle \dv -1; Expose Position -1 against you.
\item \tag{Water-Right}: Acts as a String for bridges/canals toll negotiation.
\item \tag{Archive}: Research/Expose \dv -1 with proof assembled here.
\end{itemize}

\subsection{Asset Play}
\textbf{Acting Through an \asset:} Stage a scene at or with an asset; apply its Tags and Strings to Position/\dv. A named staffer may act as a \follower. \\
\textbf{Upkeep \& Yield (Each Cycle):}
\begin{itemize}[noitemsep]
\item \textbf{Upkeep:} Pay 1-2 of coin/favor/rite. If skipped: tick Resources -1 and start Creditor \clock{4} or Inspection \clock{4}.
\item \textbf{Yield:} If Resources $\geq$ 3 and no active Accident/Inspection, gain one: coin, Favor, Clue, or Leverage (1).
\end{itemize}
\textbf{\asset Orders (Between Sessions):} Choose up to 2 orders per asset per session (1 if Resources $\leq$ 2).
\begin{itemize}[noitemsep]
\item \textbf{Operate:} Generate Yield with risk (on 1s, tick Heat or Integrity +1).
\item \textbf{Improve:} Advance a Project Clock.
\item \textbf{Secure:} Reduce Heat -1 or add a Security tag for one scene.
\item \textbf{Expand:} Start a new Project to add a String.
\item \textbf{Audit:} Convert Favor into Standing (Tag).
\item \textbf{Exploit:} Trade Integrity for coin+Favor; mark Exposure +1.
\end{itemize}

\subsection{Worldly Patrons}
\subsubsection{Patron Sheet (Template)}
\textbf{[\patron NAME]} \\
\textbf{Type:} noble / guild / office / temple / factor \\
\textbf{Stance:} Allied / Wary / Hostile \\
\textbf{Strings (3):} writs, permits, routes, rites \\
\textbf{Boons:} what they can grant (seals, escorts, stipends) \\
\textbf{Obligations:} tithes, appearances, tasks, ideological lines \\
\textbf{Tracks:}
\begin{itemize}[noitemsep]
\item \textbf{Favor Ledger:} favors owed or granted.
\item \textbf{Sanction \clock{4}:} warning $\rightarrow$ censure $\rightarrow$ seizure $\rightarrow$ hunt.
\item \textbf{Patron Exposure \clock{6}:} public risk to them from association with you.
\end{itemize}
\textbf{Patron \tier:} 0 Contact / 1 Sponsor / 2 Patron / 3 Protector \\
\textbf{Legacy Status:} Historical role, ongoing influence.

\subsubsection{Patron Tiers \& Benefits}
\begin{itemize}[noitemsep]
\item \textbf{\tier 0 — Contact:} 1 Audience/session; one minor seal.
\item \textbf{\tier 1 — Sponsor:} +1 Position once/scene in their venues; 1 Stipend/session.
\item \textbf{\tier 2 — Patron:} Cancel one Social \sbtext per session; \dv -1 on Petition to their offices.
\item \textbf{\tier 3 — Protector:} Start public scenes Dominant; Endorsement creates a Bandwagon clock \clock{4}.
\item \textbf{Advance a \tier:} Hold Standing or convert 3 favors and complete a Patron Task.
\end{itemize}

\subsubsection{Obligations \& Sanctions}
\begin{itemize}[noitemsep]
\item Skipping Obligations for two cycles: tick Sanction +1.
\item \textbf{Sanction \clock{4}:} Admonish $\rightarrow$ Censure $\rightarrow$ Seizure/Revocation $\rightarrow$ Hunt/Prosecution.
\item Your public failures tick Patron Exposure; at fill, they distance (\tier -1) or flip to Hostile.
\end{itemize}

\subsection{Campaign Continuity}
\textbf{\asset Legacy System:} Assets evolve. Positive legacy grants +1 Resources/Integrity at campaign start. Negative legacy starts with Heat +2 or Sanction +1. \\
\textbf{\patron Relationship Continuity:} Relationships persist. Active Patrons reduce \tier by 1 but retain Standing. Dormant Patrons convert to Favor Ledger entries. \\
\textbf{Campaign Transition Events:} At campaign end/beginning, roll for economic, political, and physical shifts affecting assets and patrons.

\section{Allies and Adversaries}

\subsection{Follower Sheets \& Templates}
\subsubsection{Follower Sheet (Template)}
\textbf{[NAME \& ROLE]} \\
\textbf{\tier:} Green / Trained / Veteran / Elite (modifies Position/\dv) \\
\textbf{Playbook:} Combat Ally / Specialist / Magical Assistant \\
\textbf{Capabilities (2-3):} e.g., ``shield wall'', ``ledger-eye'', ``silent step'' \\
\textbf{Edges (1-2):} What they do exceptionally well. \\
\textbf{Liabilities (1-2):} What gets them in trouble. \\
\textbf{Strings (1-3):} permits, keys, access. \\
\textbf{Costs:} wages / shares / favor / oath \\
\textbf{Tracks:}
\begin{itemize}[noitemsep]
\item \textbf{Loyalty \clock{6}:} long-term commitment.
\item \textbf{Morale \clock{6}:} short-term willingness to risk.
\item \textbf{Harm \clock{3}:} minor / serious / broken.
\end{itemize}
\textbf{Notes:} relationships, ambitions.

\subsection{Followers On-Scene}
At the start of a scene, declare a role for each present \follower:
\begin{itemize}[noitemsep]
\item \textbf{Assist:} Bolster a PC's action. Effect: Improve Position +1 or treat \dv -1 (once/scene). On any 1 in the roll, tick \follower Morale or Harm.
\item \textbf{Cover:} Absorb heat. Effect: Once/scene, reduce a GM Social \sbtext spend by 1 or cancel a tag flip. Cost: Tick \follower Exposure or Morale +1.
\item \textbf{Delegate:} Take an independent action (legwork, duel, etc.). Resolve with a PC's coordinating roll. On a hit, create a useful tag or fill a clock; on a partial, suffer Harm or Morale +1.
\end{itemize}

\subsection{Loyalty \& Morale}
\begin{itemize}[noitemsep]
\item \textbf{Loyalty \clock{6}:} Ticks up when promises kept, credit shared, ambitions advanced. Ticks down for oath-breaking, humiliation. At 0: leave or flip neutral. At 6: trigger a Loyalty Event (offer an oath, ask a Price).
\item \textbf{Morale \clock{6}:} 4+: bold. 2--3: hesitant; Delegate starts worse. 0--1: balk; require a Reassure roll to act. Reset with rest, praise, respect.
\end{itemize}

\subsection{Organizations}
\subsubsection{Organization Sheet (Template)}
\textbf{[\organisation NAME]} \\
\textbf{\tier:} street / guild / city / crown \\
\textbf{Aim (this season):} what the org wants now. \\
\textbf{Cohorts (2-4):} ``dockside crew'', ``auditors'', ``pickets''. \\
\textbf{Strings (3):} permits, routes, seals. \\
\textbf{Tracks:}
\begin{itemize}[noitemsep]
\item \textbf{Cohesion \clock{6}-\clock{8}:} unity/discipline; fill $\rightarrow$ schism/strike/coup.
\item \textbf{Exposure \clock{6}:} heat/visibility; fill $\rightarrow$ audit/raid/purge.
\item \textbf{Resources \clock{6}:} operational slack; empty $\rightarrow$ shortages.
\end{itemize}
\textbf{Bandwidth:} 1-3 orders per cycle. \\
\textbf{Entanglements:} debts, rivals, promises.

\subsubsection{Using Cohorts}
A cohort acts like a Veteran/Elite \follower group. Wounds/fear tick Cohesion instead of individual Harm.

\subsubsection{Orders \& Entanglements (Between Sessions)}
\begin{enumerate}[noitemsep]
\item \textbf{Allocate Bandwidth:} Choose orders (Audit, Guard, Smuggle, Petition, etc.).
\item \textbf{Resolve Each Order:} Directing PC rolls. Apply Org \tier as \dv -1 within its wheelhouse.
\item \textbf{On Hit:} Fill a Project Clock or create a lasting Tag. On 1s, tick Exposure or Cohesion +1.
\item \textbf{Entanglement Roll:} For each banked \sbtext, trigger a minor entanglement (rival, inspector visit).
\end{enumerate}

\subsection{Relationship Dynamics}
\begin{itemize}[noitemsep]
\item \textbf{Bonds \& Frictions:} Start a Bond \clock{4} or Friction \clock{4} between a PC and \follower. Shared scenes and actions fill or clear the clock, leading to permanent tags (Devoted, Soured).
\item \textbf{Reputation \& Standing:} \followers can hold Standing with institutions, granting once/scene Position +1 or \dv -1 in that venue.
\end{itemize}


\section{Integrated Procedures}

\subsection{Events, Audits, \& Entanglements}
Roll or draw 1-2 between sessions per active asset/patronage/followers.
\begin{itemize}[noitemsep]
\item \textbf{Inspection:} Start Inspection \clock{4}; on fill, tick Integrity +1 or Sanction +1.
\item \textbf{Accident:} Accident \clock{4} threatens staff; on fill, Integrity +1 and Audience: Fearful.
\item \textbf{Shortages:} Resources -1 and start Short Rations \clock{4}.
\item \textbf{Favor Called:} A \patron demands service; refuse $\rightarrow$ Sanction +1.
\item \textbf{Rival Poaches a Cohort:} An \organisation loses a cohort to a rival.
\item \textbf{Family Claim Interrupts:} A \follower's personal life causes a complication.
\end{itemize}

\subsection{Example of Play}
\textbf{Setup:} PCs lease a \textbf{Tollhouse} (\asset: Licensed, Prestige). Their \textbf{Specialist \follower} (ledger-eye) has ties to the \textbf{Bridge-Lords} (\patron, \tier 1). \\
\textbf{Scene — Delegation:} The Specialist is sent to secure a ledger from the Archivolt. Position: Dominant (Prestige tag + \patron venue). \dv 3. Partial Success: ledger secured but Morale +1. A rolled 1 gives GM an \sbtext to start a Rumor \clock{4}. \\
\textbf{Orders Phase:} PC directs the Tollhouse \asset to \textbf{Operate} (yield coin) and \textbf{Improve} an Archive Annex project. The party's Guild \organisation runs an \textbf{Audit} order. \\
\textbf{Event Phase:} An \textbf{Inspection \clock{4}} event occurs. The PCs use a \textbf{Secure} order next cycle to mitigate it. \\
\textbf{Dilemma:} A rival \patron offers better \tier, starting a \textbf{Split Loyalty \clock{4}} clock. This tests the \follower's Loyalty and the \asset's stability.

\subsection*{GM Reference}
\begin{itemize}[noitemsep]
\item \textbf{\asset Tracks:} Integrity \clock{6} (fill = shutdown), Resources \clock{6} (0 = shortages), Heat \clock{6} (fill = audit).
\item \textbf{\follower Tracks:} Loyalty \clock{6} (0 = leave, 6 = event), Morale \clock{6} (0-1 = balk), Harm \clock{3}.
\item \textbf{\organisation Tracks:} Cohesion \clock{6}-\clock{8}, Exposure \clock{6}, Resources \clock{6}, Bandwidth 1-3.
\item \textbf{\patron Tiers:} 0 Contact $\rightarrow$ 3 Protector. Sanctions: Admonish $\rightarrow$ Hunt.
\item \textbf{Core Loop:} Describe action $\rightarrow$ Set Position/\dv $\rightarrow$ Roll $\rightarrow$ Spend \sbtext/\boons $\rightarrow$ Tick Clocks $\rightarrow$ Move the fiction.
\end{itemize}

    \section{Asset Play}
    \subsection{Acting Through an Asset (On-Scene)}
    Stage a scene at or with an asset; apply its Tags and Strings to Position/DV like any venue. A named staffer may act as a \emph{Follower} or as a cohort (tick \textbf{Integrity} instead of Harm on failures).
    
    \subsection{Upkeep \& Yield (Each Cycle)}
    \begin{itemize}
      \item \textbf{Upkeep:} Pay 1--2 of coin/favor/rite. If skipped: tick \textbf{Resources --1} and start \textbf{Creditor [4]} or \textbf{Inspection [4]}.
      \item \textbf{Yield:} If \textbf{Resources $\geq$ 3} and no active \textbf{Accident/Inspection}, gain one: coin, \textbf{Favor (narrow)}, \textbf{Clue}, or \textbf{Leverage (1)} themed to the asset.
    \end{itemize}
    
    \subsection{Asset Orders (Between Sessions)}
    Choose up to \textbf{2 orders} per session per asset (1 if Resources $\leq$ 2):
    \begin{itemize}
      \item \textbf{Operate:} generate Yield with risk (on 1s, tick \textbf{Heat +1} or \textbf{Integrity +1}).
      \item \textbf{Improve:} advance a \textbf{Project} [4--8].
      \item \textbf{Secure:} reduce \textbf{Heat --1} or add a \textbf{Security} tag for one scene.
      \item \textbf{Expand:} start a new Project to add a String (route, permit, office).
      \item \textbf{Audit:} convert one unspent Favor at this venue into \textbf{Standing (Tag)} after proof.
      \item \textbf{Exploit:} trade 1 \textbf{Integrity} for immediate coin + Favor; mark \textbf{Exposure +1} to someone.
    \end{itemize}
    \paragraph{Resolution} The directing PC rolls once per order (appropriate action). Tier/Tags adjust Position/DV. On hits, apply the order effect; on 1s, GM banks SB and hits Heat/Integrity/Exposure.
    
    \section{Campaign Continuity}
    \subsection{Asset Legacy System}
    Assets evolve and leave marks across campaigns through their \textbf{Legacy Notes}:
    \begin{itemize}
      \item \textbf{Reputation Effects:} Notorious (feared), Respected (trusted), Infamous (recognized), Legendary (historical significance)
      \item \textbf{Historical Events:} Siege Survived, Scandal Weathered, Expansion Completed, Crisis Averted
      \item \textbf{Relationship Changes:} Patron Gained/Lost, Rival Created/Resolved, Community Bond/Feud
      \item \textbf{Physical Evolution:} Renovated, Damaged, Expanded, Relocated
    \end{itemize}
    
    \textbf{Legacy Benefits:} Assets with positive legacy gain +1 Resources or +1 Integrity at start of new campaigns.\\
    \textbf{Legacy Burdens:} Assets with negative legacy start with Heat +2 or Sanction +1 with relevant patrons.
    
    \subsection{Patron Relationship Continuity}
    Worldly Patron relationships persist with modified standing:
    \begin{itemize}
      \item \textbf{Active Patrons:} Reduce Tier by 1 but retain 1 Standing tag
      \item \textbf{Dormant Patrons:} Convert Tier to Favor Ledger entries; can be reactivated
      \item \textbf{Former Patrons:} Become Rivals [4] or Contacts [4] based on ending relationship
      \item \textbf{Legacy Patrons:} Historical figures whose mandates still influence current politics
    \end{itemize}
    
    \subsection{Campaign Transition Events}
    At campaign end/beginning, roll for transition events:
    \begin{itemize}
      \item \textbf{Economic Shift (d6):} 1-2 Resources +1, 3-4 No change, 5-6 Resources -1
      \item \textbf{Political Change (d6):} 1-2 Gain new patron opportunity, 3-4 Status quo, 5-6 Lose a patron string
      \item \textbf{Physical Event (d6):} 1-2 Asset improves, 3-4 No change, 5-6 Asset takes Integrity -1
    \end{itemize}

      \item \textbf{Tier 3 --- Protector:} start public scenes \textbf{Dominant} in their venues; \textbf{Endorsement} creates a 4-clock \emph{Bandwagon} on targets.
    \end{itemize}
    \paragraph{Advance a Tier} Hold \textbf{Standing} with them or convert \textbf{3 favors} into Standing and complete a \textbf{Patron Task [4--6]} on-screen.
    
    \subsection{Obligations \& Sanctions}
    \begin{itemize}
      \item \textbf{Obligations:} tithe; keep scandal quiet; show at rites; take contracts; avoid rivals; uphold a doctrine. Skipping two cycles: tick \textbf{Sanction +1}.
      \item \textbf{Sanction [4]:} \emph{Admonish} (lose once/scene Position boost) $\to$ \emph{Censure} (no stipends; add \textbf{Audience: Skeptical}) $\to$ \emph{Seizure/Revocation} (lose a String or asset tag) $\to$ \emph{Hunt/Prosecution} (start \textbf{Warrant [6]}).
      \item \textbf{Patron Exposure [6]:} Your public failures can tick this; at fill they \emph{distance} (Tier --1) or \emph{flip} to Hostile.
    \end{itemize}
    
    \subsection{Bargaining Procedure (Negotiation Scene)}
    \textbf{Frame:} what boon you want and what obligations you accept.\\
    \textbf{Set:} venue tags; your Reputation/Standing apply; use \emph{Petition/Broker/Expose}.\\
    \textbf{On hit:} gain the boon; write the Obligation and tick Favor Ledger $\pm$ as appropriate.\\
    \textbf{On 1s:} GM may add a secret clause, leak leverage, or start \textbf{Rival Patron [4]}.
    
    \subsection{Multiple Patrons}
    You may keep two active patrons without penalty. A third creates \textbf{Split Loyalty [4]}; on fill, one patron issues an ultimatum: choose, public denunciation, or give up a String.
    
    \section{Events, Audits, \& Market Shocks}
    Roll or draw 1--2 between sessions per active asset/patronage.
    \begin{itemize}
      \item \textbf{Inspection:} start \textbf{Inspection [4]}; on fill, tick \textbf{Integrity +1} or \textbf{Sanction +1}.
      \item \textbf{Accident:} \textbf{Accident [4]} threatens staff; on fill, Integrity +1 and \textbf{Audience: Fearful}.
      \item \textbf{Shortages:} \textbf{Resources --1} and start \textbf{Short Rations [4]}.
      \item \textbf{Tax/Lease Hike:} choose coin cost or \textbf{Sanction +1}.
      \item \textbf{Rival Claim:} assert prior right; start \textbf{Litigation [6]} or \textbf{Duel of Proof [4]}.
      \item \textbf{Rumor Run:} if \emph{Crowd-Facing}, add \textbf{Skeptical}; if \emph{Prestige}, add \textbf{Fascinated} (cuts both ways).
      \item \textbf{Favor Called:} a patron demands service; refuse $\to$ \textbf{Sanction +1}.
    \end{itemize}
    
    \section{Blueprints (Projects \& Upgrades)}
    Pick a \textbf{Project [4--8]} to install a blueprint; on fill, add the effect and tag.
    \begin{itemize}
      \item \textbf{Safehouse (4):} add \emph{Fortified} + \emph{Secret}; gain \textbf{Leverage (1)} once/session from stashed goods.
      \item \textbf{Workshop (6):} add \emph{Hazardous}; once/session, turn coin $\to$ \textbf{Clue/Prototype} tag.
      \item \textbf{Archive Annex (6):} add \emph{Archive}; \emph{Research/Expose} DV --1 here; bank \textbf{1 Clue} each cycle if Upkeep is paid.
      \item \textbf{Shrine-Nave (6):} add \emph{Sanctified}; once/session \emph{Host Rite} starts \textbf{Dominant}.
      \item \textbf{Barge Fleet (8):} add \emph{Mobile} + \emph{Water-Right}; once/session move a scene to adjacent district at same Position.
      \item \textbf{Guard Contract (6):} add \emph{Licensed}; once/session \emph{Call the Watch} auto-succeeds at minor level.
      \item \textbf{Union Hall (6):} add \emph{Unionized}; once/session flip a \textbf{Strike} counter one step toward peace (if dues paid).
      \item \textbf{Playhouse Front (4):} add \emph{Crowd-Facing}; create Audience tags more easily; once/session convert \textbf{Audience: Warm} to \textbf{Favor (narrow)}.
    \end{itemize}

        \section*{Design Goals}
        \begin{itemize}
          \item \textbf{Make holdings matter:} Assets have tags, strings, upkeep, and project clocks that change scenes.
          \item \textbf{Patrons with teeth:} Patronage brings Position, seals, and coin---and obligations, audits, and scandals.
          \item \textbf{Use the core:} Position/DV, SB, clocks, Favor/Leverage/Exposure, Reputation, Strings. No new dice math.
          \item \textbf{Low bookkeeping:} Track names, tags, and a few clocks; orders and events move the world.
          \item \textbf{Campaign continuity:} Assets and patrons persist and evolve across story arcs.
        \end{itemize}
        
        \section*{Quickstart (2 minutes)}
        \begin{enumerate}
          \item Pick/design an \textbf{Asset}; mark \textbf{Integrity [6]} and \textbf{Resources [6]}.
          \item Name a \textbf{Patron} (optional); set \textbf{Patron Tier} and \textbf{Obligations}.
          \item Attach 1--2 \textbf{Strings} (permits, seals, routes, rites).
          \item Choose 1--2 \textbf{Project Clocks [4--8]} to improve or expand the asset.
          \item Each session: issue \textbf{Asset Orders}, confront \textbf{Events \& Audits}, and tick \textbf{Upkeep}.
        \end{enumerate}

        \subsection{Asset Tags (pick 2--4)}
        \begin{itemize}
          \item \textbf{Fortified} (doors, shutters, hidden room) --- \emph{Shield/Petition} here starts \textbf{Dominant}.
          \item \textbf{Sanctified} (rites, hospitality) --- breaking rites here ticks \textbf{Exposure +1} (offender).
          \item \textbf{Licensed} (writ, seal) --- \emph{Broker/Petition} \textbf{DV --1} in scope of license.
          \item \textbf{Secret} (concealed use) --- first \emph{Expose} against you here starts \textbf{Desperate} for the attacker.
          \item \textbf{Crowd-Facing} (shop, theater) --- \emph{Audience tags} created here persist an extra scene.
          \item \textbf{Hazardous} (kilns, reagents) --- on 1s, GM may start \textbf{Accident [4]} instead of other SB spend.
          \item \textbf{Mobile} (barge/caravan) --- can act in adjacent districts without penalty.
          \item \textbf{Prestige} (old name, art) --- \emph{Petition} here \textbf{DV --1} with patricians; \emph{Blackmail} \textbf{+1 SB} against you if scandal hits.
          \item \textbf{Ward-Woven} (sigils, bells) --- \emph{Infiltrate} here starts \textbf{Desperate} unless key is held.
          \item \textbf{Unionized} --- \emph{Strike} is a valid Event; \emph{Broker} with unions here starts \textbf{Dominant}.
          \item \textbf{Shadowed} (underworld ties) --- \emph{Smuggle} \textbf{DV --1}; \emph{Expose} \textbf{Position --1} against you.
          \item \textbf{Water-Right} --- acts as a \textbf{String} for bridges/canals toll negotiation.
          \item \textbf{Archive} --- \emph{Research/Expose} \textbf{DV --1} with proof assembled here.
        \end{itemize}
        
        \subsection{Integrity \& Resources}
        \begin{itemize}
          \item \textbf{Integrity [6]} ticks from sabotage, audits, disasters, or neglect. At fill: choose \textbf{Shutdown}, \textbf{Seizure}, or a \textbf{Catastrophe} scene.
          \item \textbf{Resources [6]} drop from upkeep, shocks, strikes; refill via Orders, scenes, or Projects. If Resources = 0, future Upkeep requires \textbf{Favor} or a concession clock.
        \end{itemize}
        
        \section{Asset Play}
        \subsection{Acting Through an Asset (On-Scene)}
        Stage a scene at or with an asset; apply its Tags and Strings to Position/DV like any venue. A named staffer may act as a \emph{Follower} or as a cohort (tick \textbf{Integrity} instead of Harm on failures).
        
        \subsection{Upkeep \& Yield (Each Cycle)}
        \begin{itemize}
          \item \textbf{Upkeep:} Pay 1--2 of coin/favor/rite. If skipped: tick \textbf{Resources --1} and start \textbf{Creditor [4]} or \textbf{Inspection [4]}.
          \item \textbf{Yield:} If \textbf{Resources $\geq$ 3} and no active \textbf{Accident/Inspection}, gain one: coin, \textbf{Favor (narrow)}, \textbf{Clue}, or \textbf{Leverage (1)} themed to the asset.
        \end{itemize}
        
        \subsection{Asset Orders (Between Sessions)}
        Choose up to \textbf{2 orders} per session per asset (1 if Resources $\leq$ 2):
        \begin{itemize}
          \item \textbf{Operate:} generate Yield with risk (on 1s, tick \textbf{Heat +1} or \textbf{Integrity +1}).
          \item \textbf{Improve:} advance a \textbf{Project} [4--8].
          \item \textbf{Secure:} reduce \textbf{Heat --1} or add a \textbf{Security} tag for one scene.
          \item \textbf{Expand:} start a new Project to add a String (route, permit, office).
          \item \textbf{Audit:} convert one unspent Favor at this venue into \textbf{Standing (Tag)} after proof.
          \item \textbf{Exploit:} trade 1 \textbf{Integrity} for immediate coin + Favor; mark \textbf{Exposure +1} to someone.
        \end{itemize}
        \paragraph{Resolution} The directing PC rolls once per order (appropriate action). Tier/Tags adjust Position/DV. On hits, apply the order effect; on 1s, GM banks SB and hits Heat/Integrity/Exposure.
        
        \section{Campaign Continuity}
        \subsection{Asset Legacy System}
        Assets evolve and leave marks across campaigns through their \textbf{Legacy Notes}:
        \begin{itemize}
          \item \textbf{Reputation Effects:} Notorious (feared), Respected (trusted), Infamous (recognized), Legendary (historical significance)
          \item \textbf{Historical Events:} Siege Survived, Scandal Weathered, Expansion Completed, Crisis Averted
          \item \textbf{Relationship Changes:} Patron Gained/Lost, Rival Created/Resolved, Community Bond/Feud
          \item \textbf{Physical Evolution:} Renovated, Damaged, Expanded, Relocated
        \end{itemize}
        
        \textbf{Legacy Benefits:} Assets with positive legacy gain +1 Resources or +1 Integrity at start of new campaigns.\\
        \textbf{Legacy Burdens:} Assets with negative legacy start with Heat +2 or Sanction +1 with relevant patrons.
        
        \subsection{Patron Relationship Continuity}
        Worldly Patron relationships persist with modified standing:
        \begin{itemize}
          \item \textbf{Active Patrons:} Reduce Tier by 1 but retain 1 Standing tag
          \item \textbf{Dormant Patrons:} Convert Tier to Favor Ledger entries; can be reactivated
          \item \textbf{Former Patrons:} Become Rivals [4] or Contacts [4] based on ending relationship
          \item \textbf{Legacy Patrons:} Historical figures whose mandates still influence current politics
        \end{itemize}
        
        \subsection{Campaign Transition Events}
        At campaign end/beginning, roll for transition events:
        \begin{itemize}
          \item \textbf{Economic Shift (d6):} 1-2 Resources +1, 3-4 No change, 5-6 Resources -1
          \item \textbf{Political Change (d6):} 1-2 Gain new patron opportunity, 3-4 Status quo, 5-6 Lose a patron string
          \item \textbf{Physical Event (d6):} 1-2 Asset improves, 3-4 No change, 5-6 Asset takes Integrity -1
        \end{itemize}

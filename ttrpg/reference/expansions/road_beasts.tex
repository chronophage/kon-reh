\section*{How to Use This Manual}
\begin{itemize}
    \item \textbf{Monsters} are built from \textbf{Harm levels} (1–3) and optional \textbf{clocks} (4–8 segments). No HP math.
    \item Each entry includes \textbf{Danger}, \textbf{Harm}, \textbf{Clocks}, \textbf{Tags}, and a \textbf{Fate's Edge twist}.
    \item \textbf{Lazy GM Table} (p.4) lets you roll any adversary in 10 seconds.
    \item \textbf{Character Inspiration} (p.5): draw two playing cards to generate a random concept + complication.
\end{itemize}

\section*{Character Inspiration: Two Cards}
\textit{At character creation or when introducing a new NPC, draw two playing cards (suit + rank). Combine a \textbf{Concept} (Heart/Diamond) with a \textbf{Complication} (Spade/Club). No jokers.}

\begin{tabular}{@{}p{0.22\textwidth}p{0.22\textwidth}p{0.22\textwidth}p{0.22\textwidth}}
\toprule
\multicolumn{2}{c}{\textbf{Concepts (Heart or Diamond)}} & \multicolumn{2}{c}{\textbf{Complications (Spade or Club)}} \\
\midrule
\emph{Rank} & \emph{Concept} & \emph{Rank} & \emph{Complication} \\
2–3 & Exiled noble & 2–3 & Hunted by a rival \\
4–5 & Grizzled scout & 4–5 & Carries a cursed heirloom \\
6–7 & Street-wise fixer & 6–7 & Owe a favor to a crime lord \\
8–9 & Failed scholar & 8–9 & Secretly ill (clock [4]) \\
10–J & Faith healer & 10–J & Witness to a forbidden rite \\
Q–K & Retired soldier & Q–K & Spouse missing (search clock) \\
A & Last of a bloodline & A & Marked by a spirit (visible) \\
\bottomrule
\end{tabular}

\textit{Then roll or pick a third card for a \textbf{Goal} (any suit):}
\begin{itemize}
    \item \textbf{Red (Heart/Diamond):} Protect someone / find a treasure.
    \item \textbf{Black (Spade/Club):} destroy a rival / avenge an insult.
    \item \textbf{Ace:} discover the truth about [a place, death, or omen].
\end{itemize}

\textit{Example:} 6♦ (Grizzled scout) + 8♣ (Secretly ill) + Q♠ (avenge an insult) = a sick tracker who wants revenge on the poacher who killed her mentor.

\section*{Quick Clocks for Monster Encounters}
Use these clocks as secondary objectives during fights.

\begin{tabular}{@{}ll}
Clock & Trigger / Effect \\
\midrule
\textit{Rumor Spreads (4)} & Each shouted word ticks. At 4, reinforcements arrive \\
\textit{Sacrifice Complete (6)} & Monster stops fighting to drain a victim; interrupt to cancel \\
\textit{Ritual of Binding (8)} & Players can tick by using a specific skill (Lore, Craft) once per round \\
\textit{Panic (4)} & Ticks when an ally falls. At 4, all enemies test Resolve or flee \\
\textit{Structural Collapse (6)} & Ticks on heavy hits. At 6, terrain becomes hazardous \\
\textit{Dawn (8)} & Vampiric / shadow creatures weaken each tick. At 8, they flee or die \\
\end{tabular}

\section*{Final Notes for the Lazy GM}
\begin{itemize}
    \item You never need to invent a monster from scratch. Use the table above and rename.
    \item For a \textbf{leader-type}, add a clock that summons minions. For a \textbf{swarm}, treat as a zone hazard (test to avoid Harm each turn).
    \item Fate's Edge monsters are not sacks of HP. They have \textbf{twists} – use them to create choices, not just fights.
    \item When a player says "I know this creature," let them roll Lore. On success, they invent one of its tags (subject to your veto).
\end{itemize}

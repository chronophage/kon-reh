\section*{What this module does}
\begin{itemize}[leftmargin=*,noitemsep]
    \item Adds a \term{grid} (square or hex, 5 ft cells) to measure distance, line of sight, and reach.
    \item Introduces \term{movement rates}, \term{terrain costs}, \term{zones of control (ZoC)}, and \term{flanking} – all converted directly into the existing \term{Position} ladder.
    \item All actions resolved with standard \dice{Attribute}{Skill} pools, \storybeats, \boon, and Position/Effect outcomes.
    \item \textbf{No HP, no new damage tracks.} Harm and Fatigue work exactly as in the core rules.
    \item Completely optional – use for one encounter, a whole campaign, or never.
\end{itemize}
\vspace{0.2cm}
\hrule
\vspace{0.6cm}

\section{Core Principles}
\begin{enumerate}[label=\arabic\textit{., leftmargin=}]
    \item \textbf{Narrative first.} The grid is a storytelling tool. Every move must be described in fiction.
    \item \textbf{No mechanics changes.} Dice pools, \storybeats, \boon, Position and Effect work as written.
    \item \textbf{Predictable tactics.} Cover, flanking, reach and facing give players concrete ways to gain Position before rolling.
\end{enumerate}
\vspace{0.2cm}
\hrule
\vspace{0.6cm}

\section{The Grid}
\subsection{Scale}
\begin{itemize}[noitemsep]
    \item \textbf{1 square (or hex) = 5 ft.} Square or hex – pick before session.
    \item Most characters occupy one cell. \term{Large} creatures (e.g. horse) occupy 2×2.
\end{itemize}
\subsection{Range Bands (grid version)}
\begin{tabular}{>{\bfseries}l l}
    Close   & Adjacent cells (melee) \\
    Near    & Up to 6 cells (≈30 ft) \\
    Far     & 7–12 cells (≈35–60 ft) \\
    Absent  & Beyond 12 cells or off‑screen \\
\end{tabular}
\noindent Core range bands still apply; the grid just gives precise counts.

\section{Movement}
\subsection{Movement Rate}
\begin{itemize}[noitemsep]
    \item \textbf{Move action:} \term{Body + 3} cells. (Example: Body 3 → 6 cells)
    \item \textbf{Dash:} Use your action to double movement. After dashing, your \emph{next defence} is at \pos{Desperate}.
    \item \textbf{Difficult terrain:} 2 movement points per cell (mud, rubble, shallow water).
    \item \textbf{Elevation:} Moving \emph{up} costs +1 movement point per level; moving down is free.
\end{itemize}

\section{Zones of Control (ZoC)}
\begin{itemize}[noitemsep]
    \item ZoC = all orthogonal (or all six) adjacent cells.
    \item \textbf{Entering} enemy ZoC ends your movement and places you in melee (\pos{Controlled} with that enemy).
    \item \textbf{Leaving} ZoC requires a \term{Disengage} test: DV 4 (plus any modifiers). Failure leaves you \pos{Desperate} for your next action.
    \item \textbf{Stacked ZoC:} +1 to Disengage DV per extra enemy.
    \item \textbf{Pushing Through ZoC:} To move through an enemy's Zone of Control without ending movement, make a Skill test (Athletics or Acrobatics) vs DV $4 + (N-1)$, where $N$ is the number of enemies exerting ZoC in the target path. On success, you may continue your movement through the ZoC. On failure, movement ends as normal (entering ZoC places you in melee). \textit{Regardless of outcome, pushing through incurs one cost:} 
    \begin{itemize}[noitemsep]
        \item \textbf{Gain 1 Fatigue},
        \item \textbf{or GM gains 1 \storybeats},
        \item \textbf{or your next action suffers \pos{Risky} (if currently \pos{Controlled}) or \pos{Desperate} (if currently \pos{Risky})}.
    \end{itemize}
    \hfill\textit{(Higher skill ranks improve success chance—veterans thread through chaos where novices falter.)}
\end{itemize}

\section{Cover \& Position}
\begin{tabular}{>{\bfseries}l l}
\hline
Partial cover (low wall)     & Defender gains \pos{Controlled} against ranged attacks only. \\
Full cover (solid wall)      & Defender gains \pos{Controlled} against all attacks; attacker gains \pos{Dominant} when defending against that target. \\
\hline
\end{tabular}
\noindent Cover can be destroyed (GM spends \storybeats).

\section{Flanking \& Rear Attacks}
\begin{itemize}[noitemsep]
    \item \textbf{Flanking:} Two allies on opposite sides of a target (orthogonal or hex). Next attacker gains \pos{Dominant} for that attack.
    \item \textbf{Rear attack:} Attacker directly behind target. Gain \pos{Dominant} \emph{and} upgrade \effect{Standard} to \effect{Great} for that strike.
\end{itemize}

\section{Weapon Tags (Grid interactions)}
\begin{tabular}{>{\bfseries}p{3cm} p{10cm}}
\toprule
\textbf{Tag} & \textbf{Grid Effect}\\
\midrule
\term{[Reach]} & Can attack up to 2 cells away (Near). Enemies without Reach suffer –1 Position when trying to close.\\
\term{[Fast]} & After a successful melee hit, shift 1 cell (no ZoC trigger).\\
\term{[Set]} & Plant weapon as an action. Any enemy that charges into that cell ($\ge4$ cells straight) triggers an automatic attack at +1 Effect.\\
\term{[Brutal]} & On a hit, push target 1 cell orthogonally. Target’s next defensive Position drops one step.\\
\bottomrule
\end{tabular}

\section{Action Economy (Tactical)}
Each turn:
\begin{enumerate}[label=\arabic\textit{., leftmargin=}]
    \item \textbf{One Action} – attack, cast, use skill, activate talent, etc.
    \item \textbf{One Move} – up to movement rate (can be taken before, after, or split around the action).
\end{enumerate}
\noindent Exceptions: Dashing uses the Action, no separate attack that turn (unless a talent gives a free attack).

\section{Position Sources – Quick Lookup}
\begin{center}
\begin{tabular}{>{\bfseries}l l}
\toprule
\textbf{Situation} & \textbf{Position Change}\\
\midrule
Partial cover (defender)     & \pos{Controlled} (ranged only)\\
Full cover (defender)        & \pos{Controlled} (all attacks)\\
Flanking (attacker)          & \pos{Dominant}\\
Rear attack (attacker)       & \pos{Dominant} + \effect{Great}\\
Dashing (defender next)      & \pos{Desperate}\\
Disengage failure (defender) & \pos{Desperate}\\
Spend a Boon before roll     & Shift Position up one step\\
\bottomrule
\end{tabular}
\end{center}
\noindent All other Position assignments follow the core rules (terrain, surprise, etc.).

\section{Line of Sight \& Concealment}
\begin{itemize}[noitemsep]
    \item Line of sight blocked by solid obstacles (walls, large creatures, dense forest). GM may call for Notice/Stealth if partially concealed.
    \item Concealment (fog, smoke, darkness) gives the defender \pos{Controlled} against ranged attacks. Melee unaffected unless attacker must enter the concealed area.
\end{itemize}

\section{Example Turn}
\begin{enumerate}[label=\arabic\textit{., leftmargin=}]
    \item Harper (Body 4, Melee 5) stands 5 cells from a guard behind a low wall (partial cover).
    \item Harper dashes 12 cells, ending adjacent (enters guard’s ZoC). Guard now \pos{Controlled}, Harper \pos{Dominant} for his attack.
    \item Valeros is already on the opposite side – flanking active.
    \item Harper attacks at \pos{Dominant} (flanking bonus). Rolls \dice{Body}{Melee} = 9 dice.
    \item 6 successes vs DV 3 → clean hit. Harper uses \term{[Brutal]} to push the guard 1 cell away, out of cover.
    \item Guard’s next defence is at \pos{Desperate} (push effect).
    \item Valeros attacks the now coverless, \pos{Desperate} guard with \term{[Fast]} dagger, shifts 1 cell after the hit.
\end{enumerate}
\noindent Dice, Boons, and Outcome matrix untouched; only Position derived from the grid.

\section{GM Guidance}
\begin{enumerate}[label=\arabic\textit{., leftmargin=}]
    \item Draw the grid early – dry‑erase board or VTT.
    \item Read Position first: “What is the attacker’s Position? What can the defender do to change it?”
    \item Spend \storybeats\ on dramatic consequences (cover collapse, reinforcements, etc.).
    \item Use Boons to upgrade Position – a powerful tactical choice.
\end{enumerate}

\chapter{Combat and Conflict}

\section{Core Philosophy}\index{Combat!Core Philosophy}
Combat is violent conflict resolved through the standard consequence mechanics. Every combat action generates potential for both triumph and complication, with consequences that cascade through the same economy as all other challenges.

\section{Resolution Procedure}\index{Combat!Resolution Procedure}
\begin{enumerate}
    \item \textbf{Declare Action}: Player states intent and approach (Attribute + Skill)\index{Combat!Declare Action}
    \item \textbf{Set Position}: GM sets Controlled, Risky, or Desperate based on tactical situation\index{Combat!Set Position}
    \item \textbf{Roll Dice}: Roll pool = Attribute + Skill\index{Combat!Roll Dice}
    \item \textbf{Count Results}: 6+ = Success, 1 = Complication Point (CP)\index{Combat!Count Results}
    \item \textbf{Apply Outcome}: Use standard Outcome Matrix\index{Combat!Apply Outcome}
    \item \textbf{Manage Consequences}: GM spends CP or draws from Consequences Deck\index{Combat!Manage Consequences}
\end{enumerate}

\section{Position States}\index{Combat!Position States}
\begin{description}
\item[Controlled] Advantageous position, minor consequences (flanking, higher ground, surprised foe)\index{Combat!Controlled}
\item[Risky] Even odds, moderate consequences (evenly matched, contested terrain)\index{Combat!Risky}
\item[Desperate] Disadvantaged, severe consequences (outnumbered, wounded, poor positioning)\index{Combat!Desperate}
\end{description}

\section{Combat-Specific Consequence Types}\index{Combat!Consequence Types}
\begin{description}
\item[Hearts] Morale, fear, command/control breakdown, psychological pressure\index{Combat!Hearts}
\item[Spades] Physical harm, positioning changes, weapon status, tactical wounds\index{Combat!Spades}
\item[Clubs] Resource depletion, gear damage, fatigue, ammunition/supply issues\index{Combat!Clubs}
\item[Diamonds] Environmental hazards, reinforcements, tactical setbacks, terrain changes\index{Combat!Diamonds}
\end{description}

\section{Harm Integration}\index{Combat!Harm Integration}
Harm tracks directly tie to CP economy:
\begin{description}
\item[Minor (-)] Generate 1 CP on next 2 rolls\index{Combat!Harm!Minor}
\item[Moderate (=)] Generate 1 CP on next roll, -1 die to relevant actions\index{Combat!Harm!Moderate}
\item[Severe (≈)] Generate 2 CP on next roll, -2 dice to relevant actions\index{Combat!Harm!Severe}  
\item[Critical (†)] Generate 3 CP on next roll, out of action until treated\index{Combat!Harm!Critical}
\end{description}

\section{Tactical Clocks}\index{Combat!Tactical Clocks}
Persistent combat conditions are tracked through clocks:
\begin{itemize}
    \item \textbf{Mob Overwhelm} (6): Enemy numbers become advantage\index{Combat!Clocks!Mob Overwhelm}
    \item \textbf{Fatigue Spiral} (4): Exhaustion affects performance\index{Combat!Clocks!Fatigue Spiral}
    \item \textbf{Morale Collapse} (6): Fear undermines effectiveness\index{Combat!Clocks!Morale Collapse}
    \item \textbf{Environmental Collapse} (8): Terrain/fire/building failure\index{Combat!Clocks!Environmental Collapse}
\end{itemize}

\section{Position Dynamics}\index{Combat!Position Dynamics}
Position can shift during combat based on CP spending:
\begin{itemize}
    \item \textbf{1 CP}: Shift position one step (GM choice)\index{Combat!Position Shift}
    \item \textbf{Player Spending}: 1 CP to improve position one step\index{Combat!Player Spending}
    \item \textbf{Narrative Triggers}: Flanking, reinforcement arrival, environmental changes\index{Combat!Narrative Triggers}
\end{itemize}

\section{Magic Combat Integration}\index{Combat!Magic Integration}
Spellcasting in combat feeds the same consequence economy:
\begin{itemize}
    \item Channel/Weave Backlash CP applies to tactical situation\index{Magic!Combat}
    \item Spells can shift position, create tactical clocks, or generate combat consequences\index{Magic!Tactical Effects}
    \item Magic consequences cascade through existing combat systems\index{Magic!Cascade Effects}
\end{itemize}

\section{Asset/Follower Combat Integration}\index{Combat!Asset Integration}
\begin{itemize}
    \item \textbf{Follower Risk}: 2+ CP spent in combat can endanger assisting followers\index{Combat!Follower Risk}
    \item \textbf{Asset Compromise}: Combat in certain locations can damage relevant assets\index{Combat!Asset Compromise}  
    \item \textbf{Offensive Activation}: 1 Boon activates asset for combat advantage\index{Combat!Asset Activation}
    \item \textbf{Initiative Actions}: Followers can take combat-relevant independent actions\index{Combat!Follower Initiative}
\end{itemize}

\section{Outcome Matrix Application}\index{Combat!Outcome Matrix}
Same as standard resolution, but consequences are combat-specific:
\begin{description}
\item[Clean Success] Intent achieved with no tactical complications\index{Combat!Outcome!Clean Success}
\item[Success \& Cost] Intent achieved, but GM spends CP for combat consequences\index{Combat!Outcome!Success \& Cost}
\item[Partial] Progress with tactical fork (accept cost OR concede ground)\index{Combat!Outcome!Partial}
\item[Miss] No progress; GM spends CP for combat consequences. Award one Boon. 
\end{description}

\section{Social Conflict}\index{Social Conflict}

\subsection{Persuasion and Influence}\index{Social Conflict!Persuasion}
Social conflicts use the same core mechanics but with different skills:
\begin{itemize}
\item Sway for negotiation and formal discourse\index{Skills!Sway}
\item Deception for lies, misdirection, and manipulation\index{Skills!Deception}
\item Performance for oratory, entertainment, and emotional appeal\index{Skills!Performance}
\item Insight for reading opponents and detecting deception\index{Skills!Insight}
\end{itemize}

\subsection{Social Position}\index{Social Conflict!Position}
\begin{description}
\item[Controlled] You have leverage, information, or social advantage\index{Social Conflict!Controlled}
\item[Risky] Even social ground; standard interaction\index{Social Conflict!Risky}
\item[Desperate] You're at a disadvantage; opponent has leverage\index{Social Conflict!Desperate}
\end{description}

\subsection{Social Consequences}\index{Social Conflict!Consequences}
Complication Points in social conflicts might manifest as:
\begin{itemize}
\item Rumors spread that harm reputation\index{Social Conflict!Reputation}
\item Allies become suspicious or hostile\index{Social Conflict!Allies}
\item Obligations or concessions must be made\index{Social Conflict!Obligations}
\item Social standing or access is compromised\index{Social Conflict!Standing}
\end{itemize}

\subsection{War Clocks System}\index{Combat!Mass Combat!War Clocks}

Large-scale conflicts are tracked through persistent war-level clocks that represent ongoing conditions and pressures.

\subsubsection{Supply Lines Clock (8 segments)}\index{Combat!Mass Combat!Supply Lines}
Tracks the flow of resources to the army:
\begin{itemize}[leftmargin=*]
    \item \textbf{0-2 segments:} Adequate supply, no penalties
    \item \textbf{3-5 segments:} Stretched supply lines, -1 die to all actions
    \item \textbf{6-7 segments:} Critical shortages, -2 dice to all actions, Fatigue 1 per day
    \item \textbf{8 segments:} Supply collapse, army becomes ineffective, begins retreat
\end{itemize}

\textbf{Advancement Triggers:}
\begin{itemize}[leftmargin=*]
    \item Enemy raids on supply convoys (+1)
    \item Extended campaign duration (+1 per week)
    \item Difficult terrain or weather (+1)
    \item Political interference (+1)
\end{itemize}

\subsubsection{Morale Clock (6 segments)}\index{Combat!Mass Combat!Morale}
Tracks troop effectiveness and willingness to fight:
\begin{itemize}[leftmargin=*]
    \item \textbf{0-1 segments:} High morale, +1 die to all actions
    \item \textbf{2-3 segments:} Standard morale, no modifiers
    \item \textbf{4-5 segments:} Low morale, -1 die to all actions, desertion risk
    \item \textbf{6 segments:} Collapse, army routs or surrenders
\end{itemize}

\textbf{Advancement Triggers:}
\begin{itemize}[leftmargin=*]
    \item Heavy casualties (+1)
    \item Loss of key leaders (+1)
    \item Defeat in battle (+2)
    \item Extended combat without rest (+1 per week)
\end{itemize}

\subsubsection{Strategic Position Clock (8 segments)}\index{Combat!Mass Combat!Strategic Position}
Tracks control of key locations and tactical advantages:
\begin{itemize}[leftmargin=*]
    \item \textbf{0-2 segments:} Superior position, +1 die to all actions
    \item \textbf{3-5 segments:} Balanced position, no modifiers
    \item \textbf{6-7 segments:} Disadvantaged position, -1 die to all actions
    \item \textbf{8 segments:} Strategic collapse, forced retreat or surrender
\end{itemize}

\textbf{Advancement Triggers:}
\begin{itemize}[leftmargin=*]
    \item Loss of key fortifications (+2)
    \item Enemy control of supply routes (+1)
    \item Superior enemy numbers (+1)
    \item Failed tactical maneuvers (+1)
\end{itemize}

\subsection{Command Actions and Leadership}\index{Combat!Mass Combat!Command}

\subsubsection{Lead Action for Armies}\index{Combat!Mass Combat!Lead Action}
\textbf{Requirements:} Presence + Command skill, appropriate position

\textbf{Effect:} Coordinate up to three army units for a combined action:
\begin{itemize}[leftmargin=*]
    \item \textbf{Controlled Position:} All units gain +1 die, one unit can act as Cap 6
    \item \textbf{Risky Position:} All units act normally, one unit can act as Cap 5
    \item \textbf{Desperate Position:} All units act at -1 die, one unit can act as Cap 4
\end{itemize}

\textbf{Complications:}
\begin{itemize}[leftmargin=*]
    \item 1-2 CP: Communication breakdown, one unit acts independently
    \item 3-4 CP: Command structure compromised, -1 die to all units next player turn
    \item 5+ CP: Complete command collapse, all units act independently with -2 dice
\end{itemize}

\section{Environmental Combat}\index{Combat!Environmental}

\subsection{Environmental Hazards and Clocks}\index{Combat!Environmental!Hazards}

Environmental factors create dynamic combat scenarios that evolve over time.

\subsubsection{Fire Hazard Clock (6 segments)}\index{Combat!Environmental!Fire}
\begin{itemize}[leftmargin=*]
    \item \textbf{0-1 segments:} Minor fire, no immediate danger
    \item \textbf{2-3 segments:} Spreading fire, -1 die to actions in affected area
    \item \textbf{4-5 segments:} Intense fire, Harm 1 to anyone in area per player turn
    \item \textbf{6 segments:} Inferno, area becomes impassable, Harm 2 to anyone caught
\end{itemize}

\textbf{Advancement:}
\begin{itemize}[leftmargin=*]
    \item Flammable materials present (+1)
    \item Strong winds (+1)
    \item Failed suppression attempts (+1)
\end{itemize}

\textbf{Suppression Actions:}
\begin{itemize}[leftmargin=*]
    \item Water/chemical suppression: Body + Athletics (DV 2), success reduces clock by 1
    \item Firebreak creation: Wits + Survival (DV 3), success reduces clock by 2
    \item Magical suppression: Wits + (Appropriate Art) (DV 2), success reduces clock by 1-3
\end{itemize}

\subsubsection{Flood Hazard Clock (8 segments)}\index{Combat!Environmental!Flood}
\begin{itemize}[leftmargin=*]
    \item \textbf{0-2 segments:} Rising water, difficult terrain
    \item \textbf{3-5 segments:} Deep water, swimming required, -1 die to actions
    \item \textbf{6-7 segments:} Strong current, risk of being swept away
    \item \textbf{8 segments:} Catastrophic flood, area submerged, automatic Harm 2
\end{itemize}

\textbf{Advancement:}
\begin{itemize}[leftmargin=*]
    \item Heavy rainfall (+1)
    \item Dam/levee failure (+2)
    \item Poor drainage (+1)
\end{itemize}

\textbf{Survival Actions:}
\begin{itemize}[leftmargin=*]
    \item Swimming: Body + Athletics (DV 2), failure results in being swept 1 zone
    \item High ground seeking: Wits + Survival (DV 3), success grants Controlled position
    \item Rescue attempts: Presence + Command (DV 2), success saves endangered allies
\end{itemize}

\subsection{Terrain Effects}\index{Combat!Environmental!Terrain}

\subsubsection{Choke Points}\index{Combat!Environmental!Choke Points}
Narrow passages that limit movement and tactical options:
\begin{itemize}[leftmargin=*]
    \item Only 2 characters can engage in melee at once
    \item Ranged attackers have -1 die due to limited targets
    \item Flanking becomes difficult (-2 dice to flanking attempts)
    \item Successful choke point control grants +2 dice to defensive actions
\end{itemize}

\subsubsection{Elevated Positions}\index{Combat!Environmental!Elevated Positions}
Higher ground that provides tactical advantages:
\begin{itemize}[leftmargin=*]
    \item Characters on high ground gain start Controlled vs. opponents below
    \item Ranged attacks from high ground gain +1 effect
    \item Melee attackers from below suffer -1 die
    \item Moving to or from elevated positions requires successful Athletics roll
\end{itemize}

\subsubsection{Cover and Concealment}\index{Combat!Environmental!Cover}
Environmental features that provide protection:
\begin{itemize}[leftmargin=*]
    \item Light cover: +1 die to defense rolls, start Controlled vs. ranged attacks
    \item Heavy cover: +2 dice to defense rolls, -1 die to attacker's actions
    \item Concealment: Start Controlled vs. detection, -1 die to attacker's perception
    \item Moving between cover positions requires successful Stealth roll
\end{itemize}

\section{Siege Warfare}\index{Combat!Siege Warfare}

\subsection{Extended Combat Scenarios}\index{Combat!Siege Warfare!Extended}

Siege warfare creates persistent conditions that affect both attackers and defenders over extended periods.

\subsubsection{Siege Conditions Clock (10 segments)}\index{Combat!Siege Warfare!Conditions}
Tracks the overall progress and strain of siege conditions:
\begin{itemize}[leftmargin=*]
    \item \textbf{0-2 segments:} Initial siege, minor inconveniences
    \item \textbf{3-5 segments:} Established siege, supply concerns begin
    \item \textbf{6-8 segments:} Prolonged siege, significant strain on defenders
    \item \textbf{9-10 segments:} Critical siege, imminent collapse or breakthrough
\end{itemize}

\textbf{Advancement Triggers:}
\begin{itemize}[leftmargin=*]
    \item Successful siege engineering (+1)
    \item Defender supply depletion (+1)
    \item Disease outbreak (+2)
    \item Reinforcements arrival (-1 to +2 depending on side)
\end{itemize}

\subsection{Resource Management During Siege}\index{Combat!Siege Warfare!Resources}

\subsubsection{Rapid Supply Clock Filling}\index{Combat!Siege Warfare!Supply}
During sieges, the Supply Clock fills more rapidly:
\begin{itemize}[leftmargin=*]
    \item \textbf{Daily Filling:} +1 segment per day without resupply
    \item \textbf{Critical Shortages:} +2 segments if supply line is cut
    \item \textbf{Rationing:} +1 segment per day when on half rations
\end{itemize}

\textbf{Mitigation Actions:}
\begin{itemize}[leftmargin=*]
    \item Foraging: Wits + Survival (DV 3), success slows supply depletion by 1 day
    \item Supply smuggling: Presence + Subterfuge (DV 4), success adds 2 segments to supply
    \item Local procurement: Presence + Sway (DV 2), success adds 1 segment to supply
\end{itemize}

\subsubsection{Fatigue Accumulation}\index{Combat!Siege Warfare!Fatigue}
Characters gain Fatigue more rapidly during extended sieges:
\begin{itemize}[leftmargin=*]
    \item \textbf{Daily Accumulation:} +1 Fatigue per day without adequate rest
    \item \textbf{Combat Conditions:} +1 additional Fatigue per combat encounter
    \item \textbf{Poor Conditions:} +1 additional Fatigue per day in unsanitary conditions
\end{itemize}

\textbf{Recovery Actions:}
\begin{itemize}[leftmargin=*]
    \item Rest and recuperation: Requires safe conditions and adequate supply
    \item Medical treatment: Wits + Medicine (DV 2), success removes 1 Fatigue
    \item Magical healing: Wits + Vitalism (DV 2), success removes 2 Fatigue
\end{itemize}

\section{Magic Duels}\index{Combat!Magic Duels}

\subsection{High-Stakes Magical Combat}\index{Combat!Magic Duels!High Stakes}

Magical duels require special considerations for pacing, risk management, and dramatic tension.

\subsubsection{Counterspelling Mechanics}\index{Combat!Magic Duels!Counterspelling}
\textbf{Interrupting Spellcasting:}
\begin{itemize}[leftmargin=*]
    \item Opposed roll: Wits + Arcana (caster) vs. Wits + Arcana (counter-speller)
    \item Success: Interrupts Channel phase, caster takes 1 CP
    \item Critical success: Completely disrupts spell, caster takes 2 CP
    \item Failure: Counter-speller takes 1 CP, original casting proceeds
\end{itemize}

\textbf{Defensive Magic:}
\begin{itemize}[leftmargin=*]
    \item Shield spells: Can be maintained as ongoing effects (1 Fatigue/round)
    \item Ward creation: Wits + Geomancy/Thaumaturgy (DV 3), creates lasting protection
    \item Spell absorption: Requires specific talents, converts incoming magic to beneficial effect
\end{itemize}

\subsubsection{Backlash Cascade Effects}\index{Combat!Magic Duels!Backlash}
In magical duels, multiple sources of CP can create cascading consequences:
\begin{itemize}[leftmargin=*]
    \item Each 1 rolled during Channel generates immediate CP for opponent to spend
    \item Failed Weave actions generate CP that can be spent by either participant
    \item Environmental magical effects can generate additional CP sources
    \item Spell collision (two spells affecting same target) creates 1 CP for each caster
\end{itemize}

\textbf{Cascade Management:}
\begin{itemize}[leftmargin=*]
    \item Players can spend Boons to mitigate incoming CP during cascade
    \item Talents like "Backlash Soothing" can reduce cascade effects
    \item Environmental control can limit cascade opportunities
\end{itemize}

\subsection{Follower Combat Integration}\index{Combat!Quick Reference!Followers}

\subsubsection{Risk Management}\index{Combat!Quick Reference!Risk}
\textbf{2+ CP Spent in Combat:}
\begin{itemize}[leftmargin=*]
    \item Follower may be endangered instead of PC
    \item GM chooses which follower if multiple present
    \item Fictional appropriateness determines eligibility
    \item Follower takes Exposure +1 or Harm 1
\end{itemize}

\textbf{Initiative Actions:}
\begin{itemize}[leftmargin=*]
    \item One per scene party-wide
    \item Cost: Exposure +1 or Harm 1 on follower
    \item Options:
    \begin{itemize}
        \item Scout Signal: Ally starts Controlled or +1 effect
        \item Distract Draw: Reduce kinetic rail by -1 tick
        \item Fetch Carry: Move object through danger
    \end{itemize}
\end{itemize}

\subsubsection{Asset Compromise}\index{Combat!Quick Reference!Assets}
\textbf{Combat in Specific Locations:}
\begin{itemize}[leftmargin=*]
    \item 1-2 CP: Asset becomes Neglected
    \item 3-4 CP: Asset becomes Compromised
    \item 5+ CP: Asset destroyed or captured
    \item Recovery requires downtime or narrative quest
\end{itemize}

\textbf{Offensive Activation:}
\begin{itemize}[leftmargin=*]
    \item Cost: 1 Boon
    \item Must have scope and reach
    \item Examples:
    \begin{itemize}
        \item Safehouse provides escape route
        \item Spy network provides intelligence
        \item Military asset provides reinforcement
    \end{itemize}
\end{itemize}

\textbf{During Extended Conflicts:}
\begin{itemize}[leftmargin=*]
    \item Update clocks at the end of each major engagement

    \item Critical thresholds trigger narrative consequences
    \item Clocks can be linked for cascading effects
\end{itemize}

\textbf{Victory Conditions:}
\begin{itemize}[leftmargin=*]
    \item Fill 2+ war clocks to critical levels
    \item Reduce enemy army effectiveness to Militia level

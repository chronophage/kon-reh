\begin{center}
{\LARGE \textbf{Riarhin — Sapphire Wolf of the Bow}}\\[4pt]
{\large Half-Wood-Elf Vigilante, Hunter of the Wicked, Shadow-Quick Sniper}
\end{center}

\section*{Profile}
\textbf{Name:} Riarhin\\
\textbf{Tier:} V (Masterwork)\\
\textbf{Concept:} Lethal archer, lightning-fast tactician, hunter of tyrants and predators\\
\textbf{Affiliations:} None pledged; rumored ties to Ikasha through blood and shadow\\
\textbf{Familiar:} \emph{Ashwing}, storm-feathered raven — keen-eyed and unsettlingly intelligent\\
\textbf{Armor:} Elven harbark — light, silent chain-mesh that moves like cloth

\section*{Attributes}
Body 3 — trained, fast-twitch strength\\
Wits 4 — snap-judgment precision, predator’s focus\\
Spirit 3 — icy calm, steel-nerved under fire\\
Presence 2 — quiet, sharp, hard to read

\section*{Skills}
Ranged 5; Stealth 4; Acrobatics 3; Awareness 3; Melee 3; Survival 3; Intimidation 2

\section*{Talents}
\begin{itemize}
  \item \textbf{Deadeye:} From Far range, either gain +1 Effect or inflict Prone on a Success.
  \item \textbf{Phantom Step:} Move between covers without provoking Opportunity on a Success.
  \item \textbf{Precision Kill:} On a 10 from Far or High Ground, ignore armor for that strike.
  \item \textbf{Cold Oath:} If outnumbered, gain +1 die to attacks until only one enemy remains.
  \item \textbf{Familiar — Ashwing (Raven):} Once per scene, may \emph{Beast Speech} — utter a brief mortal-tongue warning, command, or omen.
\end{itemize}

\section*{Signature Move — Phantom Shot}
\textbf{Trigger:} Attacking from Far \emph{or} acting from \emph{Dominant} Position.\\
\textbf{Cost:} Spend 1 Boon (ignore cost if already \emph{Dominant}).\\
\textbf{Effect:}
\begin{itemize}
  \item The attack ignores Cover and Shields for this strike.
  \item Gain +1 die on the attack. On a 10, either deal +1 Harm \emph{or} pin a second foe (they lose their next Move).
\end{itemize}
\textbf{Fiction:} A feint-arrow draws the eye; the true line kills.

\section*{Signature Relics}
\begin{itemize}
  \item \textbf{\emph{Irisil}, Elven Bow — Sapphire Wolf (Relic, Attuned):}  
  An ancient bow of moonlight-lacquered yew.  
  \begin{itemize}
    \item When setting an ambush, the first successful attack each scene counts as a Critical if it deals Harm.
    \item If she waits motionless with a clear line-of-sight, treat her first attack as \emph{Dominant}.
  \end{itemize}

  \item \textbf{\emph{Toothpick} — Stiletto (Relic):}  
  Needle-thin blade hidden in wrist-sheath.  
  \begin{itemize}
    \item Treated as Concealed until drawn.
    \item When drawn against a surprised or shaken target, gain +1 die.
  \end{itemize}

  \item \textbf{\emph{The Last Blade} — Ykrul Dirk (Relic):}  
  Forged for silent endings.  
  \begin{itemize}
    \item When it tastes blood, gain +1 die to Stealth for the next scene.
  \end{itemize}

  \item \textbf{\emph{Remorse} — Curved Sword (Relic, Veiled Power):}  
  The blade that judges the unworthy.  
  \begin{itemize}
    \item When unsheathed, \emph{Silence the Field}: shouting muffles; panic fails to spread.
    \item On first engagement each scene, foes must test Fear (DV by GM).
    \item True power remains veiled; the sword “grooms leaders,” yet offers no comfort.
  \end{itemize}

  \item \textbf{Elven Harbark (Light Armor):}  
  Whisper-silent chain-mesh.  
  \begin{itemize}
    \item Grants armor without noise penalty.
    \item When braced and unseen, gain +1 die to initiate a Stealth ambush.
  \end{itemize}
\end{itemize}

\section*{Familiar — Ashwing, Storm-Feather Raven}
Ink-black feathers banded with sapphire sheen. One eye is pale sky-blue.  
\textbf{Traits:}
\begin{itemize}
  \item Sharp-eyed spotter: advantage to Awareness from high perch
  \item Ghost-flight: barely a whisper of wings
  \item \emph{Beast Speech} (1/scene): words crackle like lightning in a storm
  \item Loyal, sardonic, and entirely unafraid of kings or killers
\end{itemize}
\emph{Bond:} “Silent, sharp, certain. The wind knows your name, Wolf.”

\section*{Methods and Tactics}
\textbf{Ambush Doctrine:} Mark sightlines, measure breath, kill the loudest first.\\
\textbf{Range Mastery:} Prefers Far or High Ground; will fight in Melee only to finish.\\
\textbf{Clean Climb}: Acrobatics to reposition along walls, trees, rigging, balconies.\\
\textbf{Fear Economics:} One body falls; the room breaks. She moves before it breathes.

\section*{Reputation}
Whispers: “The Sapphire Wolf.”\\
Saves small towns from tyrants, hunts slavers and war-thieves, vanishes without pay.\\
Never signs a charter. Never asks permission. Never misses twice.

\bigskip
\noindent\emph{GM Hooks}  
\begin{itemize}
  \item Someone powerful marked her family for death; she marked them back.
  \item Remorse hung once above a king’s throne; someone wants it returned.
  \item Ashwing has spoken a prophecy only once: “The quiet crown will fall.”
  \item Ikasha’s shadows do not follow her—they wait.
\end{itemize}

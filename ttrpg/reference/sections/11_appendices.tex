\subsection{Core Mechanic Quick Reference}\label{subsec:core_mechanic_ref}
\index{Core resolution!quick reference}

\begin{enumerate}
    \item \textbf{Approach:} Player states intent and method (Attribute + Skill combination).\index{Core resolution!approach}
    \item \textbf{Execution:} Roll dice pool of d10s. Each 6+ is a success; each 1 generates 1 Story Beat.\index{Core resolution!execution}
    \item \textbf{Description Ladder:}
        \begin{itemize}
            \item \textbf{Basic} --- Roll as‑is; all 1s generate SB.
            \item \textbf{Detailed} --- Re‑roll one die showing 1.
            \item \textbf{Intricate} --- Re‑roll all 1s; add a positive flourish on success.
        \end{itemize}
\end{enumerate}

\subsection{Upkeep (Condensed)}\label{subsec:upkeep_srd}
\index{Upkeep}\index{Assets!upkeep}\index{Followers!upkeep}

\textbf{Frequency:} Once per Downtime period.

\begin{itemize}
    \item \textbf{Efficient (Higher XP, Less Time):} Upkeep XP = \(\max\bigl(1, \frac{\text{XP Acquisition Cost}}{3}\bigr)\). Minimal effort (delegation/check‑in).
    \item \textbf{Intensive (Lower XP, More Time):} 1 XP + one dedicated Downtime action of significant personal attention.
\end{itemize}

\textbf{Failure:} If unpaid, resource degrades:
\begin{itemize}
    \item \emph{Follower:} becomes \textbf{Wary} (or \textbf{Seized} if already Wary).
    \item \emph{Asset:} becomes \textbf{Neglected} (or \textbf{Compromised} if already Neglected).
\end{itemize}

\section{Deck Usage Reference}\label{sec:deck_reference}
\index{Decks!reference}

\subsection{Deck Types and Meanings}\label{subsec:deck_types}
\begin{description}
    \item[\textbf{Travel Decks} (regional, 52‑card)] Used for journey content and location‑based adventures.
        \begin{itemize}
            \item Spade $=$ Place/Location
            \item Heart $=$ Actor/Faction
            \item Club $=$ Pressure/Complication
            \item Diamond $=$ Leverage/Reward
        \end{itemize}
    \item[\textbf{Deck of Consequences} (scene drama)] Used for immediate complications and narrative twists.
        \begin{itemize}
            \item Hearts $=$ Social/Emotional fallout
            \item Spades $=$ Harm/Escalation
            \item Clubs $=$ Material cost/Resource drain
            \item Diamonds $=$ Magical/Spiritual disturbance
        \end{itemize}
\end{description}
\textbf{Important:} Never mix suit meanings across decks.

\subsection{Deck Usage Procedure}\label{subsec:deck_procedure}
After a roll generating Story Beats:
\begin{enumerate}
    \item \textbf{Direct Spend:} Translate SB into immediate consequences or clock ticks.
    \item \textbf{Deck Draw:} Draw up to \(\min(\text{SB},\,3)\) cards and synthesise a single twist.
    \item Interpret cards based on suit meanings and highest rank.
\end{enumerate}

\subsection{Rank Severity Guide}\label{subsec:rank_severity}
\begin{description}
    \item[Ace--3] Minor inconvenience or flavour complication.
    \item[4--6] Moderate setback with narrative impact.
    \item[7--9] Significant consequence altering the scene.
    \item[10--King] Major fallout introducing new problems or lasting effects.
\end{description}

\section{Common Rules Questions}\label{sec:rules_questions}
\index{Rules questions}

\subsection{Core Mechanic Clarifications}\label{subsec:core_clarifications}
\paragraph{Q: Can players re‑roll 1s to remove Story Beats?}
A: No. Re‑rolling 1s does not remove SB already generated. If re‑rolled dice show 1 again, they generate additional SB.

\paragraph{Q: When does a miss or partial success award Boons?}
A: Only when all three conditions are met: procedure followed, stakes stated, and consequence lands immediately.

\paragraph{Q: Can players assist each other on every action?}
A: Yes, but total assist dice are capped at +3 from all sources combined.

\subsection{Magic System Questions}\label{subsec:magic_questions}
\paragraph{Q: Can a character use multiple magic paths?}
A: Yes, but each path has its own tracking (Backlash, Obligation, Symbol states). Specialising is more efficient.

\paragraph{Q: How does Crack the Seal work for Invokers?}
A: Convert a ritual to instant casting by setting the Symbol to \emph{Compromised} and marking +2/+3 Obligation.

\subsection{Combat and Conflict Questions}\label{subsec:combat_questions}
\paragraph{Q: How does the Over‑Stack rule work?}
A: If the party \emph{enters} a scene with $\geq 3$ structural advantages, either start one challenge at +1 difficulty \emph{or} bank +1 SB.

\paragraph{Q: Can players spend Boons to improve Position?}
A: Yes. 1 Boon improves Position by 1 step for the current action.

\subsection{Condition \& Upkeep}
\begin{description}
    \item[\textbf{Neglected}] Weapons: --1 die until maintained. Armor: conversion reduced by 1 step (Light=0; Medium treats 2$\rightarrow$1F; Heavy treats 3$\rightarrow$2F).

\textit{Maintenance:} A Short Rest with tools removes \textbf{Neglected}. A scene of work or a Smith removes \textbf{Compromised}.

\begin{itemize}
    \item \textbf{Aim:} +1 die \emph{or} +1 Effect next shot (lost if you Move or are disrupted).
    \item \textbf{Volley:} Spend extra ammo for +1 die (max +2).
    \item \textbf{Suppress:} Zone under fire; foes there act at --1 die or Limited Effect until they shift cover.
    \item \textbf{Overwatch:} Ready a Controlled shot on a clear trigger crossing your line.
\end{itemize}

\section{The Whispered Grimoire (Rituals: Condensed SRD)}\label{sec:rituals_condensed}
\index{Rituals!condensed}

\begin{tcolorbox}[title=Rituals (SRD Format),colback=gray!4,colframe=black!60]
Each ritual lists: \textbf{Purpose}, \textbf{Prereqs} (minimum pools), \textbf{Time}, \textbf{Duration}, \textbf{Test} (typical DV), \textbf{Effect}, \textbf{Backlash (SB tiers)}.\\
\emph{Position:} Unless stated, assume \textbf{Controlled}. Consecrated/strong leverage may grant \textbf{Dominant}; hostile site/opposition may impose \textbf{Desperate}.\\
\emph{Criticals:} A roll of \textbf{10} counts as \emph{two successes}; does not auto‑succeed if total $< \mathrm{DV}$.
\end{tcolorbox}

\subsection*{Exorcism (Core)}
\textbf{Purpose:} Expel a possessing entity. \\
\textbf{Prereqs:} Spirit 3+, Lore 3+, Insight 2+. \quad \textbf{Team:} 1+ helpers (Lore/Spirit/Insight/Command/Presence). \\
\textbf{Time:} Significant Time (phased). \quad \textbf{Duration:} Encounter.\\
\textbf{Structure:} \emph{Contested clock} = \(\max(4,\min(8, 2 + \text{Subject Tier}))\). Each \textbf{segment} resolves three beats:
\begin{enumerate}
    \item \textbf{Invocation:} Spirit+Lore vs DV (Tier).
    \item \textbf{Confrontation:} Presence+Insight \emph{or} Command vs DV (Tier).
    \item \textbf{Binding:} Wits + relevant (Tinker/Arcana/Rites) vs DV (Tier).
\end{enumerate}
\textbf{Effect (Success):} Three consecutive beats exceeding DV by 2+ across segments expel the entity; subject stabilises (Harm 2 care); participants mark Fatigue 1 per entity Tier. Mark 1 XP for major spiritual conflict.\\
\textbf{Failure:} Clock fills $\rightarrow$ exorcism fails; entity empowered (+1 die on future possessions), subject worsens (Harm +1 level or Corruption), all participants Fatigue 2; start \emph{Haunting [6]}.\\
\textbf{Backlash (SB):} 1--2 SB minor phenomena; 3--4 SB splintering influence (bank +2 SB later); 5+ SB possession flares (impose Desperate next beat).

\subsection*{Cleansing Fire}
\textbf{Purpose:} Purify curses/contamination; weaken hauntings.\\
\textbf{Prereqs:} Spirit 4+, Arcana 3+, Lore 2+. \quad \textbf{Time:} Significant Time. \quad \textbf{Duration:} Immediate; site/object clean for 1 month.\\
\textbf{Test:} Spirit+Lore (prep) then Presence+Arcana to enact, DV 3--4 (by strength).\\
\textbf{Effect:} Remove one curse/taint; beneficiaries gain +1 die to resist similar effects for a scene. If target possessed, entity suffers --1 die this scene.\\
\textbf{Backlash:} 1--2 SB partial; returns at half strength next day. 3--4 SB agitation (+2 SB next encounter). 5+ SB inward scorch (caster Harm 1; minor fire‑themed flaw).

\subsection*{Bell of Truce}
\textbf{Purpose:} Compel parley before violence.\\
\textbf{Prereqs:} Presence 4+, Spirit 3+, Command 2+. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Some Time (~30 min).\\
\textbf{Test:} Spirit+Command then Presence+Performance, DV 3.\\
\textbf{Effect:} Sentients within ~200y must pass Spirit+Resolve (DV 3) to initiate violence. Early violators create +2 SB when violence starts; resisters gain +1 die to Sway with party.\\
\textbf{Backlash:} 1--2 SB weak tone (--1 die to Sway next social). 3--4 SB spillover (GM banks +2 SB in next crowd). 5+ SB perceived weakness (--2 dice to Command 1 session).

\subsection*{Mist‑Road Veil}
\textbf{Purpose:} Route‑bound concealment for stealthy travel.\\
\textbf{Prereqs:} Wits 3+, Arcana 2+, Survival 2+. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Significant Time (~1 hr).\\
\textbf{Test:} Wits+Survival (channel) then Wits+Arcana (weave), DV 3.\\
\textbf{Effect:} ~100y misted path; party gains +2 dice to Stealth on path; sound halved; mundane tracking nearly impossible.\\
\textbf{Backlash:} 1--2 SB patchy (--1 die to Stealth). 3--4 SB hunters notice (+2 SB next encounter). 5+ SB whiteout (--2 dice to navigation for a scene).

\subsection*{Lantern Weir}
\textbf{Purpose:} Calm/guide crowds; prevent panic surges.\\
\textbf{Prereqs:} Wits 3+, Arcana 3+, Performance 2+. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Significant Time (~1 hr).\\
\textbf{Test:} Wits+Performance then Presence+Arcana, DV 3.\\
\textbf{Effect:} 50‑ft soft light fence; +2 dice to crowd management; reduce panic effects by 2 SB inside; moonlight illumination.\\
\textbf{Backlash:} 1--2 SB flicker (--1 die to crowd control). 3--4 SB wrong attention (+2 SB next social). 5+ SB glare (--1 die to Perception nearby).

\subsection*{Oath of Shared Breath}
\textbf{Purpose:} Distribute Fatigue; blunt chip damage.\\
\textbf{Prereqs:} Presence 4+, Spirit 3+, Sway 2+; 2+ willing. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Significant Time (~1 hr).\\
\textbf{Test:} Spirit+Sway then Presence+Command, DV 2--3.\\
\textbf{Effect:} Pool Fatigue, divide evenly (round up). When one would take Harm 1, group may instead each take 1 Fatigue. Sense ally health + location (~100y).\\
\textbf{Backlash:} 1--2 SB uneven load (+1 Fatigue to one). 3--4 SB clingy bond (--1 die to social with outsiders this scene). 5+ SB overshare (unwanted empathic bleed).

\subsection*{Key of Locked Gates}
\textbf{Purpose:} Open magical seals/wards/locks.\\
\textbf{Prereqs:} Wits 4+, Arcana 4+, Tinker 2+. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Instant; bypass may persist 1 day.\\
\textbf{Test:} Wits+Arcana (analyse) then Presence+Tinker (unravel), DV 3--4 (by complexity).\\
\textbf{Effect:} Bypass one seal/ward; +2 dice to further lock/ward analysis rolls this session; may mint a one‑day passkey.\\
\textbf{Backlash:} 1--2 SB partial (relatches soon; --1 die to keep open). 3--4 SB alarms/wards trigger (+2 SB next area scene). 5+ SB containment breached (new threat enters play).

\subsection*{Stone's Memory}
\textbf{Purpose:} Read imprinted history of places/objects.\\
\textbf{Prereqs:} Spirit 4+, Insight 3+, Lore 2+. \quad \textbf{Time:} Some Time. \quad \textbf{Duration:} Scene (insights last session).\\
\textbf{Test:} Spirit+Insight (attune) then Wits+Lore (parse), DV 2--3.\\
\textbf{Effect:} Witness 1--3 significant echoes; +2 dice to related Investigation/Lore this session; detect current magical influence.\\
\textbf{Backlash:} 1--2 SB fragmented (--1 die to related Investigation this scene). 3--4 SB psychic residue (Fatigue 1; --1 die social for a scene). 5+ SB memory snare (outside aid required).

\subsection*{Mirror's Gaze}

\textbf{Effect:} Perceive target's senses; if willing, +1 die to Investigation/Stealth via their view; guide them remotely.\\

\section{Disabling Traps and Wards}\label{sec:disabling_wards}
\index{Traps}\index{Wards}

\textbf{Traps} are mechanical obstacles (barriers, Harm, Fatigue/poison). \textbf{Wards} are the magical analog (sensors, bindings, anchored effects). Resolve like any obstacle: state the fiction, set Position, assign DV, roll, spend SB.

\subsection*{At the Table: Fast Structure}
\begin{enumerate}[label=\textbf{Step~\arabic*:}, leftmargin=2.1em]
    \item \textbf{Name the Ward/Trap.} What it senses, what it does, and where it anchors.
    \item \textbf{Reveal Two+ Handles.} Always surface multiple approaches in fiction:
        \begin{itemize}
            \item \textbf{Arcane Pattern} (\emph{Wits+Arcana})
            \item \textbf{Mechanism/Anchor} (\emph{Wits+Tinker})
            \item \textbf{Pathing/Timing} (\emph{Body+Agility/Move})
        \end{itemize}
    \item \textbf{Set Position} then \textbf{Set DV (from access/pressure)}.
        \begin{itemize}
            \item \textbf{Position:} \emph{Dominant} (one fail die re‑roll), \emph{Controlled}, \emph{Desperate} (severe consequences).
            \item \textbf{Typical DV:} DV 2 (clear access, right tools), DV 3 (partial access, time pressure), DV 4--5+ (hostile field, active trigger).
        \end{itemize}
    \item \textbf{Declare Stakes.} If relevant, put a \textsc{Ward Clock} and an \textsc{Alert Clock} on the table.
\end{enumerate}

\subsection*{Common Ward Types \& Anchors}
\begin{tabularx}{\linewidth}{>{\bfseries}l X X}
\toprule
Type & Trigger & Anchor Examples \\
\midrule
Sentry & Threshold, line‑of‑sight, cadence & Lintel sigil, eye‑gem, bell filament \\
Binding & Touch item, break seal, oath‑breach & Wax seal, chain knot, iron brand \\
Punitive & Trespass + timer, named foe & Idol tooth, ash glyph under rug, curse knot \\
Displacement & Path inversion, gravity twist, dream‑step & Floor mandala, mirror post, sleep pin \\
Anti‑Magic & Casting in zone, symbol/pact use & Null plate, cold‑iron frame, void ampoule \\
\bottomrule
\end{tabularx}

\subsection*{Outcome Palette}
\begin{itemize}
    \item \textbf{Strong Hit:} Disabled or cleanly redirected. Optionally \emph{bank a tell} (you recognise this signature later).
    \item \textbf{Partial:} Suppressed but unstable; pick one: tick \textsc{Alerted Attention}+1; apply a small elemental/Patron backlash; leave a trace that can be read later.
    \item \textbf{Miss:} Ward holds or barks. GM may (a) advance \textsc{Ward Integrity} toward rupture, (b) trigger a reduced effect now, (c) spend SB for a hard complication.
\end{itemize}

\subsection*{Counter‑Ward (Arcana focus; optional)}
On a success choose: \textbf{Suppress} (one scene) \emph{or} \textbf{Disarm} (permanent; pay rare material or mark +1 Obligation) \emph{or} \textbf{Invert} (first trigger redirects to a safe sink; start \textsc{Unintended Echo} 1/4).

\subsection*{Difficulty Dials (GM)}
\begin{itemize}
    \item \textbf{Scale:} Vast/ancient lattice: +1 DV or add a second clock.
    \item \textbf{Haste:} Rushing worsens \emph{Position} one step.
    \item \textbf{Right of Names:} Knowing the warder's true name improves \emph{Position} one step \emph{or} grants +1d.
    \item \textbf{Opposed Patron:} Using hostile symbols imposes --1d \emph{or} +1 DV (table call).
\end{itemize}

\subsection*{Worked Mini‑Example}
\emph{Vault Sentry (DV 3, \textbf{Desperate}).} Anchor: eye‑gem set in an idol. Crew uses \emph{Wits+Tinker} with ward‑ink and a mirror shard. \textbf{Partial:} \textsc{Ward Integrity}+1, \textsc{Alerted Attention}+1; beam is shunted for this scene. Later, a rival recognises the mirror signature---trace pays off.

\section{Poison as Story Beats}\label{sec:poison_sb}
\index{Poison}\index{Story Beats}

\textbf{Principle:} Poison should move the scene forward, not just drain health. It begins with \textbf{Fatigue}, escalates via \textbf{SB}, and resolves through fiction.

\subsection*{Core Model}
\begin{itemize}
    \item Poison applies \textbf{Fatigue} first.
    \item GM may \textbf{spend SB} to escalate: Fatigue $\to$ Condition $\to$ Harm. Always narrate the change.
    \item Track with a \textsc{Poison Clock} (e.g. \emph{Neurotoxin Load} 1/4).
    \item Resist as normal; antidotes, masks, or purge rites improve Position or cancel ticks.
\end{itemize}

\subsection*{Outcomes}

\section{Boss Generator}\label{sec:boss_generator}
\index{Bosses!generator}

Bosses mirror player archetypes:
\begin{itemize}
    \item \textbf{Solo Boss (The Brute):} A singular, overwhelming threat with unique mechanics.
    \item \textbf{Mixed Boss (The Commander):} A strong figure who empowers and directs minions.
    \item \textbf{Mastermind Boss (The System):} A schemer, institution, or environment that opposes players through assets and schemes.
\end{itemize}

\subsection{Deck Structure}
\paragraph{Standard Deck (52 Cards)}
\textbf{Suits Define Core Aspects:}
\begin{itemize}
    \item \textbf{♠ Spades (Structure):} Body, form, core mechanics.
    \item \textbf{♥ Hearts (Drive):} Motivations, goals, psychology.
    \item \textbf{♣ Clubs (Complication):} Hazards, minions, environmental twists.
    \item \textbf{♦ Diamonds (Reward):} Loot, knowledge, or narrative shifts.
\end{itemize}
\textbf{Ranks Define Scale/Intensity:}
\begin{itemize}
    \item \textbf{2--5 (Minor):} Elite foe. 4‑segment Phase Clock.
    \item \textbf{6--10 (Standard):} Significant boss. 6‑segment Phase Clock.
    \item \textbf{J/Q/K (Major):} Set‑piece villain. 8‑segment Phase Clock.
    \item \textbf{A (Pivotal):} Campaign‑defining threat. 10‑segment Phase Clock.
\end{itemize}

\subsection{Boss Archetypes}
\paragraph{1. The Solo Boss (Brute)}
\begin{itemize}
    \item \textbf{Focus:} Raw presence, unique mechanics, escalating phases.
    \item \textbf{Structure (♠):} Defines resistances, body, attack type.
    \item \textbf{Drive (♥):} Shapes targeting and tactics.
    \item \textbf{Complication (♣):} Ongoing hazard or condition.
    \item \textbf{Reward (♦):} Narrative payoff, artifact, or twist.
\end{itemize}
\emph{Example:} ♠A Titan, ♥J Vengeance, ♣10 Phased Attacks, ♦Q Command Obedience → a colossal vengeful giant who phases through attacks, defeated only to yield dominion over its kin.

\paragraph{2. The Mixed Boss (Commander)}
\begin{itemize}
    \item \textbf{Focus:} A dangerous leader plus followers. Defeating them requires managing both.
    \item \textbf{Structure (♠):} Body plus a \emph{Minion Clock}. Followers act each round unless disrupted.
    \item \textbf{Drive (♥):} Directs how the leader uses minions.
    \item \textbf{Complication (♣):} Minions escalate scene hazards.
    \item \textbf{Reward (♦):} Control of followers, faction leverage, or treasure guarded by both.
\end{itemize}
\emph{Example:} ♠6 Swarm Core, ♥K Domination, ♣J Minion Control, ♦10 Faction Favor → a commander directing a swarm; defeating them secures the loyalty of a faction.

\paragraph{3. The Mastermind Boss (Systemic)}
\begin{itemize}
    \item \textbf{Focus:} A schemer, institution, or environment resolved through multiple scenes.
    \item \textbf{Structure (♠):} Organisation or environment. Represented by multiple linked clocks.
    \item \textbf{Drive (♥):} The villain's overarching scheme.
    \item \textbf{Complication (♣):} Ongoing schemes/assets---rival factions, crises, economic pressure.
    \item \textbf{Reward (♦):} Narrative leverage: reform, control, or expose the system.
\end{itemize}
\emph{Example:} ♠Q Hive Mind, ♥A Cosmic Hunger, ♣K Reality Distortion, ♦A Worldly Truth → a cult network feeding a god‑fragment, warping reality until the players dismantle the cult's assets.

\subsection{GM Procedure}
\begin{enumerate}
    \item \textbf{Choose Archetype:} Solo, Mixed, or Mastermind.
    \item \textbf{Draw Four:} One per suit. Highest rank = Phase Clock size.
    \item \textbf{Interpret:} Combine Structure, Drive, Complication, Reward.
    \item \textbf{Mechanise:}
        \begin{itemize}
            \item \textbf{Solo:} Pure mechanics and hazards.
            \item \textbf{Mixed:} Add Minion Clock(s).
            \item \textbf{Mastermind:} Use Scheme/Asset clocks, faction moves.
        \end{itemize}
    \item \textbf{Phase \& Escalate:} Break into 2--3 phases. Increase dice pools or trigger complications.
    \item \textbf{Tie to Story Beats:} SB fuel Drives and Complications---rage, reinforcements, systemic backlash.
\end{enumerate}

\subsection{Terrestrial Patrons}
Not all patrons are gods, demons, or cosmic forces. Mortals create power too: nobles, guilds, conspiracies, temples, syndicates, and commanders. A Terrestrial Patron represents an ongoing relationship with a powerful mortal faction.

\begin{description}
    \item[\textbf{Why They Matter}] A terrestrial patron doesn"t grant magic. They grant \emph{leverage}: protection, resources, sanctuary, information, and political shifts.
    \item[\textbf{Obligation (Terrestrial)}] Use the same Obligation track, but consequences are social, legal, or economic instead of supernatural. When you call on a Patron's influence, add +1 Obligation.
    \item[\textbf{Getting a Patron}] Complete one of: a major job, a sworn Oath, legal/financial binding, blackmail or shared crime. Mark them on your sheet and write one sentence: "They want me because \_\_\_"
    \item[\textbf{Perks}] Each Patron offers 2--3 repeatable benefits (sanctuary, legal relief, black‑market goods, elite followers, forged documents, military backing, rumours). Using a Perk never requires a roll.
    \item[\textbf{Demands}] Terrestrial Patrons always want something back: silence, loyalty, a job, a name, a secret. Refusing raises Obligation by 1. Betrayal may have immediate consequences.
    \item[\textbf{When Obligation Fills}] At 6 Obligation, the Patron acts. Choose one: you do a job you cannot refuse; you pay a severe price; they strike first. Reduce Obligation to 3 after the consequence lands.
    \item[\textbf{Cutting Ties}] Lose all current perks, gain a new Rival faction, take a Curse, Bounty, or Scandal that follows you.
    \item[\textbf{Redemption or Favor}] A monumental service reduces Obligation by 2 and grants a permanent Favor: title, land, permanent access, unique asset.
\end{description}

\subsection*{Quick Example}
The Black Ledger smuggling syndicate gives sanctuary and illegal gear. Rellan calls on the Ledger for a smuggled border crossing. The GM rules it succeeds automatically, but adds +1 Obligation. Rellan now owes the Ledger. Later, the Ledger demands he silence a witness. If he refuses, Obligation rises again. If Obligation ever reaches 6, the Ledger collects: accounts frozen, bounty posted, or a rival informant sent after him.

\chapter{Core Mechanics}\label{chap:core_mechanics}

\section{The Core Resolution Cycle}\label{sec:resolution_cycle}

When a player rolls dice, they engage the world through risk, consequence, and discovery.

\subsection{Step‑by‑Step Resolution}\label{subsec:steps}

\begin{enumerate}
    \item \textbf{Declare Action \& Approach:} Intent + Attribute + Skill.
    \item \textbf{Set Difficulty Value (DV):} GM sets DV (2--7+) based on stakes.
    \item \textbf{Roll Dice Pool:} Attribute + Skill d10s (max 10 dice; extras = auto-successes).
    \item \textbf{Count Results:}
        \begin{itemize}
            \item 6--9 = 1 success
            \item 10 = 2 successes
            \item 1 = 1 Story Beat (SB) for GM
        \end{itemize}
    \item \textbf{Compare to DV} → Outcome Matrix (see \autoref{tab:outcome_matrix}).
    \item \textbf{Spend SB:} GM introduces complications (see \autoref{sec:sb_spend_menu}).
\end{enumerate}

\subsection{Difficulty Value (DV)}\label{subsec:difficulty_value}

\begin{tcolorbox}[title=Default DV Ladder, colback=white!97!gray, colframe=black!80!gray, sharp corners]
\begin{center}
\begin{tabular}{c p{3.5in}}
\toprule
\textbf{DV} & \textbf{Typical Case} \\
\midrule
2 & Routine: clear intent, modest stakes, controlled environment \\
3 & Pressured: time limits, mild resistance, partial information \\
4 & Hard: hostile conditions, active opposition, precision required \\
5+ & Extreme: stacked constraints, dangerous failure, high drama \\
\bottomrule
\end{tabular}
\end{center}
\end{tcolorbox}
Default DV = 3.

\subsection{Position \& Effect}\label{subsec:position_effect}
\begin{description}
    \item[Position (risk)] Dominant (safe), Controlled (balanced), Desperate (high stakes).
    \item[Effect (reward)] Limited, Standard, Great.
\end{description}
GM sets both based on fiction.

\subsection{Critical Success (10s)}\label{subsec:critical_success}
If roll succeeds, each 10 adds narrative weight. Effects apply only on success.
\begin{itemize}
    \item \textbf{1 ten:} Strong success -- improved Position or minor advantage.
    \item \textbf{2 tens:} Exceptional -- choose two benefits (e.g., increased Effect + free Boon).
    \item \textbf{3 tens:} Legendary -- decisive victory; clear 1 clock segment.
    \item \textbf{4+ tens:} Mythic -- reshape scene; clear 1--2 clock segments.
\end{itemize}
Tens are never rerolled by Position effects.

\subsection{Dice Pool Limit \& Auto‑Successes}\label{sec:dice_pool_cap}
Max dice = 10d10. Extra dice become auto-successes (no SB generated).

\subsection{Boons}\label{subsec:boon_economy}
\paragraph{Earning Boons}
\begin{itemize}
    \item Partial (0 < successes < DV): 1 Boon
    \item Miss (successes = 0): 2 Boons (meaningful roll only)
    \item Bond‑driven action (once per bond/session): 1 Boon
    \item GM award
\end{itemize}
\noindent \textbf{Limit:} Max 2 Boons from failures per character per scene.

\paragraph{Spending Boons (1 Boon)}
\begin{itemize}
    \item Reroll one die (after result)
    \item Activate an on‑screen Asset
    \item Improve Position by one step before roll
    \item Power certain Rites/abilities
    \item Convert 2 Boons $\rightarrow$ 1 XP (once/session downtime, max 2 XP)
\end{itemize}
\paragraph{Carryover:} Max 5 Boons; at scene end, reduce to 2.

\subsection{Story Beats (SB)}\label{subsec:story_beats}
Each die showing 1 = 1 SB for GM. Rerolling a 1 does not erase its SB; another 1 adds extra.

\paragraph{SB Spend Menu}\label{sec:sb_spend_menu}
\begin{tabular}{ll}
\toprule
\textbf{Cost} & \textbf{Effect} \\
\midrule
1 SB & Minor: noise, trace, tick clock +1, worsen Position \\
2 SB & Moderate: alarm raised, lose cover/Position, lesser foe appears \\
3 SB & Serious: reinforcements, key gear breaks, major twist, advance front clock +1 \\
4+ SB & Major: trap springs, authority arrives, scene shift, Patron intervenes \\
\bottomrule
\end{tabular}

\section{Player‑Managed Modules}\label{sec:player_managed_modules}
Players track their own: Obligation, Corruption, Leash, Asset States.

\begin{tcolorbox}[title=What Players Track, colback=black!3, colframe=black!40!white]
\begin{tabular}{lll}
\toprule
\textbf{Module} & \textbf{Owner} & \textbf{Tick Triggers (examples)} \\
\midrule
Obligation (per Patron) & Player & Invoke Rite, Push, Crack the Seal \\
Corruption Clock & Player & Cantor Push, Resonant Rite, GM SB on occult actions \\
Leash (Summoning) & Player & Spirit takes Harm, acts against nature, crosses [WARD] \\
Asset State (Symbol) & Player & Neglect, Crack the Seal $\rightarrow$ Compromised \\
\bottomrule
\end{tabular}
\end{tcolorbox}

\paragraph{Obligation Capacity} = Spirit + Presence.
\begin{itemize}
    \item Over capacity: +1 Fatigue per segment.
    \item Double capacity: Clear Fatigue, +1 Harm, Patron Intrusion.
\end{itemize}
\paragraph{Corruption Full} Apply last Patron's boon & burden, reset to 0.
\paragraph{Leash Full} Spirit acts once to its nature then departs/turns hostile.
\paragraph{Symbol State} Compromised -- repair downtime (DV 3) or 1 XP.

\section{Turn Economy}\label{sec:turn_economy}
Each turn: \textbf{1 Action + 1 Move}.

\subsection{Movement}
\begin{itemize}
    \item \textbf{Move:} Shift one band (Close $\leftrightarrow$ Near $\leftrightarrow$ Far) or stand from prone.
    \item \textbf{Dash (Action):} Move two bands; next defense Desperate.
    \item \textbf{Disengage (Action):} Then Move does not provoke opportunity attacks; cannot Attack this turn.
\end{itemize}

\subsection{Action Options}
\begin{itemize}
    \item \textbf{Attack} (melee/ranged)
    \item \textbf{Disengage} (as above)
    \item \textbf{Charge (full turn):} Move two bands toward enemy + melee attack. Costs 1 Fatigue. Next defense Desperate. Requires open ground.
    \item \textbf{Skirmish (full turn):} Move two bands away + ranged attack. Costs 1 Fatigue. Next defense Desperate. Requires open ground.
    \item \textbf{Observe / Change Position:} +1 Position once per turn.
    \item \textbf{Setup:} Grant ally +1 Position or step up Effect.
    \item \textbf{Assist:} Spend 1 Boon for +1 die (max +3 from all assists).
    \item \textbf{Defend, Channel/Weave, Cast Rite/Song, Interact,} etc.
\end{itemize}
Talents (Effortless Charge, Relentless, etc.) modify these.

\subsection{Initiative}
No fixed order. GM spotlights based on fiction; ambushers act first.

\subsection{Opportunity Attacks}
Leaving an enemy's Close without Disengage provokes one free melee attack (Controlled Position, no action cost).

\section{Fatigue \& Harm}\label{sec:fatigue_harm}

\subsection{Fatigue Track}
Size = Body. Mark from exertion, magic, stress.

\paragraph{Effects}
\begin{itemize}
    \item Each Fatigue worsens Position: Dominant $\rightarrow$ Controlled $\rightarrow$ Desperate.
    \item If already Desperate, each Fatigue → --1 die.
    \item \textbf{Overflow:} When track fills, increase Harm by 1 level, clear all Fatigue.
\end{itemize}
\paragraph{Recovery:} Short rest removes 2 Fatigue; full night removes all. Requires adequate Supply.

\subsection{Harm Levels}
\begin{tabular}{ll}
\toprule
\textbf{Harm} & \textbf{Effect} \\
\midrule
Harm 1 & --1 die on related actions \\
Harm 2 & --1 die on most actions until treated \\
Harm 3 & Incapacitated or dying \\
\bottomrule
\end{tabular}
Recovery: medical or magical care.

\subsection{Taking Harm \& Fatigue -- The Procedure}
When hit:
\begin{enumerate}
    \item \textbf{Convert} Harm to Fatigue using armor (see \autoref{sec:armor_conversion}). Minimum 1 Fatigue per hit.
    \item \textbf{Apply remaining Harm:} If any, increase Harm level and clear all existing Fatigue.
    \item \textbf{Add Fatigue} from step 1. If track fills, resolve overflow (Harm +1, clear Fatigue).
\end{enumerate}

\subsection{Armor Conversion Table}\label{sec:armor_conversion}
\begin{tabular}{lll}
\toprule
\textbf{Armor Type} & \textbf{Conversion (Harm → Fatigue)} & \textbf{Minimum Fatigue} \\
\midrule
Light  & 1 → 1 & 1 \\
Medium & 2 → 1 & 1 \\
Heavy  & 3 → 2 & 1 \\
\bottomrule
\end{tabular}
\emph{Example:} Light armor hit with Harm 2 → convert 1→1 Fatigue, remaining Harm 1 → increase Harm 1, clear Fatigue, then add 1 Fatigue. Result: Harm 1, Fatigue 1.

\subsection{Fear Effects}
When Fear Track escalates (Shaken → Frightened → Panicked), roll d10 or choose:
\begin{center}
\begin{tabular}{cl}
\toprule
d10 & Effect \\
\midrule
1 & Freeze (cannot act) \\
2 & Flee \\
3 & Drop held item \\
4 & Beg/bargain \\
5 & Hide \\
6 & Panic attack (nearest target) \\
7 & Blunder into danger \\
8 & Obey simple command \\
9 & Break down (useless until aided) \\
10 & Catatonia \\
\bottomrule
\end{tabular}
\end{center}

\section{Experience Points (XP)}\label{sec:xp}

\subsection{Awarding XP (per session)}
\begin{itemize}
    \item Attendance: +2
    \item Major objective: +2--4
    \item Discoveries: +1--2
    \item Hard choices: +1--2
    \item Complication spotlight: +1--3
    \item Bond/flag driven: +1--2
    \item GM curveball: +0--3
\end{itemize}
\noindent End of major arc: +8--12 XP to all.

\subsection{Spending XP}
\begin{itemize}
    \item \textbf{Attribute:} new rating ×3 XP, downtime = new rating days.
    \item \textbf{Skill:} new level ×2 XP, downtime = new level days.
    \item \textbf{Talents:} 2--12+ XP per tier.
    \item \textbf{Follower:} Cap² XP, downtime 1--3 days.
    \item \textbf{Asset:} Minor (4 XP, 1d), Standard (8 XP, 1w), Major (12 XP, 1 month).
\end{itemize}
\paragraph{Rush Rule:} Skip downtime → start Haste [4] clock. On fill, ability gains flaw/hitch.

\subsection{Reputation Tiers}
\begin{description}
    \item[Novice (0--40 XP)] Local, prestige locked.
    \item[Seasoned (41--90 XP)] Regional, some prestige unlock.
    \item[Veteran (91--150 XP)] National, second follower slot.
    \item[Paragon (151--220 XP)] Continental, rivals emerge.
    \item[Mythic (221+ XP)] Legendary, kingdoms react.
\end{description}

\section{Game Structure}\label{sec:game_structure}
\subsection{Narrative Time}
\textbf{A Moment} (heartbeat), \textbf{Some Time} (minutes), \textbf{Significant Time} (hours), \textbf{Days}.

\subsection{Game Units}
\textbf{Scene} (conflict/question), \textbf{Session} (3--6 hrs, 2--4 scenes), \textbf{Campaign} (6--20+ sessions).

\section{Assets \& Followers (Quick Reference)}\label{sec:assets_followers}

\subsection{Assets (Off‑Screen)}
Costs: Minor (4 XP, 1d), Standard (8 XP, 1w), Major (12 XP, 1m).
\begin{itemize}
    \item \textbf{Free off‑screen use:} once/session.
    \item \textbf{Boon activation:} 1 Boon for on‑screen scene.
    \item \textbf{Conditions:} Maintained $\rightarrow$ Neglected $\rightarrow$ Compromised.
\end{itemize}

\subsection{Followers (On‑Screen)}
Cost: Cap² XP, downtime 1--3d.
\begin{itemize}
    \item \textbf{Assistance:} +min(Cap, relevant skill) dice, capped at +3 total.
    \item \textbf{Independent action:} once/scene (party‑wide). Cost: 1 Exposure or 1 Harm.
    \item \textbf{Conditions:} Exposure, Harm tracks.
\end{itemize}

\subsection{Upkeep (per downtime)}
\begin{itemize}
    \item \textbf{Efficient:} XP = max(1, acquisition/3).
    \item \textbf{Intensive:} 1 XP + dedicated downtime action.
\end{itemize}
Failure to pay: follower → Wary → Seized; asset → Neglected → Compromised.

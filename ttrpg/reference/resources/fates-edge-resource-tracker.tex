\newcommand{\trackerbox}[1]{%
  \begin{tcolorbox}[colback=white, colframe=fe-dusk, sharp corners, boxrule=0.8pt, width=0.9\linewidth, boxsep=2pt, left=4pt, right=4pt, top=2pt, bottom=2pt,breakable]
    #1
  \end{tcolorbox}
}

\newcommand{\segmenttracker}[3]{%
  \begin{tikzpicture}
    \foreach \i in {1,\dots,#1} {
      \filldraw[fill=white, draw=#2, thick] (\i*0.75cm,0) circle (0.35cm);
      \ifnum\i<#1
        \draw[#2, thick] (\i*0.75cm+0.35cm,0) -- (\i*0.75cm+0.75cm-0.35cm,0);
      \fi
    }
    \node[anchor=west, font=\small\bfseries, color=#2] at (#1*0.75cm+0.5cm,0) {#3};
  \end{tikzpicture}%
}

\newcommand{\boontracker}{%
  \begin{tikzpicture}
    \foreach \i in {1,\dots,5} {
      \filldraw[fill=white, draw=fe-fate, thick] (\i*0.6cm,0) circle (0.3cm);
    }
    \node[anchor=west, font=\small\bfseries, color=fe-fate] at (3.5cm,0) {Boons (max 5)};
  \end{tikzpicture}%
}

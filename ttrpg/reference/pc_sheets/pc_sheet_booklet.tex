\section*{Character Creation Walkthrough}
\addcontentsline{toc}{section}{Character Creation Walkthrough}

\begin{tipbox}{Before You Begin: The Philosophy}
    \textbf{\textit{``Do not try to build a 'perfect' character. Build one who owes a debt, carries a scar, and has a name that someone will remember. The numbers will follow the story---not the other way around.''}}
    \hfill --- The Gray Wanderer
    
    \vspace{0.5em}
    \textbf{Key principles to embrace:}
    \begin{itemize}
        \item \textbf{Flavor is free} — describe your character however you want; mechanics don’t limit creativity.
        \item \textbf{Boons are not XP tokens} — spend them freely; hoarding starves the table of drama.
        \item \textbf{Embrace failure} — partials and misses make the story interesting.
        \item \textbf{You manage your own resources} — track your own Obligation, Fatigue, Boons.
    \end{itemize}
\end{tipbox}

\subsection*{Step 1: Concept Seed (5 minutes)}
\begin{enumerate}
    \item \textbf{Who are you?} (ancestry, origin)
    \item \textbf{What do you do?} (profession, role)
    \item \textbf{What defines you?} (a trait, a debt, a scar)
\end{enumerate}

\begin{charbox}{Sample Concepts}
    \begin{tabular}{p{3cm} p{9cm}}
        \toprule
        \textbf{Concept} & \textbf{Sentence} \\
        \midrule

        Hedge Witch & ``An Aelaerem village healer who learned magic from her grandmother---and inherited her debts.'' \\
        Runekeeper & ``An Aeler tunnel‑wright who broke a seal in the deep ways and now serves the Patron who answered.'' \\

\end{charbox}

\textbf{Sample Build (Sage):} \textit{``A Vhasian duelist who lost his family's charter in a rigged trial. He carries his father’s sword and a vow to restore his house’s name---or die trying.''}

\begin{tipbox}{Core Sheet — First Steps}
    \textbf{Start here.} This sheet works for any character. Fill out Identity, Attributes, Skills, then track resources.
    
    \textbf{Key Choices:}
    \begin{itemize}
        \item \textbf{Primary Attribute:} Body (melee), Wits (rogue/caster), Spirit (magic), Presence (face)
        \item \textbf{Key Skills:} Choose 2‑3 at level 2, rest at 0‑1
        \item \textbf{Talents:} 1‑3 talents (2‑6 XP each)
        \item \textbf{Bonds:} Take 2 bonds with other PCs (+4 XP total)
    \end{itemize}
    
    \textbf{Resource Management:}
    \begin{itemize}
        \item \textbf{Boons:} Max 5; trim to 2 at scene end. Spend freely!
        \item \textbf{Fatigue:} Each point worsens Position. Full track → Harm + 1
        \item \textbf{Obligation:} Track per Patron; Capacity = Spirit + Presence
    \end{itemize}
\end{tipbox}

\section*{Melee Combat Specialist --- Build Guidance}
\addcontentsline{toc}{section}{Melee Combat Specialist --- Build Guidance}

\begin{tipbox}{The Tank / Striker --- Frontline Combat}
    \textbf{Role:} Stand in the shield wall, absorb harm, deal damage. Primary: Body.
    
    \textbf{Core Talents (Must Have):}
    \begin{itemize}
        \item \textbf{Weapon Mastery (5 XP)} -- +2 dice with chosen weapon
        \item \textbf{Disciplined Body (3 XP)} -- Convert 1 Harm to Fatigue, +1 die unarmed
        \item \textbf{Second Wind (2 XP)} -- Clear 1 Fatigue
    \end{itemize}
    
    \textbf{Progression Path:}
    \begin{itemize}

    \begin{charbox}{Freecaster Talents}
        \textbf{Spellcraft (6 XP):} Required to free cast.\\
        \textbf{Controlled Burn (4 XP):} Once/scene, convert a 1 into 1 Boon (GM does not gain SB).\\
        \textbf{Weave Resilience (4 XP):} Reduce first Backlash each scene by one step.\\
        \textbf{Elemental Attunement (5 XP):} Choose an element; reroll one failed die per casting.\\
        \textbf{Spell Echo (8 XP):} Once/session, reroll a 1 on a casting roll.
    \end{charbox}

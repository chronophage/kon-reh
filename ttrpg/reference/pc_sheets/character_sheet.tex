% JavaScript for auto-calculation (must be at start of document)
\begin{Form}[action={% Auto-calculate on load
    var attrs = ['body','wits','spirit','presence'];
    var skillNames = ['melee','ranged','brawl','tactics','athletics','stealth','endurance','craft','survival','sway','command','deception','performance','insight','lore','investigation','medicine','arcana','ritual'];
    var attrFields = {};
    var skillFields = {};
    
    function attrCost(r) {
        if (r <= 1) return 0;
        return 3 * (r * (r + 1) / 2 - 1);
    }
    
    function skillCost(l) {
        if (l <= 0) return 0;
        return 2 * (l * (l + 1) / 2 - 1);
    }
    
    function calculateAll() {
        // Calculate Attributes
        var total = 0;
        for (var i = 0; i < attrs.length; i++) {
            var val = parseInt(this.getField('attr_' + attrs[i]).value);
            var cost = attrCost(val);
            total += cost;
            this.getField('cost_attr_' + attrs[i]).value = cost;
        }
        this.getField('attr_total').value = total;
        
        // Calculate Skills
        var skillTotal = 0;
        for (var j = 0; j < skillNames.length; j++) {
            var val = parseInt(this.getField('skill_' + skillNames[j]).value);
            var cost = skillCost(val);
            skillTotal += cost;
            this.getField('cost_skill_' + skillNames[j]).value = cost;
        }
        this.getField('skill_total').value = skillTotal;
    }
    
    this.calculateAll = calculateAll;
}]}

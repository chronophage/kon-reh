% CAMPAIGN HOOKS
% ======================
\sectionheader{Campaign Hooks}
\begin{enumerate}[leftmargin=*]
\item \textbf{Dean's Quiet Panic} — Recover 3 ``misfiled'' theses (they're people).\\
\emph{Starter:} Containment Breach +1. \emph{Complication:} Disavowed if it's loud.
\item \textbf{The Coughing Manuscript} — Track a soot-coughing book to its shelf.\\
\emph{Starter:} Knowledge Corruption +1. \emph{Complication:} It knows your name.
\item \textbf{Missing Custodian Cart} — Find a Bone Custodian's cart of keys.\\
\emph{Starter:} Wandering every Turn. \emph{Complication:} Vertebrae argue.
\item \textbf{Inheritance Clause} — Retrieve/destroy a sealed question.\\
\emph{Starter:} Obligation (Library) +1. \emph{Complication:} The question is you.
\item \textbf{A Scholar's Last Route} — Map 3-node safe path for first-years.\\
\emph{Starter:} Library starts hostile. \emph{Complication:} Your map corrects you.
\item \textbf{The Failing Annotation} — Stabilize a loved one bound in pages.\\
\emph{Starter:} Annotation Fade [4]=1. \emph{Complication:} They ask not to save them.
\item \textbf{The Choir Without Voices} — Prove a sect's silencing with a Memory Core.\\
\emph{Starter:} First Echo hostile. \emph{Complication:} They offer you silence.
\item \textbf{Grant Funding} — Secure a marketable artifact for a donor.\\
\emph{Starter:} Rivals one layer ahead. \emph{Complication:} Donor backs rivals too.
\item \textbf{The Syllabus War} — Arbitrate a curriculum via Codex clause.\\
\emph{Starter:} Grey Scribe tails a PC. \emph{Complication:} Clause wants a living cite.
\item \textbf{The Door That Finds You} — The door appears wherever you sleep.\\
\emph{Starter:} Library Reach 1. \emph{Complication:} Friends dream floorplans.
\item \textbf{Paper Storm} — Fetch Sealed Sigil to shield the city archive.\\
\emph{Starter:} Breach +2 in 24h. \emph{Complication:} Sigil flags a PC as contraband.
\item \textbf{Immortality Buyer} — Deliver ``proof'' (or lie) and live.\\
\emph{Starter:} Rumor Heat [6]=2. \emph{Complication:} Buyer is their own Annotation.
\end{enumerate}

\sectionheader{Side NPCs}

\begin{npcbox}
\textbf{The Witness (Curator of Containment)}\\
\emph{Demeanor:} Precise, weary. \emph{Leverage:} Halt a scene's time one exchange.\\
\emph{Offer:} Bind your truth to spare you. \emph{Secret:} Lied about immortality.\\
\emph{Move:} Re-index the room (reposition all). \emph{Clock:} Patience [6].\\
\emph{Tags:} Neutral Lawful; Cannot Lie; Hates Waste.
\end{npcbox}

\begin{npcbox}
\textbf{The Hollow Librarian (Echo of the Ancient)}\\
\emph{Demeanor:} Kind, dusty rain. \emph{Leverage:} Trade memory for shortcut.\\
\emph{Offer:} Safe passage for a regret. \emph{Secret:} Wants unwriting.\\
\emph{Move:} Helpful mislabel (truth later, cost now). \emph{Clock:} Mercy Owed [4].\\
\emph{Tags:} Ancient; Gentle; Dangerous if cornered.
\end{npcbox}

\begin{npcbox}
\textbf{The Void Keeper (Hand of the Ninth)}\\
\emph{Demeanor:} Gaps and wind. \emph{Leverage:} Remove 1 harmful tag (erase a bond).\\
\emph{Offer:} Erase debt, enemy, or you. \emph{Secret:} Bound by hospitality.\\
\emph{Move:} Null Glyph (1s count twice). \emph{Clock:} Unwriting Tide [6].\\
\emph{Tags:} Patient; Implacable; Polite.
\end{npcbox}

\begin{npcbox}
\textbf{Professor Isola Thren (Echo Scholar)}\\
\emph{Demeanor:} Bright eyes, ink nails. \emph{Leverage:} Route to Heart (for your middle name).\\
\emph{Offer:} Exchange one truth each. \emph{Secret:} An Annotation in a body.\\
\emph{Move:} Footnote a PC (can be ``referenced'' once). \emph{Clock:} Loaned Breath [3].\\
\emph{Tags:} Charming; Transactional; Homesick.
\end{npcbox}

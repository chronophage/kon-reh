\section{The Banner Swap in the Wind}
\emph{A Black Banners Adventure for Fate's Edge (no gunpowder)}

\subsection*{Adventure Overview}
\begin{tabular}{@{}ll@{}}
\textbf{Tier:} & II--III \\
\textbf{Length:} & 2--3 sessions \\
\textbf{Theme:} & Banner politics, village defense, condotta manipulation \\
\textbf{Hook:} & A Vilikari village caught between two rival banner companies offers the party a chance \\
& to play the companies against each other for the village's survival.
\end{tabular}

\subsection*{Setup: The Village of Saltmere}
Saltmere sits in a strategic valley where two trade routes converge. The \textbf{Iron Lanterns} (siege specialists who promise ``no fire in hospitals'') and the \textbf{Thornwrights} (sappers/road-cutters who price bridges by the breath) hold \emph{conflicting charters} to secure the valley. Village elders approach the PCs to manipulate both banners into terms that spare Saltmere.

\paragraph{Saltmere at a Glance}
\begin{itemize}[leftmargin=*]
  \item \textbf{Assets:} Winter granaries, shrine-hospital (old stone), river ford, timberyards, waymarket green.
  \item \textbf{Strings:} Ledger of Small Kindnesses (debts owed), ford-right witness-stone, bell-token set (for parley).
  \item \textbf{Clocks:} \emph{Granary Seizure [4]}, \emph{Shrine Fire Rumors [4]}, \emph{Village Panic [6]}.
\end{itemize}

\subsection*{The Companies \& What They Want (concrete aims)}
\begin{multicols}{2}
\subsubsection*{Iron Lanterns --- Siege Specialists}
\textbf{Captain Thorne Ironwright} (methodical, honor-bound) seeks:
\begin{itemize}[leftmargin=*]
  \item \textbf{Garrison Charter:} permanent \emph{winter garrison} with stipend.
  \item \textbf{Sanctuary Credit:} public recognition of Lantern hospital rites.
  \item \textbf{Bridge Priority:} stamp rights if siege breaks a crossing.
\end{itemize}
\textbf{Quartermaster Bray} wants predictable \emph{Payday} and \emph{rations}.
\columnbreak
\subsubsection*{Thornwrights --- Sappers \& Road-Cutters}
\textbf{Captain ``Cut'' Marrick} (reckless innovator) seeks:
\begin{itemize}[leftmargin=*]
  \item \textbf{Demolition Writs:} license to \emph{fire-set \& wedge} old footings.
  \item \textbf{Reconstruction Fees:} toll schedule until \emph{the bridge is paid}.
  \item \textbf{Exclusive Road Works:} first refusal on all valley roads.
\end{itemize}
\textbf{Engineer Tessa Chain} wants materials and \emph{no audits}.
\end{multicols}

\subsection*{Articles of the Camp (examples \& violations)}
\begin{center}\small
\begin{tabular}{p{0.44\textwidth}p{0.5\textwidth}}
\toprule
\textbf{Article (exemplar text)} & \textbf{Typical Violation / Evidence} \\
\midrule
No open flame nor fire-setting within twenty paces of a sanctuary. & Char streaks, heat-stress in stone, pitch jars and braziers in shrine yard. \\
Take no road you will not defend. & Abandoned picket, unpaid tolls during raid, missing watch roster. \\

\textit{Article Cite (mechanical):} Citing a relevant Article under witness grants \textbf{Position +1} for that parley; presenting \emph{material evidence} adds \textbf{DV --1}.

\subsection*{Session Structure \& Pacing}
\paragraph{Act I --- Council (45--60 min).} Frame crisis, place clocks, hand out charter cards.
\paragraph{Interlude Orders (10--15 min).} Each PC may take \emph{1 Between-Scene Order}: \emph{Scout}, \emph{Audit}, \emph{Canvas}, or \emph{Secure Proof}. Resolve with a single roll; deliver leads as cards.
\paragraph{Act II --- Manipulations (60--90 min).} Run Parley, Debts, False Orders, and/or \emph{Forgery Mini-Score} (below).
\paragraph{Act III --- Crisis \& Charter Endgame (45--75 min).} Roll Allegiances-in-Flux once mid-talks; resolve a \emph{Synthesis}, \emph{Gambit}, or \emph{Third Way}.

\subsection*{Investigation Beats (choose order)}
\begin{enumerate}[leftmargin=*]
  \item \textbf{Charter Analysis} (\emph{Wits+Lore or Diplomacy (Broker) DV 3}): Overlaps with \emph{different end terms} (\emph{first frost} vs. \emph{bridge is paid}). Note \emph{witness gaps}.
  \item \textbf{Ledger of Small Kindnesses} (\emph{Wits+Investigation DV 4}): Concrete debts: Lanterns owe \emph{sanctuary upkeep}; Thornwrights owe \emph{bridge toll arrears}.
  \item \textbf{Scout Reports} (\emph{Wits+Survey DV 3}): Sightings: mantlets staged; \emph{lever-frames, pitch-kettles, and rope-saws} positioned at old footings.
\end{enumerate}
\textit{Between-Scene Order Option:} PCs can push any one beat off-screen for speed.

\subsection*{Forged Ledger, Real Stakes (mini-score)}
Make \emph{The False Ledger} an active heist in four steps; three successes needed before 2 SB are generated.
\begin{enumerate}[leftmargin=*]
  \item \textbf{Materials} (\emph{Craft+Wits or Subterfuge DV 3}): Acquire proper paper, stamps, inks. \emph{SB: Shortfall---add Watermark Test later.}
  \item \textbf{Language} (\emph{Lore+Wits DV 4}): Match charter idiom; fill \emph{said/meant}. \emph{SB: Anachronism---adds DV +1 to Delivery.}
  \item \textbf{Witness Seal} (\emph{Diplomacy+Presence DV 3}): Coax a minor clerk/bell-witness to ``remember'' witnessing. \emph{SB: Loose Tongue---\emph{Rumor [2]}}.
  \item \textbf{Delivery} (\emph{Subterfuge+Stealth DV 4} or \emph{Performance+Presence DV 4}): Plant or present without cross-exam. \emph{On hit:} target unit \emph{Withdraws 1 ring} or suffers \emph{Cohesion +1}. \emph{On miss:} \emph{False Orders [4]} flips on PCs.
\end{enumerate}

\subsection*{Bell Dawn Parley (set-piece)}
\emph{Presence+Sway DV 4 under witness; Wits+Diplomacy may Assist.} Ring three bells; both captains attend. If \emph{Ledger Debt} is presented, choose 1:
\begin{itemize}[leftmargin=*]
  \item Impose \emph{No-Seizure of Granaries [4]} jointly.
  \item Establish \emph{Single Banner Hour} (only one company shows colors in town per day).
\end{itemize}

\subsection*{Condotta Flip (Allegiances in Flux)}
Roll 1d6 at scene starts or on a PC trigger; interpret via their play:
\begin{enumerate}[leftmargin=*,label=\Alph*.]
  \item \textbf{Charter Challenge:} Accuse Thornwrights of storing \emph{pitch-kettles and resin jars} in shrine bounds; Lantern inquiry (Position +1 to press).
  \item \textbf{Debt Reminder:} Thornwrights fluster; Captain orders an audit (their Payday +1 if unfunded).
  \item \textbf{False Orders:} Conflicting movements begin (\emph{False Orders [4]} ticks).
  \item \textbf{Broker's Market:} Exclusive lane offers narrow claims (gain \emph{Bridge Peace [4]} on hit).
  \item \textbf{Village Leverage:} Threaten provisioning the rival (on partial: \emph{Village Panic +1}).
  \item \textbf{Article Violation:} Formal accusation under witness; force parley or lose face.
\end{enumerate}

\subsection*{Charter Synthesis Workshop (make it tactile)}
Draft a new charter by choosing \textbf{2 clauses} (and 1 \emph{cost}) from each list; then roll \emph{Broker+Petition DV 5} with Position based on choices made.
\paragraph{Lantern Clauses (pick 2)}
\begin{itemize}[leftmargin=*]
  \item Winter Garrison capped at \emph{20 spears} and \emph{no levy}.
  \item Sanctuary care funded by \emph{toll tithe} (1 in 20 breaths).
  \item Bridge priority stamps only \emph{during declared emergencies}.
\end{itemize}
\paragraph{Thornwright Clauses (pick 2)}
\begin{itemize}[leftmargin=*]
  \item Toll schedule sunsets after \emph{two harvests} or \emph{Bridge Paid [6]}.
  \item \textbf{Fire-setting} limited to \emph{downriver footings}; \emph{no flame} within sanctuary radius.
  \item First refusal on \emph{road repairs} for one year.
\end{itemize}
\paragraph{Costs (pick 1)}
\begin{itemize}[leftmargin=*]
  \item Village owes \emph{Indemnity [4]} if raids spike.
  \item PCs pledge \emph{Witness Duty} (be present for violations; Reputation at stake).
  \item Market day tithe funds a neutral \emph{Broker's Bell}.
\end{itemize}
\textit{On hit:} both patrons accept; gain \emph{Broker's Favor} x2. \textit{On partial:} one accepts; the other demands \emph{Indemnity [4]}. \textit{On miss:} \emph{Political Pressure [4]} enters play.

\subsection*{Endgame Options}
\subsubsection*{The Village Gambit}
\emph{Lore+Diplomacy DV 4 under witness.} Invoke \emph{Ledger Law}: old debts narrow present claims. If upheld, both banners pull to outer rings; village keeps core.

\subsubsection*{The Third Way}
\emph{Command+Presence DV 4.} Joint defense against a common threat; start \emph{Shared Defense [6]}. On fill, both banners leave with honor; village gains \emph{Watch Rights}.

\subsection*{Mechanical Hooks \& SB Menu}
\paragraph{Key Dice Rolls}
\begin{itemize}[leftmargin=*]
  \item \textbf{Charter Reading:} Wits+Lore (DV 3--4); +1 die with \emph{witness gaps}.
  \item \textbf{Banner Law Parley:} Presence+Sway (DV 4); \textit{Article Cite} grants Position +1.
  \item \textbf{Village Negotiation:} Diplomacy or Broker (DV 3--4).
  \item \textbf{False Evidence:} Subterfuge+Lore (DV 4--5) after the mini-score setup.
  \item \textbf{Supply Cut/Restore:} Tactics+Streetwise or Engineering (DV 3--4).
\end{itemize}

\paragraph{Story Beats (Keeper Spend)}
\begin{itemize}[leftmargin=*]
  \item \textbf{False Orders} (colors cross at the ford)
  \item \textbf{Ally Turns} (shamed officer flips)
  \item \textbf{Cut Supply Lines} (granary cordon tightens)
  \item \textbf{Camp-Fever} (restless troops)
  \item \textbf{Gear Failure} (\emph{mantlet collapse; wedge-snap rockfall})
  \item \textbf{Authority Intervention} (lord's reeve arrives)
  \item \textbf{Winter Omens} (first frost early)
\end{itemize}

\subsection*{Outcomes \& Reputation}
\paragraph{Success}
\begin{itemize}[leftmargin=*]
  \item \textbf{Clean Victory:} Both withdraw honor-intact; PCs gain \emph{Broker's Favor} with both.
  \item \textbf{Pragmatic:} One stays as protector on narrowed charter; PCs gain a \emph{Village String} (ford free).
  \item \textbf{Innovative:} Synthesis adopted; PCs tagged as \emph{banner mediators}.
\end{itemize}
\paragraph{Failure}
\begin{itemize}[leftmargin=*]
  \item \textbf{Village Lost:} Saltmere destroyed; PCs earn a \emph{Black Mark}.
  \item \textbf{Banner War:} Companies clash; \emph{Skirmish Spiral [6]} begins.
  \item \textbf{Patron Wrath:} \emph{Political Pressure [4]} follows the party.
\end{itemize}

\subsection*{GM Aids}
\paragraph{When you want to... (Order guidance)}

  \item \textbf{Bind with paper:} Use \emph{Charter Synthesis Workshop}.
  \item \textbf{Make it tactile:} Trigger the \emph{Forgery Mini-Score}.

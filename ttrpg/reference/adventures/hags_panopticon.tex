\section{Overview \& Themes}
\subsection{Premise}
Shipwrecked (or captured), the PCs awaken on an island ``rehabilitation colony'' run by a Hag. The colony’s social order is engineered so no one can say who is \emph{prisoner} and who is \emph{guard}. Roles invert mid-scene; memory and narrative contradict. Escape is possible, overthrow is possible, submission is tempting.

\subsection{Themes}
\begin{itemize}
  \item \textbf{Masks and Roles:} Identity is a costume; truth is a performance.
  \item \textbf{Paranoia as Power:} Doubt fuels control; trust becomes a resource.
  \item \textbf{Bargains Over Bars:} Walls exist, but \emph{social architecture} is the real prison.
\end{itemize}

\subsection{Tone Guide}
Folk-horror unease, candlelit communal halls, whispers through the walls, public rituals of ``confession,'' kindness with a hook.

\subsection{Outcomes at a Glance}
\begin{itemize}
  \item \textbf{Escape:} Flee by sea; the Hag’s influence lingers.
  \item \textbf{Overthrow:} Break her control; inherit a broken community.
  \item \textbf{Submission:} Walk free as her agents beyond the island.
\end{itemize}

\section{Act Structure}
\subsection{Act I: Arrival \& Orientation}
\textbf{Hook:} PCs awaken in the infirmary; they are ``new arrivals.'' A communal meal follows.
\begin{itemize}
  \item \textbf{Read:} smiles, contradictions, missing names. 
  \item \textbf{Tell:} no one states their role plainly.
  \item \textbf{Clue seeds (choose 1--2):}
  \begin{itemize}
    \item A ``guard'' takes orders from a ``prisoner'' without comment.
    \item A hymn switches verses mid-phrase, as if someone else is speaking.
    \item A masked figure watches from a reflection, not the doorway.
  \end{itemize}
\end{itemize}

\subsection{Act II: The Crack in the Facade}
Assign mundane labor (wall repair, kitchens, docks). During the task, reveal the first \emph{break}.
\begin{itemize}
  \item \textbf{Factions:} Conspirators (escape), True Believers (devotion), Traders (barter to survive).
  \item \textbf{Beat:} a public ``discipline'' where roles invert: today’s enforcer is punished as a traitor.
\end{itemize}

\subsection{Act III: The Hag’s Game}
The Hag contacts a PC (dream, whisper, or proxy). Offer three bargains:
\begin{enumerate}
  \item \textbf{Submission:} Serve as her watchers (leave later as agents).
  \item \textbf{Escape:} She opens a path if they betray someone.
  \item \textbf{Overthrow:} ``If you think you can.'' (She tightens the screws.)
\end{enumerate}

\subsection{Act IV: Resolution}
\begin{itemize}
  \item \textbf{Escape:} Reach the saboted skiff; storms and watchers oppose you.
  \item \textbf{Overthrow:} Confront the Hag’s locus (chapel mirror, cistern, or masked warden). Shatter the mechanism of control.
  \item \textbf{Submission:} The Hag grants passage; PCs bear her mark.
\end{itemize}

\begin{callout}
\textbf{Positioning Advice:} Social scenes start \emph{Dominant}; a public reveal pushes to \emph{Desperate}. Use Boons to step up Position or seed Assists; 1s feed your SB.
\end{callout}

\section{NPC Roster}
\subsection{The Hag (Warden Mother)}
\textbf{Face:} Rarely seen directly; appears in reflections, as a masked ``warden,'' or as a kindly voice.
\begin{itemize}
  \item \textbf{Agenda:} Preserve the panopticon; harvest secrets and bargains.
  \item \textbf{Levers:} Role inversion, dream-bargains, public rituals of confession.
  \item \textbf{If cornered:} She does not flee; she \emph{recasts the scene}: who is confessing now?
\end{itemize}

\subsection{Seris (The Kindly Healer)}
\textbf{Mask:} Prisoner (nurturing, soft-spoken). \ \textbf{Truth:} Senior watcher.
\begin{itemize}
  \item \textbf{Wants:} Convert PCs into cooperative assets.
  \item \textbf{Tells:} Always sits with her back to mirrors; never states a role.
  \item \textbf{Offer:} Protection if you ``help'' identify agitators.
\end{itemize}

\subsection{Gorvik (The Gruff Enforcer)}
\textbf{Mask:} Guard (stern, dutiful). \ \textbf{Truth:} Desperate conspirator.
\begin{itemize}
  \item \textbf{Wants:} Out---but won’t risk others lightly.
  \item \textbf{Tells:} Winces when giving orders; works hardest at menial tasks.
  \item \textbf{Offer:} Quiet favors; asks for proof PCs won’t betray him.
\end{itemize}

\subsection{Three Quick Extras}
\begin{itemize}
  \item \textbf{The Believer:} Reports everything. (Is it piety, or leverage?)
  \item \textbf{The Skeptic:} Knows too much; paralyzed by fear.
  \item \textbf{The Ghost:} Swears they died here and woke up again.
\end{itemize}

\begin{callout}
\textbf{Portrayal Note:} Swap an NPC’s displayed role mid-scene once per act. Let the table feel the rug-pulls.
\end{callout}

\section{Key Locations}
\subsection{Longhouse (Common Hall)}
\textbf{Read:} Smoke-sweet air, communal benches, masks hung like trophies.  
\textbf{Secrets:} A slat behind the choir dais hides a crawlspace to the Chapel.  
\textbf{Tension Beat:} A public confession each evening; roles invert mid-ritual.

\subsection{Infirmary (Wards \& Whispers)}
\textbf{Read:} Herb-bitter linen, quiet footfalls, mirror above the washbasin.  
\textbf{Secrets:} The mirror is her favored locus; speaking a name to it summons a watcher.  
\textbf{Tension Beat:} A “patient” orders a “nurse”; no one reacts as if odd.

\subsection{Chapel (Masks \& Reflections)}
\textbf{Read:} Beeswax, iron bell rope, a silvered altar.  
\textbf{Secrets:} The altar’s reflection reveals the Hag’s masked warden even when she is absent.  
\textbf{Tension Beat:} The hymn shifts voice mid-verse as though the island sings back.

\subsection{Workyard (Labor \& Leverage)}
\textbf{Read:} Salt air, pile of nets, wall repair scaffold.  
\textbf{Secrets:} A loose stone conceals a smuggled lockpick and a conspirator’s note.  
\textbf{Tension Beat:} An enforcer is punished as a traitor—today’s guard is tomorrow’s prisoner.

\subsection{Docks (Hope \& Sabotage)}
\textbf{Read:} Splintered skiff, barnacled pilings, treacherous reef.  
\textbf{Secrets:} The only seaworthy skiff needs a sparkstone (kept in the Chapel).  
\textbf{Tension Beat:} Fog arrives unnaturally fast; a watcher “was always here.”

Start \emph{Dominant}. Any public reveal, accusation, or role inversion can drop to \emph{Desperate}.

\section{The Hag’s Low Rites (Paranoia-Fueled)}
All are \emph{Low Rites} intended for social/intrigue tension. They mirror Mab’s toolkit but twist toward gaslighting and identity flux.

\subsection{Wear the Mask}
\textbf{Scope:} Scene; Self; Resisted. \ \textbf{Materials:} Mask/veil/shadow-smear.  
\textbf{Effect:} Present as guard or prisoner at will; NPCs accept it unless violently contradicted.  
\textbf{Push:} Learn one target’s secret (GM supplies). Costs 1 Corruption on a \emph{Miss}.

\subsection{Whisper of Betrayal}
\textbf{Scope:} Scene; Close; Resisted. \ \textbf{Materials:} Shared water; hushed phrase.  
\textbf{Effect:} Target treats an ally as suspicious; -1 die to Assist/Teamwork with that ally.  
\textbf{Push:} Paranoia spreads to the whole group for one exchange.

\subsection{Exchange of Shadows}
\textbf{Scope:} Instant; Near; Unresisted. \ \textbf{Materials:} Overlapped shadows.  
\textbf{Effect:} You and a target swap apparent roles; onlookers “remember” it as true.  
\textbf{Push:} Lingers into Downtime; fallout scene required to clear.

\subsection{Eyes in the Walls}
\textbf{Scope:} Scene; Self; Unresisted. \ \textbf{Materials:} Crack, keyhole, splinter.  
\textbf{Effect:} Observe a nearby scene remotely.  
\textbf{Push:} Receive one \emph{truth} and one \emph{lie}; you don’t know which is which (GM marks 1 SB).

\subsection{The Hag’s Bargain}
\textbf{Scope:} Scene; Touch; Resisted. \ \textbf{Materials:} Blood drop or mirror shard.  
\textbf{Effect:} Grant an ally +1 die; if they \emph{Miss}, you mark 1 Corruption.  
\textbf{Push:} If they \emph{Crit}, you still mark 1 Corruption (the Hag always profits).

\section{Corruption: The Hag (Masks \& Paranoia)}
\label{subsec:hag-corruption}
\index{Corruption!Hag}

\subsection{Subtle Tells (Narrative)}
\begin{itemize}
  \item \textbf{Stage 1:} Whisper-twitch; face-covering habits; compulsive “are you lying?” asides.
  \item \textbf{Stage 2:} False titles; aversion to mirrors; tattletale reflex to unseen authorities.
  \item \textbf{Stage 3:} Shadow out of sync; gaslight instinct; her tone slips into your voice.
  \item \textbf{Stage 4:} Mirror-glint eye; role amnesia; compulsion to spread her bargains.
\end{itemize}

\subsection{SRD Track (Condensed)}
\begin{description}
  \item[\indexterm{1--3 Corruption}] Subtle habits emerge: covering the face, pausing as if listening, probing others for lies.
  \item[\indexterm{4--6 Corruption}] Roles blur: false titles, mirror aversion, reflex to report others’ actions.
  \item[\indexterm{7--9 Corruption}] Paranoia deepens: shadow slip, memory-twisting, the Hag’s tones in your speech.
  \item[\indexterm{10+ Corruption}] Revelation: one eye mirrors dim light; you doubt your role and dream of spreading her bargains.
\end{description}

\section{GM Tools: Paranoia \& Control}

\subsection{Reveal Ladder (Escalation Cues)}
Use these stages to track how overt the Hag’s influence becomes.
\begin{enumerate}
  \item \textbf{Masks in Motion:} PCs notice minor behavioral tics in NPCs (whisper-twitch, compulsive reporting).  
  \item \textbf{Role Confusion:} Guards and prisoners invert in public scenes; records contradict themselves.  
  \item \textbf{Living Walls:} PCs catch the island itself “watching” — whispers in cracks, faces in reflections.  
  \item \textbf{The Hag’s Voice:} Commands ripple through NPCs; PCs hear her in their own speech.  
  \item \textbf{Revelation:} One PC or ally is unmasked as her agent. This is a turning point — confrontation, betrayal, or bargain.  
\end{enumerate}

\subsection{Clock Suite (Examples)}
\begin{itemize}
  \item \textbf{Trust Frays} (4): Allies hesitate to Aid; social Position worsens.  
  \item \textbf{Watcher’s Eye} (6): The Hag intervenes directly; insert bargain/punishment.  
  \item \textbf{Sabotage at the Docks} (4): Escape route collapses unless countered.  
  \item \textbf{Identity Collapse} (6): One PC begins to doubt their own role; must Act to recover.  
\end{itemize}

\subsection{Downtime SB Spend Menu (After Each Session/Scene)}
\begin{longtable}{@{}p{1.5cm}p{13.2cm}@{}}
\toprule
\textbf{SB} & \textbf{Spend (Downtime)} \\
\midrule
1 & A rumor spreads in the cells: PCs suffer -1 die to the next social action unless they quell it. \\
1 & Nightmares visit: mark 1 Fatigue unless a PC takes Watch. \\
2 & A mask “migrates” overnight: swap two NPC roles at the GM’s choice. \\
2 & A PC receives a whispered bargain in their dreams; accept a small boon, or mark 1 Corruption. \\
3 & An ally NPC vanishes; next session they return with subtle Stage 2 Corruption. \\
4 & The Hag herself appears in a dream tribunal; one PC must confess or face a scene start at \emph{Desperate}. \\
\bottomrule
\end{longtable}

\subsection{GM Reminders}
\begin{itemize}
  \item Always spend SB to destabilize trust, never just to “add hit points.”  
  \item Swap roles subtly; the fiction should feel like sand slipping through fingers.  
  \item Corruption is narrative fuel; feed paranoia slowly, then escalate into overt reveals.  
  \item Use Clocks sparingly; fewer, more visible ones emphasize mounting tension.  
\end{itemize}

\section{Player Handout: The Island of Masks}

\subsection{Premise (What You Know)}
You are confined on an isolated prison island ruled by a mysterious Hag.  
No one can say with certainty who is a prisoner and who is a guard.  
Every face wears suspicion; every mask hides a bargain.  
You have one goal: \textbf{survive long enough to find escape or overthrow her hold.}

\subsection{Your Immediate Goals}
\begin{itemize}
  \item Identify potential allies and conspirators.
  \item Avoid exposure as traitor or pawn.
  \item Secure a path off the island --- skiff, dock, or revolt.
  \item Decide whether you will confront the Hag directly or flee her grasp.
\end{itemize}

\subsection{What You Suspect}
\begin{itemize}
  \item The Hag’s servants are hidden among you.  
  \item Masks and roles may shift without warning.  
  \item Bargains are binding, but always slanted in her favor.  
\end{itemize}

\subsection*{Sidebar: Spotting the Hag’s Servants}
\begin{callout}{Tells in Plain Sight}
Keep watch for these subtle behaviors:
\begin{itemize}
  \item Covering the face or speaking in half-pauses, as if “listening.”  
  \item Refusal to meet mirrors or reflective surfaces.  
  \item Compulsively reporting on others’ actions to anyone listening.  
  \item Shadows that move just slightly out of step.  
  \item Speech that slips into strange tones or repeats the Hag’s phrases.  
\end{itemize}
Each of these hints may mark a servant already sliding under her influence.  
It is up to you whether to expose them, trust them, or bargain yourself.
\end{callout}

\subsection{Player Reminders}
\begin{itemize}
  \item Your Position (Dominant/Dominant/Desperate) matters more than raw damage.  
  \item Boons fuel team momentum; Fatigue and Corruption are long-term costs.  
  \item The Hag thrives on paranoia --- play into it, but remember: \textbf{cooperation is still your strongest weapon.}  
\end{itemize}

\section{Closing GM Section: Flow \& Resolution}

\subsection{Adventure Flow (3 Acts)}

\paragraph{Act I: Arrival \& Unease}
\begin{itemize}
  \item Introduce the Longhouse or Infirmary; emphasize masks and role confusion.
  \item Clock: \emph{Trust Frays} (4). Every misstep fuels suspicion.
  \item Seed subtle corruption tells (whisper-twitch, false titles).
\end{itemize}

\paragraph{Act II: Fractures \& Conspiracies}
\begin{itemize}
  \item PCs navigate shifting alliances. Push SB spends from the Social Paranoia table.
  \item Clock: \emph{Watcher’s Eye} (6). Advance it whenever bargains are refused.
  \item Include a reveal ladder step (Role Confusion or Living Walls).
\end{itemize}

\paragraph{Act III: Escape or Overthrow}
\begin{itemize}
  \item Scene pivot to Docks or Chapel. Choose \emph{Escape} or \emph{Confrontation}.
  \item Escape uses the \emph{Escape Pressure} SB menu; Overthrow uses \emph{Overthrow \& Confrontation}.
  \item Endgame corruption: highlight one PC or ally tipping into Stage 3 or 4.
\end{itemize}

\subsection{Final Confrontation Options}
\begin{description}
  \item[Escape] PCs secure the skiff or another route. Costs: Fatigue + SB; some prisoners left behind.  
  \item[Bargain] PCs accept a deal from the Hag. Corruption flows, but safe passage is granted.  
  \item[Overthrow] PCs confront the Hag in her locus (Chapel/altar reflection). Use Overthrow SB menu.  
\end{description}

\subsection{Aftermath Table}
Roll 1d6 or decide based on play outcomes.
\begin{longtable}{@{}p{2cm}p{13cm}@{}}
\toprule
\textbf{Roll} & \textbf{Aftermath} \\
\midrule
1 & The Hag’s power fractures; she vanishes, but masks linger across the isles. PCs gain infamy. \\
2 & Escape succeeds, but one PC carries her Stage 4 Corruption --- a new vector for her influence. \\
3 & Bargain accepted: safe passage in exchange for a permanent Mark (all PCs +1 ongoing Corruption track). \\
4 & Prison revolt succeeds; the Hag’s locus destroyed. Survivors establish a fragile new order. \\
5 & The Hag withdraws, promising return. PCs carry whispers of her bargains into future play. \\
6 & PCs overthrew her utterly. Yet a mirror shard follows them, glinting with potential return. \\
\bottomrule
\end{longtable}

\subsection{GM Wrap Reminders}
\begin{itemize}
  \item Escalate paranoia gradually: from subtle habits to overt reveals.
  \item Use SB to warp trust and identity, not to simply add “difficulty.”
  \item Always end with a clear player choice: flee, bargain, or confront.
  \item Corruption is the long shadow — even victory may carry her mark forward.
\end{itemize}

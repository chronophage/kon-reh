\section{Premise \& Tone}
Frosthollow’s solstice rite, the \emph{Feast of the Endless Night}, has curdled. The village cantor, \textbf{Elara Nightweaver}, wears a living crown of holly and sings with a voice not wholly hers. Revelers decorate with blood–bright berries and do not sleep. Three Powers lean in: \textbf{Thrysos} (ecstasy and excess), the \textbf{Pale Shepherd} (thresholds and the passage), and the \textbf{Silent Choir} (mercy through silence). PCs arrive as storm and song rise together. Their choices decide whether Frosthollow remembers joy—or becomes its grave.

\begin{FEhook}
\textbf{Offer three clear entries:}
\begin{itemize}
  \item \textbf{Shelter in a Storm:} Snow closes behind; bells peal off–rhythm.
  \item \textbf{A Missing Brother:} A hunter’s letter begs you to fetch him from the Feast.
  \item \textbf{Ritual Competence:} A nearby abbot sends you to audit a troubled solstice mass.
\end{itemize}
\end{FEhook}

\section{Fronts \& Clocks}
\begin{multicols}{2}
\begin{FEclock}
\textbf{Village Hysteria [6]} advances when the party indulges, delays, or escalates spectacle.\\
\emph{1–2:} Harmless mania. \emph{3–4:} Harm 1 from exposure/exhaustion. \emph{5–6:} Stampede, collapse, or freezing injuries (Harm 2).
\end{FEclock}

\begin{FEclock}
\textbf{Holly Crown’s Hold [6]} advances when Elara sings unopposed or offerings are made.\\
\emph{1–2:} Subtle compulsion (\(-1\)d to resist revelry). \emph{3–4:} \(-1\) Position near the square. \emph{5–6:} Crown roots through flesh; Elara becomes \emph{Avatar of Revel}.
\end{FEclock}

\begin{FEclock}
\textbf{Storm’s Mercy [4]} advances on time passage or failed shelter efforts.\\
\emph{1:} Whiteout pockets. \emph{2:} Paths vanish. \emph{3:} Avalanches threaten. \emph{4:} Outlying homes cut off (people die without aid).
\end{FEclock}

\begin{FEclock}
\textbf{Silent Choir Mandate [4]} advances when harm spreads or pleas for rest go unheard.\\
\emph{4:} Choir imposes \emph{Great Silence}—speech fails in the square until the rite is resolved.
\end{FEclock}
\end{multicols}

\section{Conditions \& Environmental Moves}
\begin{FElore}
\textbf{Condition — Revelry:} You feel \emph{wonderful}. \(-1\)d to resist joining songs, dances, or feasts; ignoring an invitation advances \textit{Hysteria}. Cleansed by cold immersion, fasting, or a counter–hymn.\\[0.25em]
\textbf{Winter Moves:} Whiteout, Black Ice, Roof–Load Collapse, Bone–Chill Seep (\emph{Harm 1}), Lost Lanterns (separate the party), Avalanche Shadow (clock tick + forced scramble).
\end{FElore}

\section{Core Scenes}

\subsection*{1. Arrival: Snow, Smoke, \& Holly}
\begin{FEscene}
\textbf{Read:} Frosthollow’s square blazes with lanterns; off–key choirs bellow carols. A crown of holly gleams wetly on the cantor’s brow. The bells are late. The storm inhales.
\end{FEscene}

\begin{FEmove}
\textbf{Challenges}
\begin{itemize}
  \item \textbf{Assess the Rite:} \emph{Lore + Insight} \dv{3} — learn the hymn is mis-keyed to force breath patterns (\(+1\) SB to GM if ignored).
  \item \textbf{Resist the Cheer:} \emph{Resolve} \dv{2} — on a miss, gain \emph{Revelry}.
  \item \textbf{Shepherd the Elderly to Shelter:} \emph{Command + Athletics} \dv{3} — on a hit, reduce \textit{Storm’s Mercy}.
\end{itemize}
\end{FEmove}

\subsection*{2. The Bell Tower — Greta’s Truth}
\begin{FEscene}
Greta Frostwhisper, trembling in the belfry, clutches a ledger of the old hymn. “They changed the breathing. Too long. Too fast. They’ll fall and never rise.”
\end{FEscene}

\begin{FEmove}
\begin{itemize}
  \item \textbf{Compare Hymns:} \emph{Lore + Craft} \dv{3} — identify a \emph{counter–refrain} that resets breathing safely.
  \item \textbf{Befriend Greta:} \emph{Sway + Presence} \dv{2} — gain a trusted guide; +1d on village social rolls.
\end{itemize}
\end{FEmove}

\subsection*{3. Elara’s House — The Red Thread}
\begin{FEscene}
Holly cuttings root in bowls; old lyrics are crossed out; a red thread binds prayer–knots to a goblet.
\end{FEscene}

\begin{FEmove}
\begin{itemize}
  \item \textbf{Unravel the Charm:} \emph{Arcana + Tinker} \dv{4} — separate the goblet (Thrysos vessel) from the holly. On partial, advance \textit{Holly Crown’s Hold}.
  \item \textbf{Quiet the Home:} \emph{Shepherd’s Bargain} — accept a memory tithe to sanctify this space (Choir \(-1\) this scene).
\end{itemize}
\end{FEmove}

\subsection*{4. Climax: Feast of the Endless Night}
\begin{FEscene}
Elara raises her hands; the village inhales as one. Snow becomes confetti. The bells strike \emph{late}.
\end{FEscene}

\begin{FEmove}
Choose one or braid them:
\begin{itemize}
  \item \textbf{Counter–Hymn:} \emph{Presence + Sway} \dv{4} — lead a new refrain; on success, reduce \textit{Hysteria} by 2, stabilize breath; on miss, \textit{Holly Hold} +1 and Choir advances.
  \item \textbf{Sunder the Crown:} \emph{Melee/Arcana + Athletics} \dv{4} — break holly roots; on partial, take \emph{Harm 1 (thorns/acid sap)}.
  \item \textbf{Triune Negotiation:} name each Patron’s \emph{concession}: \emph{Insight + Lore} \dv{5} — forge a pact: \emph{short feast, long rest, and a hymn of remembrance}. On hit, all clocks freeze; on miss, one Patron takes primacy (GM move).
\end{itemize}
\end{FEmove}

\begin{FEout}
\textbf{Resolutions}
\begin{itemize}
  \item \textbf{Mercy Kept (Best):} Feast becomes vigil; the storm parts; Elara lives, crown withers. PCs gain 2 Boons (Community/Thresholds).
  \item \textbf{Joy Saved but Costly:} Revelry quiets; 1–2 deaths; Elara scarred. PCs gain 1 Boon; choose: spare a life or spare a memory.
  \item \textbf{Silence Falls (Grim):} Silent Choir imposes Great Silence; Feast ends forever. PCs gain Favor with the Choir; village loses its song.
  \item \textbf{Rapture Eternal (Dark):} Thrysos claims the square; Frosthollow dances into legend. PCs marked with \emph{Revelry} until exorcised.
\end{itemize}
\end{FEout}

\section{Elara as Avatar (If the Crown Consumes)}
\begin{FEstat}
\textbf{Elara, Avatar of Revel}\\
\textbf{Tags:} \tagbox{Elite} \tagbox{Mythic Song} \tagbox{Zone Control}\\
\textbf{Aura — Crescendo:} Each round, all in the square test \emph{Resolve} \dv{3} or gain/advance \emph{Revelry}.\\
\textbf{Song–Strike:} \emph{Presence + Arcana} vs. the crowd; on hit, split \emph{Harm 2 (breathless)} among up to three targets.\\
\textbf{Roots of Holly:} On melee hit against Elara, attacker tests \emph{Wits} \dv{2} or be \emph{Entangled}.\\
\textbf{Weakness:} Counter–hymn sung by a \emph{beloved voice} (Greta or a PC who earned the choir’s trust) reduces her Position by one step this round.
\end{FEstat}

\section{GM Dials \& Tools}
\begin{FEdial}
\begin{itemize}
  \item \textbf{Patron Prominence:} Feature 1, 2, or all 3. For one–shot pace, lead with Thrysos, let Shepherd save, let Choir judge.
  \item \textbf{Combat Light or Heavy:} Swap \emph{Sunder the Crown} to a pure ritual (\emph{Arcana + Insight}) or to a brawl in the choir loft.
  \item \textbf{Hard Mode:} Add \emph{Breath Debt}: each failed Revelry resist deals \emph{Harm 1 (asphyxia)} that only the counter–hymn heals.
  \item \textbf{Mechanical Rewards:} \emph{If Mercy Wins:} +1d on future \emph{Comfort/Recovery} scenes. \emph{If Rapture Tempted:} bank 1 \emph{Joy Token}—cash for +1d on one grand celebration later.
\end{itemize}
\end{FEdial}

\section{Quick Start for PCs}
\begin{itemize}
  \item \textbf{Runekeeper (Thresholds):} Bargain with the Shepherd for safe passage; pay in story or scar.
  \item \textbf{Cantor / Bard:} Compose the counter–hymn; lead call–and–response to cleanse \emph{Revelry}.
  \item \textbf{Sellsword:} Guard the bell ropes; hold the square against stampede or falling ice.
  \item \textbf{Healer:} Replace \emph{Cheer} with \emph{Breath}—synchronized breathing drills under the eaves.
\end{itemize}

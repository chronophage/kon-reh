\subsection*{Story Beat (SB) Spend Menu}
\begin{itemize}
    \item \textbf{1 SB}: Minor pressure: noise, trace, +1 Supply segment.
    \item \textbf{2 SB}: Moderate setback: alarm, lose cover, lesser foe.
    \item \textbf{3 SB}: Serious trouble: reinforcements, key gear breaks, rail tick.
    \item \textbf{4+ SB}: Major turn: trap springs, authority arrives, scene shifts.
\end{itemize}

\subsection*{Position Descriptions}
\begin{itemize}
    \item \textbf{Dominant}: Act on your terms. Consequences are manageable.
    \item \textbf{Controlled}: Normal risk/reward. Failure has a real cost.
    \item \textbf{Desperate}: Odds stacked against you. Failure could be catastrophic.
\end{itemize}

\subsection*{Boon Economy Quick Guide}
\begin{itemize}
  \item \textbf{Holding cap:} 5 Boons maximum.
  \item \textbf{Carryover:} End of scene → trim to 2 Boons.
  \item \textbf{Conversion:} 2 Boons → 1 XP, once per session during downtime (max 2 XP via conversion).
  \item \textbf{Spending:} Re‑roll a die, activate an asset, improve Position, or power certain Rites.
\end{itemize}

\subsection*{Mechanical Constraints}
\begin{itemize}
  \item \textbf{Assist Max}: +3 dice total per roll (Exceptional Coordination: +4 from one follower).
  \item \textbf{Boon Max}: 5 total; 2→1 XP conversion once/session (max 2 XP via conversion).
  \item \textbf{Initiative}: 1 Follower Action per scene (party‑wide).
  \item \textbf{Over‑Stack}: 2+ structural advantages → start one obstacle at +1 DV or GM banks +1 SB.
\end{itemize}

\subsection*{Range Bands}
\begin{description}
\item[Close] Arm’s length, grappling.
\item[Near] Same room or immediate area.
\item[Far] Visible but not immediately reachable.
\item[Absent] Off‑screen; requires scene change.
\end{description}

\subsection*{Movement Actions}
\begin{itemize}
  \item \textbf{Move}: Shift one band (Close↔Near or Near↔Far).
  \item \textbf{Dash}: Shift two bands as your full action.
  \item \textbf{Disengage}: Test to leave Close range when threatened.
  \item \textbf{Sprint}: Rapid movement across the battlefield.
\end{itemize}

\subsection*{Supply Clock States}
\begin{description}
\item[Full (0)] Well‑provisioned, no penalty.
\item[Low (2)] Minor complications.
\item[Dangerously Low (3)] Each PC gains Fatigue 1.
\item[Empty (4)] Severe penalties; starvation risk.
\end{description}

\subsection*{Travel Clock Sizes}
\begin{description}
\item[4 segments] Short, straightforward.
\item[6 segments] Standard leg.
\item[8 segments] Extended or complex.
\item[10 segments] Epic or highly dangerous.
\end{description}

\subsection*{Upkeep (Condensed)}
\textbf{Frequency:} Once per Downtime.
\begin{itemize}
  \item \textbf{Efficient (XP, little time):} Upkeep XP = $\max(1, \frac{\text{Acquisition XP}}{3})$.
  \item \textbf{Intensive (1 XP, action):} One dedicated downtime action.
  \item \textbf{Failure:} Follower → Wary (or Seized); Asset → Neglected (or Compromised).
\end{itemize}

\subsection*{Grid‑Based Combat}
Use a square or hex grid with Zones of Control (ZoC).
\paragraph{ZoC rules:}
\begin{itemize}
  \item Entering enemy ZoC ends movement (you are engaged).
  \item Cannot move through enemy ZoC.
  \item Leaving requires Disengage (DV 4–6) or spend 1 Boon.
  \item Flank: +1 die; Rear: +1 die and +1 Effect.
\end{itemize}
\paragraph{Push‑Through:} $\text{Body}+\text{Athletics}$ vs. DV $=2+\text{enemy Tier}$; choose a cost (Position –1, –1 die, or opportunity attack).

\subsection*{Detailed Warfare}
Treat an army as a Follower with its own clocks:
\begin{itemize}
  \item \textbf{Supply Clock} (6–8): runs out → penalties.
  \item \textbf{Morale Clock} (6–8): fills → routs.
  \item Command rolls advance objective clocks (e.g., \emph{Break Their Line} [6]).
  \item \textbf{Battle’s Edge} – SB from command rolls spent on fog‑of‑war complications.
\end{itemize}

\subsection*{Hex‑Based Exploration}
Map a region in 6‑mile hexes. Entering a hex requires $\text{Wits}+\text{Survival}$ (DV by terrain). Outcomes determine discovery, danger, or getting lost.

\subsection*{Miniatures and Tactical Layer}
\begin{tcolorbox}[title=Miniatures Mode — Speed Defaults, colback=white!98!gray]
\textbf{DV:} $\mathrm{DV}=\mathrm{Tier}+2+\text{Keywords}$ (elevation +1, [WARD] +1, Disengage = 4).\\
\textbf{Crit:} Bump Position one step; if Dominant, Push/Pull 1 hex \emph{or} +1 Success.\\
\textbf{ZOC:} Enter/leave an adjacent hex provokes 1 Reaction (Free Strike or Shove 1 hex).\\
\textbf{Heat:} On a Crit, GM spends 1 Heat to degrade Position or trigger terrain.
\end{tcolorbox}
\begin{tcolorbox}[title=Hex Keywords, colback=white!98!gray]
\textbf{Difficult:} 2 MP/hex \textbf{Elevation:} +1 DV from below \textbf{ZOC:} Reaction on cross\\
\textbf{Altar [WARD]:} –1 die (or attacker takes 2 Fatigue to ignore)\\
\textbf{Incorporeal:} Ignore Difficult; pass through occupied hexes, not end there
\end{tcolorbox}

\subsection*{Deck Types and Meanings}
\begin{description}
  \item[Travel Decks (regional, 52‑card)] Spade=Place, Heart=Actor, Club=Pressure, Diamond=Leverage.
  \item[Deck of Consequences (scene drama)] Hearts=social/emotional, Spades=harm/escalation, Clubs=material cost, Diamonds=magical/spiritual.
\end{description}
\textbf{Important:} Never mix suit meanings across decks.

\subsection*{Deck Usage Procedure}
After a roll generating SB:
\begin{enumerate}
  \item \textbf{Direct Spend}: Translate SB into immediate consequences/clock ticks.
  \item \textbf{Deck Draw}: Draw up to $\min(\text{SB},3)$ cards, synthesize a single twist guided by suit and highest rank.
\end{enumerate}

\subsection*{Rank Severity Guide}
\begin{description}
  \item[Ace–3] Minor inconvenience/flavor.
  \item[4–6] Moderate setback.
  \item[7–9] Significant consequence.
  \item[10–King] Major fallout, lasting effects.
\end{description}

\subsection*{Deck of Consequences Interpretation}
\begin{description}
  \item[Hearts (Emotional/Social)] Ace–3: awkward moment; 4–6: relationship strain; 7–9: betrayal; 10–King: exile, shattered alliance.
  \item[Spades (Harm/Escalation)] Ace–3: bruise, fatigue; 4–6: wound, gear damaged; 7–9: severe injury; 10–King: death, permanent loss.
  \item[Clubs (Material/Cost)] Ace–3: minor loss; 4–6: gear failure, debt; 7–9: major asset lost; 10–King: total ruin.
  \item[Diamonds (Magical/Spiritual)] Ace–3: omen, whisper; 4–6: curse, spirit appears; 7–9: arcane backlash; 10–King: reality bends.
\end{description}

\section*{Appendix D: Campaign Management}
\addcontentsline{toc}{section}{Appendix D: Campaign Management}

\subsection*{Campaign Clock Examples}
\paragraph{Mandate Advancement} Public victories, crisis resolution, recognition by powerful factions.
\paragraph{Crisis Advancement} Rival faction gains influence, asset neglect/betrayal, scandal.

\subsection*{Follower and Asset Condition States}
\begin{description}
  \item[Maintained] Full capability.
  \item[Neglected] –1 die penalty; narrative wear.
  \item[Compromised] Unavailable until repaired or recovered.
\end{description}

\subsection*{Faction Loyalty Scale (Optional)}
\begin{description}
  \item[-3 Enemy] Actively works against players.
  \item[0 Neutral] Indifferent.
  \item[+3 Ally] Will sacrifice for players.
\end{description}

\subsection*{Campaign Pacing (Arc Structure)}
\begin{enumerate}
  \item \textbf{Introduction} (1–2 sessions): establish stakes, hook players.
  \item \textbf{Development} (2–4 sessions): complications, alliances.
  \item \textbf{Climax} (1–2 sessions): confrontation, critical choices.
  \item \textbf{Resolution} (1 session): consequences, new status quo.
\end{enumerate}

\section*{Appendix E: Safety and Inclusivity}
\addcontentsline{toc}{section}{Appendix E: Safety and Inclusivity}

\subsection*{X‑Card System}
\begin{itemize}
  \item \textbf{X‑Card}: “I’m not comfortable – change direction.”
  \item \textbf{O‑Card}: “I’m excited – lean into it.”
  \item \textbf{N‑Card}: “I need a break – pause.”
\end{itemize}
Keep cards visible. Model appropriate use as GM.

\subsection*{Script Change}
Ask at appropriate moments:
\begin{itemize}
  \item \textbf{Pause}: “Are you okay with this content?”
  \item \textbf{Rewind}: “Would you like to go back?”
  \item \textbf{Fast Forward}: “Would you like to skip ahead?”
\end{itemize}

\subsection*{Inclusive Worldbuilding}
\begin{itemize}
  \item Research real cultures thoroughly.
  \item Avoid stereotypes.
  \item Provide character options not reliant on real‑world cultural markers.
  \item Use high‑contrast, readable materials for accessibility.
\end{itemize}

\subsection*{Conflict Resolution}
When conflicts arise:
\begin{enumerate}
  \item Pause gameplay.
  \item Discuss privately if needed.
  \item Focus on understanding, not fault.
  \item Find collaborative solutions.
  \item Document agreements.
\end{enumerate}
Common solutions: GM ruling stands for the session; research later; adjust future play.

\section{Core Philosophy \& Safety}
%=====================================================================

\subsection{Narrative First}
Mechanics serve the story. Describe what you do; the dice decide how it goes wrong or right.

\subsection{Golden Rule}
When in doubt, the GM makes a ruling that keeps the story moving. Default: \dv\ 3, \position\ Controlled.

\subsection{Safety Tools}
\begin{itemize}
    \item \textbf{Lines:} Content that never appears.
    \item \textbf{Veils:} Content that happens off-screen.
    \item \textbf{X-Card:} Any player may stop and rewind a scene. No questions asked.
\end{itemize}
The GM is also a player.

%=====================================================================
\section{The 90-Second Play Loop}
%=====================================================================

\begin{enumerate}[label=\textbf{\arabic*}.]
    \item \textbf{Declare intent} -- ``I want to X by doing Y.''
    \item \textbf{GM sets \position\ and \dv} -- Risk and Difficulty.
    \item \textbf{Roll} -- Attribute + Skill d10s.
    \item \textbf{Count successes} (6+) and \sbs\ (1s).
    \item \textbf{Apply outcome} -- see Outcome Matrix.
    \item \textbf{GM spends \sbs} -- complications, timers, scene shifts.
    \item \textbf{Pass the spotlight} -- next player acts.
\end{enumerate}

%=====================================================================
\section{Dice and Outcomes}
%=====================================================================

\subsection{Dice Pool}
\index{Dice pool}
Dice pool = \textbf{Attribute + Skill} (both 1--5).  
Success = die showing \textbf{6--10}. A \textbf{10} counts as two successes.  
\sbs = die showing \textbf{1}. The GM gains one SB for each 1 rolled.

\subsection{Outcome Matrix}
\index{Outcome matrix}

\begin{coretable}[Outcome Matrix]
\begin{tabular}{>{\bfseries}p{0.28\linewidth} p{0.62\linewidth}}
\toprule
\textbf{Result} & \textbf{What Happens} \\
\midrule
Clean Success & Intent achieved, no complication. \\
Success with SB & Intent achieved; GM spends SB for a twist. \\
Partial & Progress made; gain 1 \boons. \\
Miss & No progress; gain 2 \boons; GM escalates. \\
\bottomrule
\end{tabular}
\end{coretable}

\textbf{Important:} \sbs are generated on the roll, not after re-rolls. Re-rolling a 1 does not remove the SB.

%=====================================================================
\section{Boons and Story Beats}
%=====================================================================

\subsection{Boons (Player Resource)}
\index{Boons}

\begin{itemize}
    \item Hold up to \textbf{5 \boons}. At scene end, reduce to 2 (excess lost).
    \item \textbf{Spend 1 \boons\ to:}
        \begin{itemize}
            \item Re-roll a die (after seeing pool)
            \item Improve \position\ by one step (before rolling)
            \item Activate an \asset
            \item Power certain abilities
        \end{itemize}
    \item \textbf{Earn \boons:} Partial (1), Miss (2), \bond-driven action (once per \bond\ per session), GM award.
    \item \textbf{Convert to XP:} 2 \boons $\rightarrow$ 1 XP, once per session, max 2 XP.
\end{itemize}

\begin{sidebar}
    \textbf{The Math of Fun:} Hoarding \boons\ for XP conversion starves the table of drama. A re-rolled failure or an improved \position\ creates a memorable moment. Spend \boons!
\end{sidebar}

\subsection{Story Beats (GM Resource)}
\index{Story Beat}

\begin{coretable}[SB Spend Menu]
\begin{tabular}{>{\bfseries}p{0.2\linewidth} p{0.7\linewidth}}
\toprule
\textbf{Cost} & \textbf{Effect} \\
\midrule
1 SB & Minor: noise, trace, +1 segment on a \timer \\
2 SB & Moderate: alarm raised, lose advantage, lesser foe appears \\
3+ SB & Major: reinforcements, key gear breaks, scene shifts \\
\bottomrule
\end{tabular}
\end{coretable}

\section{Position and Difficulty}

\subsection{Position (Risk)}
\index{Position}
Position is set by fiction (cover, surprise, danger).

\section{Combat in Brief}
%=====================================================================

\begin{itemize}
    \item \textbf{Initiative:} Narrative -- GM spotlights based on fiction. Ambushes act first.
    \item \textbf{Turn:} Move (one range band) + Action.
    \item \textbf{Range Bands:} Close $\rightarrow$ Near $\rightarrow$ Far $\rightarrow$ Absent.
    \item \textbf{Opportunity Attack:} Leaving Close range without Disengage provokes one free melee attack.
\end{itemize}

\subsection{Actions}
\begin{coretable}[Common Actions]
\begin{tabular}{>{\bfseries}p{0.3\linewidth} p{0.6\linewidth}}
\toprule
\textbf{Action} & \textbf{Effect} \\
\midrule
Attack & Melee or ranged attack. \\
Disengage & Move without provoking opportunity attacks. \\
Charge (full turn) & Move two bands + melee attack; costs 1 Fatigue; next defense Desperate. \\
Defend & Improve Position by one step for defense; protect allies. \\
Assist & Spend 1 \boons\ to give +1 die to ally (max +3 from all assists). \\
\bottomrule
\end{tabular}
\end{coretable}

%=====================================================================
\section{Harm and Fatigue}
%=====================================================================

\subsection{Fatigue Track}
\index{Fatigue}
\begin{itemize}
    \item Each character has Fatigue = Body rating.
    \item Each Fatigue worsens \position: Dominant $\rightarrow$ Controlled $\rightarrow$ Desperate.
    \item If already Desperate, each Fatigue gives --1 die instead.
    \item \textbf{Full track:} Increase Harm by 1 step, clear all Fatigue.
\end{itemize}

\subsection{Harm Levels}
\index{Harm}
\begin{coretable}[Harm Levels]
\begin{tabular}{>{\bfseries}p{0.3\linewidth} p{0.6\linewidth}}
\toprule
\textbf{Harm Level} & \textbf{Effect} \\
\midrule
Harm 1 & --1 die on related actions \\
Harm 2 & --1 die on most actions until treated \\
Harm 3 & Incapacitated or dying \\
\bottomrule
\end{tabular}
\end{coretable}

\subsection{Taking Harm (Procedure)}
\begin{enumerate}
    \item Convert incoming Harm to Fatigue using armor (see table below).
    \item Apply remaining Harm -- if any Harm remains, increase Harm level and clear all existing Fatigue (the shock resets exhaustion).
    \item Add converted Fatigue; if track fills, overflow triggers Harm increase.
\end{enumerate}

\subsection{Armor Conversion}
\begin{coretable}[Armor Conversion]
\begin{tabular}{>{\bfseries}p{0.2\linewidth} >{\bfseries}p{0.2\linewidth} >{\bfseries}p{0.2\linewidth} >{\bfseries}p{0.2\linewidth}}
\toprule
\textbf{Armor} & \textbf{Harm 1} & \textbf{Harm 2} & \textbf{Harm 3} \\
\midrule
Light & Fatigue 1 & Harm 1 + Fatigue 1 & Harm 2 + Fatigue 1 \\
Medium & Fatigue 1 & Fatigue 1 & Harm 2 + Fatigue 1 \\
Heavy & Fatigue 1 & Fatigue 1 & Fatigue 2 \\
\bottomrule
\end{tabular}
\end{coretable}

\subsection{Recovery}
\begin{itemize}
    \item \textbf{Fatigue:} Short rest (1 hour) clears 2; full night's rest clears all.
    \item \textbf{Harm:} Requires medical care and time (downtime). Magical healing exists but has costs (see Magic section).
\end{itemize}

%=====================================================================
\section{Player-Managed Resources}
%=====================================================================

Players track their own:
\begin{itemize}
    \item \obligation\ (per Patron or Symbol) -- Capacity = Spirit + Presence. Exceeding capacity causes 1 Fatigue per segment over.
    \item \corruption\ timer (Cantors)
    \item Leash timer (Summoners)
    \item \asset\ states (Maintained / Neglected / Compromised)
    \item \follower\ conditions (Exposure, Harm)
    \item Fatigue and Harm
    \item \boons
\end{itemize}

\textbf{Your duties:} Mark immediately when a rule says ``mark +X''. Declare thresholds (``My \obligation\ is full!''). Keep trackers visible to the table.

\section{Magic in One Page}
%=====================================================================

\subsection{Free Caster (Spellcraft)}
Requires \textbf{Spellcraft} talent (6 XP). No Patron.

\begin{itemize}
    \item \textbf{Casting:} Describe spell using \textbf{TAGS}. \dv = 1 + number of tags. Dangerous tags (Teleport, Transform, Dominate) add +2 \dv.
    \item \textbf{Roll:} Wits + Arcana.
    \item \textbf{Backlash (on Miss or GM spending SB):} Choose one: Harm 2 (arcane), +2 Fatigue, \corruption\ +1, or catastrophic side effect (GM describes).
\end{itemize}

\subsection{Sample Spells}
\begin{coretable}[Sample Spells]
\begin{tabular}{>{\bfseries}p{0.35\linewidth} p{0.35\linewidth} p{0.2\linewidth}}
\toprule
\textbf{Spell} & \textbf{TAGS} & \textbf{DV} \\
\midrule
Light a candle & [ILLUMINATE] & 2 \\
HEAL Fatigue & [HEAL] & 2 \\
Shield of Air & [WARD] [DEFEND] & 3 \\
Flame Cloak & [BURNING] [AREA] & 3 \\

\bottomrule
\end{tabular}
\end{coretable}

\subsection{Runekeeper (The Reliable Path)}
\textbf{Requirements:} Familiar (Thiasos, 2 XP) + Codex (4 XP) + Patron.

\begin{itemize}
    \item \textbf{Invoking a Rite:} 1 action. Mark +1 \obligation. Effect works as written.
    \item \textbf{Push It:} Once per scene, mark +1 \obligation\ to amplify (+1 die or +1 Effect).
    \item \textbf{Obligation Capacity:} Spirit + Presence.
\end{itemize}

\textbf{For first magical characters, play a Runkeeper.}

%=====================================================================
\section{Timers and Pressure}
%=====================================================================

\index{Clock}
\timer s are visible trackers for threats, progress, and looming changes.

\textbf{Size:} 4 (short scene), 6 (standard), 8 (arc), 10 (epic).  
\textbf{Advance a \timer:} on a Miss, certain Partials, or when GM spends SB.

\subsection{Three-Timer Guideline (per scene)}
\begin{itemize}
    \item \textbf{Immediate \timer} -- what's about to go wrong (e.g., Collapsing Bridge [4])
    \item \textbf{Situational \timer} -- pressure shaping decisions (e.g., Crowd Turns Hostile [6])
    \item \textbf{Long-term \timer} -- echoes forward (e.g., Faction Retaliation [8])
\end{itemize}

%=====================================================================
\section{Bonds and Teamwork}
%=====================================================================

\subsection{Bonded Actions}
\index{Bond}
\begin{itemize}
    \item Once per \bond\ per session: when you help a bonded ally with an intricate description, \textbf{both gain 1 \boons} after the action.
    \item At \textbf{Tier II+ (41+ XP)}, you may transfer up to 2 \boons\ to a bonded ally as a free action (once per scene).
\end{itemize}

\subsection{Assist Limit}
Max +3 dice from all assists per roll. One \follower\ may assist (adds min(Cap, relevant skill) dice).

\subsection{Selling the Action (GM may grant +1 Position or bank a Boon)}
\begin{itemize}
    \item \textbf{Counts:} stating a clear stake, invoking a \bond, revealing a fear, making a hard choice visible.
    \item \textbf{Does not count:} piling on adjectives, passive sensory lists, repeated descriptions.
\end{itemize}

%=====================================================================
\section{Your Turn Checklist}
%=====================================================================

\begin{enumerate}
    \item Declare intent and approach (Attribute + Skill)
    \item GM tells you \position\ and \dv
    \item Roll dice, count successes (6+) and 1s (SB for GM)
    \item Apply outcome (GM will tell you Clean/Success with SB/Partial/Miss)
    \item Mark \boons\ (1 on Partial, 2 on Miss) and any Fatigue/Harm
    \item If you rolled 1s, GM may spend SB immediately
\end{enumerate}

\section{Quick Build \& Advancement}
%=====================================================================

\subsection{Starting Characters (32 XP)}
\begin{itemize}
    \item \textbf{Attributes:} each starts at 1. Raise to 2 (6 XP), 3 (9 XP), etc. (new rating $\times$3 XP)
    \item \textbf{Skills:} start at 0. Raise to 1 (2 XP), 2 (4 XP), etc. (new level $\times$2 XP)
    \item \textbf{Minor \asset\ (4 XP):} safehouse, contact, special tool
    \item \textbf{Minor Talent (2--3 XP):} Keen Senses, Second Wind, etc.
    \item \textbf{\bond s (up to 2, +2 XP each)}
    \item \textbf{Complications (up to 2, +2 XP each)} -- each unresolved adds +1 banked SB to early scenes.
\end{itemize}

\subsection{Advancement per Session (typical)}
\begin{itemize}
    \item Attendance: +2 XP
    \item Major objectives: +2--4 XP
    \item Discoveries: +1--2 XP
    \item Hard choices: +1--2 XP
    \item Personal goals: +1--2 XP
\end{itemize}
\textbf{Game pace options:} Gritty (4--6 XP/session), Standard (6--10), Epic (10--14).

\section{Glossary}
%=====================================================================

\begin{itemize}
    \item \textbf{\gameterm{Asset}} -- Off-screen resource (safehouse, spy network, workshop). Costs XP. Can be activated with 1 \boons.
    \item \textbf{\gameterm{Bond}} -- Mutual connection between two PCs. Once per session, when you help your bonded ally with intricate description, both gain 1 \boons.
    \item \textbf{\gameterm{Boons}} -- Player resource earned on Partials and Misses. Spend to re-roll, improve Position, activate Assets.
    \item \textbf{\gameterm{Cap}} -- Follower capability rating (1--5). Adds min(Cap, skill) dice when assisting.
    \item \textbf{\gameterm{Clock}} -- Visible progress tracker (4/6/8/10 segments). Advances on Misses, Partials, or SB spends.
    \item \textbf{\gameterm{Compromised}} -- State of an Asset or Symbol: unusable until repaired.
    \item \textbf{\gameterm{Corruption}} -- Cantor resource. Timer fills $\rightarrow$ gain Patron's burden/benefit, reset to Tier minimum.
    \item \textbf{\gameterm{Crisis}} -- Campaign timer (0--6). High = enemies stronger.
    \item \textbf{\gameterm{Difficulty Value (DV)}} -- Target number of successes. Default 3.
    \item \textbf{\gameterm{Exposure}} -- Follower stress track. Leads to Neglected/Compromised.
    \item \textbf{\gameterm{Fatigue}} -- Exhaustion track. Worsens Position; full track $\rightarrow$ Harm+1, clear Fatigue.
    \item \textbf{\gameterm{Follower}} -- Named NPC ally. Costs Cap$^2$ XP. Can assist or take independent actions.
    \item \textbf{\gameterm{Harm}} -- Injury track (1--3). Imposes die penalties.
    \item \textbf{\gameterm{Leash}} -- Summoner's control track. When full, spirit acts independently.
    \item \textbf{\gameterm{Mandate}} -- Campaign timer (0--6). High = party is respected.
    \item \textbf{\gameterm{Miss}} -- 0 successes. Player gains 2 \boons; GM gains SB and escalates.
    \item \textbf{\gameterm{Neglected}} -- State of Asset or Follower: --1 die until attention given.
    \item \textbf{\gameterm{Obligation}} -- Debt to Patron (or terrestrial patron). Capacity = Spirit + Presence. Over capacity $\rightarrow$ Fatigue.
    \item \textbf{\gameterm{Partial}} -- $0 <$ successes $<$ DV. Player gains 1 \boons; GM gains SB.
    \item \textbf{\gameterm{Patron}} -- Cosmic entity granting Rites. Runkeeper = Obligation; Invoker = Symbol; Cantor = Corruption.
    \item \textbf{\gameterm{Position}} -- Risk level: Dominant (safe), Controlled (normal), Desperate (dangerous).
    \item \textbf{\gameterm{Story Beat (SB)}} -- GM resource. Earned on any 1. Spent to create complications.
    \item \textbf{\gameterm{String}} -- Narrative leverage (favor owed, secret known). Can be spent like currency.
    \item \textbf{\gameterm{TAGS}} -- Keywords for freeform magic (e.g., [FIRE], [AREA], [HEAL]). DV = 1 + number of tags.
    \item \textbf{\gameterm{Terrestrial Patron}} -- Mortal faction offering aid in exchange for Debt Timer .
    \item \textbf{\gameterm{Thiasos}} -- Familiar/token for Runekeepers (2 XP). Required for Patron's Gift.
    \item \textbf{\gameterm{XP}} -- Experience Points. Earned at session end. Spent on Attributes, Skills, Assets, Talents.
\end{itemize}

\section{Session Zero Checklist}
%=====================================================================

Use this checklist to start your campaign.
\begin{enumerate}
    \item \textbf{Lines \& Veils} -- What content is off-limits or only implied?
    \item \textbf{Tone} -- Cinematic action? Grim survival? Political intrigue? Dark horror?
    \item \textbf{Campaign length} -- One-shot, short arc (3--6 sessions), or long campaign (12+)?
    \item \textbf{Character hooks} -- Each player shares one goal and one fear.
    \item \textbf{Patron interest} -- Which Patrons (if any) are already watching the party?
    \item \textbf{Starting situation} -- Where does the story begin?
    \item \textbf{House rules} -- Any optional subsystems the GM plans to use?
    \item \textbf{Scheduling \& attendance} -- How often do you meet? What if a player misses?
    \item \textbf{Player-GM contract} -- ``We are co-creators, not adversaries.''
    \item \textbf{Resource mindset} -- ``Boons are not XP tokens. Spend them. Embrace failure.''
\end{enumerate}

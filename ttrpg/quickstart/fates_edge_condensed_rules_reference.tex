\section{Core Philosophy & Safety}
%=====================================================================

\subsection{Narrative First}
Mechanics serve the story. Describe what you do; the dice decide how it goes wrong or right.

\subsection{Golden Rule}
When in doubt, the GM makes a ruling that keeps the story moving. Default: \dv\ 3, \position\ Controlled.

\subsection{Safety Tools}
\begin{itemize}
    ℹtem \textbf{Lines:} Content that never appears.
    ℹtem \textbf{Veils:} Content that happens off−screen.
    ℹtem \textbf{X−Card:} Any player may stop and rewind a scene. No questions asked.
\end{itemize}
The GM is also a player.

%=====================================================================
\section{The 90−Second Play Loop}
%=====================================================================

\begin{enumerate}[label=\textbf{\arabic*}.]
    ℹtem \textbf{Declare intent} – ``I want to X by doing Y.''
    ℹtem \textbf{GM sets \position\ and \dv} – Risk and Difficulty.
    ℹtem \textbf{Roll} – Attribute + Skill d10s.
    ℹtem \textbf{Count successes} (6+) and \sbs\ (1s).
    ℹtem \textbf{Apply outcome} – see Outcome Matrix.
    ℹtem \textbf{GM spends \sbs} – complications, timers, scene shifts.
    ℹtem \textbf{Pass the spotlight} – next player acts.
\end{enumerate}

%=====================================================================
\section{Dice and Outcomes}
%=====================================================================

\subsection{Dice Pool}
∈dex{Dice pool}
Dice pool = \textbf{Attribute + Skill} (both 1–5).  
Success = die showing \textbf{6–10}. A \textbf{10} counts as two successes.  
\sbs = die showing \textbf{1}. The GM gains one SB for each 1 rolled.

\subsection{Outcome Matrix}
∈dex{Outcome matrix}

\begin{coretable}[Outcome Matrix]
\begin{tabular}{>{\bfseries}p{0.28łinewidth} p{0.62łinewidth}}
⊤rule
\textbf{Result} & \textbf{What Happens} \\
∣rule
Clean Success & Intent achieved, no complication. \\
Success with SB & Intent achieved; GM spends SB for a twist. \\
Partial & Progress made; gain 1 \boons. \\
Miss & No progress; gain 2 \boons; GM escalates. \\
⊥tomrule
\end{tabular}
\end{coretable}

\textbf{Important:} \sbs are generated on the roll, not after re−rolls. Re−rolling a 1 does not remove the SB.

%=====================================================================
\section{Boons and Story Beats}
%=====================================================================

\subsection{Boons (Player Resource)}
∈dex{Boons}

\begin{itemize}
    ℹtem Hold up to \textbf{5 \boons}. At scene end, reduce to 2 (excess lost).
    ℹtem \textbf{Spend 1 \boons\ to:}
        \begin{itemize}
            ℹtem Re−roll a die (after seeing pool)
            ℹtem Improve \position\ by one step (before rolling)
            ℹtem Activate an \asset
            ℹtem Power certain abilities
        \end{itemize}
    ℹtem \textbf{Earn \boons:} Partial (1), Miss (2), \bond−driven action (once per \bond\ per session), GM award.
    ℹtem \textbf{Convert to XP:} 2 \boons $→$ 1 XP, once per session, max 2 XP.
\end{itemize}

\begin{sidebar}
    \textbf{The Math of Fun:} Hoarding \boons\ for XP conversion starves the table of drama. A re−rolled failure or an improved \position\ creates a memorable moment. Spend \boons!
\end{sidebar}

\subsection{Story Beats (GM Resource)}
∈dex{Story Beat}

\section{Combat in Brief}
%=====================================================================

\begin{itemize}
    ℹtem \textbf{Initiative:} Narrative – GM spotlights based on fiction. Ambushes act first.
    ℹtem \textbf{Turn:} Move (one range band) + Action.
    ℹtem \textbf{Range Bands:} Close $→$ Near $→$ Far $→$ Absent.
    ℹtem \textbf{Opportunity Attack:} Leaving Close range without Disengage provokes one free melee attack.
\end{itemize}

\subsection{Actions}
\begin{coretable}[Common Actions]

%=====================================================================
\section{Harm and Fatigue}
%=====================================================================

\subsection{Fatigue Track}
∈dex{Fatigue}
\begin{itemize}
    ℹtem Each character has Fatigue = Body rating.
    ℹtem Each Fatigue worsens \position: Dominant $→$ Controlled $→$ Desperate.
    ℹtem If already Desperate, each Fatigue gives –1 die instead.
    ℹtem \textbf{Full track:} Increase Harm by 1 step, clear all Fatigue.
\end{itemize}

\subsection{Harm Levels}
∈dex{Harm}
\begin{coretable}[Harm Levels]
\begin{tabular}{>{\bfseries}p{0.3łinewidth} p{0.6łinewidth}}
⊤rule
\textbf{Harm Level} & \textbf{Effect} \\
∣rule
Harm 1 & –1 die on related actions \\
Harm 2 & –1 die on most actions until treated \\
Harm 3 & Incapacitated or dying \\
⊥tomrule
\end{tabular}
\end{coretable}

\subsection{Taking Harm (Procedure)}
\begin{enumerate}
    ℹtem Convert incoming Harm to Fatigue using armor (see table below).
    ℹtem Apply remaining Harm – if any Harm remains, increase Harm level and clear all existing Fatigue (the shock resets exhaustion).
    ℹtem Add converted Fatigue; if track fills, overflow triggers Harm increase.
\end{enumerate}

\subsection{Armor Conversion}
\begin{coretable}[Armor Conversion]
\begin{tabular}{>{\bfseries}p{0.2łinewidth} >{\bfseries}p{0.2łinewidth} >{\bfseries}p{0.2łinewidth} >{\bfseries}p{0.2łinewidth}}
⊤rule
\textbf{Armor} & \textbf{Harm 1} & \textbf{Harm 2} & \textbf{Harm 3} \\
∣rule
Light & Fatigue 1 & Harm 1 + Fatigue 1 & Harm 2 + Fatigue 1 \\
Medium & Fatigue 1 & Fatigue 1 & Harm 2 + Fatigue 1 \\
Heavy & Fatigue 1 & Fatigue 1 & Fatigue 2 \\
⊥tomrule
\end{tabular}
\end{coretable}

\subsection{Recovery}
\begin{itemize}
    ℹtem \textbf{Fatigue:} Short rest (1 hour) clears 2; full night's rest clears all.
    ℹtem \textbf{Harm:} Requires medical care and time (downtime). Magical healing exists but has costs (see Magic section).
\end{itemize}

%=====================================================================
\section{Player−Managed Resources}
%=====================================================================

Players track their own:
\begin{itemize}
    ℹtem øbligation\ (per Patron or Symbol) – Capacity = Spirit + Presence. Exceeding capacity causes 1 Fatigue per segment over.
    ℹtem \corruption\ timer (Cantors)
    ℹtem Leash timer (Summoners)
    ℹtem \asset\ states (Maintained / Neglected / Compromised)
    ℹtem \follower\ conditions (Exposure, Harm)
    ℹtem Fatigue and Harm
    ℹtem \boons
\end{itemize}

\textbf{Your duties:} Mark immediately when a rule says ``mark +X''. Declare thresholds (``My øbligation\ is full!''). Keep trackers visible to the table.

\section{Magic in One Page}
%=====================================================================

\subsection{Free Caster (Spellcraft)}
Requires \textbf{Spellcraft} talent (6 XP). No Patron.

\begin{itemize}
    ℹtem \textbf{Casting:} Describe spell using \textbf{TAGS}. \dv = 1 + number of tags. Dangerous tags (Teleport, Transform, Dominate) add +2 \dv.
    ℹtem \textbf{Roll:} Wits + Arcana.
    ℹtem \textbf{Backlash (on Miss or GM spending SB):} Choose one: Harm 2 (arcane), +2 Fatigue, \corruption\ +1, or catastrophic side effect (GM describes).
\end{itemize}

\subsection{Runekeeper (The Reliable Path)}
\textbf{Requirements:} Familiar (Thiasos, 2 XP) + Codex (4 XP) + Patron.

\begin{itemize}
    ℹtem \textbf{Invoking a Rite:} 1 action. Mark +1 øbligation. Effect works as written.
    ℹtem \textbf{Push It:} Once per scene, mark +1 øbligation\ to amplify (+1 die or +1 Effect).
    ℹtem \textbf{Obligation Capacity:} Spirit + Presence.
\end{itemize}

\textbf{For first magical characters, play a Runkeeper.}

%=====================================================================
\section{Timers and Pressure}
%=====================================================================

∈dex{Clock}
\timer s are visible trackers for threats, progress, and looming changes.

\textbf{Size:} 4 (short scene), 6 (standard), 8 (arc), 10 (epic).  
\textbf{Advance a \timer:} on a Miss, certain Partials, or when GM spends SB.

\subsection{Three−Timer Guideline (per scene)}
\begin{itemize}
    ℹtem \textbf{Immediate \timer} – what's about to go wrong (e.g., Collapsing Bridge [4])
    ℹtem \textbf{Situational \timer} – pressure shaping decisions (e.g., Crowd Turns Hostile [6])
    ℹtem \textbf{Long−term \timer} – echoes forward (e.g., Faction Retaliation [8])
\end{itemize}

%=====================================================================
\section{Bonds and Teamwork}
%=====================================================================

\subsection{Bonded Actions}
∈dex{Bond}
\begin{itemize}
    ℹtem Once per \bond\ per session: when you help a bonded ally with an intricate description, \textbf{both gain 1 \boons} after the action.
    ℹtem At \textbf{Tier II+ (41+ XP)}, you may transfer up to 2 \boons\ to a bonded ally as a free action (once per scene).
\end{itemize}

\subsection{Assist Limit}
Max +3 dice from all assists per roll. One \follower\ may assist (adds min(Cap, relevant skill) dice).

\subsection{Selling the Action (GM may grant +1 Position or bank a Boon)}
\begin{itemize}
    ℹtem \textbf{Counts:} stating a clear stake, invoking a \bond, revealing a fear, making a hard choice visible.
    ℹtem \textbf{Does not count:} piling on adjectives, passive sensory lists, repeated descriptions.
\end{itemize}

%=====================================================================
\section{Your Turn Checklist}
%=====================================================================

\begin{enumerate}
    ℹtem Declare intent and approach (Attribute + Skill)
    ℹtem GM tells you \position\ and \dv
    ℹtem Roll dice, count successes (6+) and 1s (SB for GM)
    ℹtem Apply outcome (GM will tell you Clean/Success with SB/Partial/Miss)
    ℹtem Mark \boons\ (1 on Partial, 2 on Miss) and any Fatigue/Harm
    ℹtem If you rolled 1s, GM may spend SB immediately
\end{enumerate}

\section{Quick Build & Advancement}
%=====================================================================

\subsection{Starting Characters (32 XP)}
\begin{itemize}
    ℹtem \textbf{Attributes:} each starts at 1. Raise to 2 (6 XP), 3 (9 XP), etc. (new rating $×$3 XP)
    ℹtem \textbf{Skills:} start at 0. Raise to 1 (2 XP), 2 (4 XP), etc. (new level $×$2 XP)
    ℹtem \textbf{Minor \asset\ (4 XP):} safehouse, contact, special tool
    ℹtem \textbf{Minor Talent (2–3 XP):} Keen Senses, Second Wind, etc.
    ℹtem \textbf{\bond s (up to 2, +2 XP each)}
    ℹtem \textbf{Complications (up to 2, +2 XP each)} – each unresolved adds +1 banked SB to early scenes.
\end{itemize}

\subsection{Advancement per Session (typical)}
\begin{itemize}
    ℹtem Attendance: +2 XP
    ℹtem Major objectives: +2–4 XP
    ℹtem Discoveries: +1–2 XP
    ℹtem Hard choices: +1–2 XP
    ℹtem Personal goals: +1–2 XP
\end{itemize}
\textbf{Game pace options:} Gritty (4–6 XP/session), Standard (6–10), Epic (10–14).

\section{Glossary}
%=====================================================================

\begin{itemize}
    ℹtem \textbf{\gameterm{Asset}} – Off−screen resource (safehouse, spy network, workshop). Costs XP. Can be activated with 1 \boons.
    ℹtem \textbf{\gameterm{Bond}} – Mutual connection between two PCs. Once per session, when you help your bonded ally with intricate description, both gain 1 \boons.
    ℹtem \textbf{\gameterm{Boons}} – Player resource earned on Partials and Misses. Spend to re−roll, improve Position, activate Assets.
    ℹtem \textbf{\gameterm{Cap}} – Follower capability rating (1–5). Adds min(Cap, skill) dice when assisting.
    ℹtem \textbf{\gameterm{Clock}} – Visible progress tracker (4/6/8/10 segments). Advances on Misses, Partials, or SB spends.
    ℹtem \textbf{\gameterm{Compromised}} – State of an Asset or Symbol: unusable until repaired.
    ℹtem \textbf{\gameterm{Corruption}} – Cantor resource. Timer fills $→$ gain Patron's burden/benefit, reset to Tier minimum.
    ℹtem \textbf{\gameterm{Crisis}} – Campaign timer (0–6). High = enemies stronger.
    ℹtem \textbf{\gameterm{Difficulty Value (DV)}} – Target number of successes. Default 3.
    ℹtem \textbf{\gameterm{Exposure}} – Follower stress track. Leads to Neglected/Compromised.
    ℹtem \textbf{\gameterm{Fatigue}} – Exhaustion track. Worsens Position; full track $→$ Harm+1, clear Fatigue.
    ℹtem \textbf{\gameterm{Follower}} – Named NPC ally. Costs Cap$²$ XP. Can assist or take independent actions.
    ℹtem \textbf{\gameterm{Harm}} – Injury track (1–3). Imposes die penalties.
    ℹtem \textbf{\gameterm{Leash}} – Summoner's control track. When full, spirit acts independently.
    ℹtem \textbf{\gameterm{Mandate}} – Campaign timer (0–6). High = party is respected.
    ℹtem \textbf{\gameterm{Miss}} – 0 successes. Player gains 2 \boons; GM gains SB and escalates.
    ℹtem \textbf{\gameterm{Neglected}} – State of Asset or Follower: –1 die until attention given.
    ℹtem \textbf{\gameterm{Obligation}} – Debt to Patron (or terrestrial patron). Capacity = Spirit + Presence. Over capacity $→$ Fatigue.
    ℹtem \textbf{\gameterm{Partial}} – $0 <$ successes $<$ DV. Player gains 1 \boons; GM gains SB.
    ℹtem \textbf{\gameterm{Patron}} – Cosmic entity granting Rites. Runkeeper = Obligation; Invoker = Symbol; Cantor = Corruption.
    ℹtem \textbf{\gameterm{Position}} – Risk level: Dominant (safe), Controlled (normal), Desperate (dangerous).
    ℹtem \textbf{\gameterm{Story Beat (SB)}} – GM resource. Earned on any 1. Spent to create complications.
    ℹtem \textbf{\gameterm{String}} – Narrative leverage (favor owed, secret known). Can be spent like currency.
    ℹtem \textbf{\gameterm{TAGS}} – Keywords for freeform magic (e.g., [FIRE], [AREA], [HEAL]). DV = 1 + number of tags.
    ℹtem \textbf{\gameterm{Terrestrial Patron}} – Mortal faction offering aid in exchange for Debt Timer .
    ℹtem \textbf{\gameterm{Thiasos}} – Familiar/token for Runekeepers (2 XP). Required for Patron's Gift.
    ℹtem \textbf{\gameterm{XP}} – Experience Points. Earned at session end. Spent on Attributes, Skills, Assets, Talents.
\end{itemize}

\section{Session Zero Checklist}
%=====================================================================

Use this checklist to start your campaign.
\begin{enumerate}
    ℹtem \textbf{Lines & Veils} – What content is off−limits or only implied?
    ℹtem \textbf{Tone} – Cinematic action? Grim survival? Political intrigue? Dark horror?
    ℹtem \textbf{Campaign length} – One−shot, short arc (3–6 sessions), or long campaign (12+)?
    ℹtem \textbf{Character hooks} – Each player shares one goal and one fear.
    ℹtem \textbf{Patron interest} – Which Patrons (if any) are already watching the party?
    ℹtem \textbf{Starting situation} – Where does the story begin?
    ℹtem \textbf{House rules} – Any optional subsystems the GM plans to use?
    ℹtem \textbf{Scheduling & attendance} – How often do you meet? What if a player misses?
    ℹtem \textbf{Player−GM contract} – ``We are co−creators, not adversaries.''
    ℹtem \textbf{Resource mindset} – ``Boons are not XP tokens. Spend them. Embrace failure.''
\end{enumerate}
